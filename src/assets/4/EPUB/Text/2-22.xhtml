٦٠٧

(بابُ محظوراتِ الإحرامِ)

أيِ: المحرَّماتِ بسببِهِ.

(وهيَ)؛ أيْ: محظوراتُهُ (تسعةٌ):

أحدُهَا: (حلْقُ الشّعرِ) مِنْ جميعِ بدنِهِ بلَا عذرٍ، يعنِي: إزالتُهُ بحلْقٍ، أوْ نتْفٍ، أوْ قلْعٍ؛ لقولِهِ تعالَى: {وَلَا تَحۡلِقُواْ رُءُوسَكُمۡ حَتَّىٰ يَبۡلُغَ ٱلۡهَدۡيُ مَحِلَّهُ} [البقرة:١٩٦].

(و) الثّانِي: (تقليمُ الأظفارِ) أوْ قصُّهُ مِنْ يدٍ أوْ رجلٍ بلَا عذرٍ؛

  • فإنْ خرجَ بعينِهِ شعرٌ،
  • أوْ كُسرَ ظُفرُهُ،
  • فأزالَهُمَا،
  • أوْ زالَا معَ غيرِهِمَا:
  • فلَا فديَةَ.
  • وإنْ حصلَ الأذَى بقرحٍ أوْ قملٍ ونحوِهِ فأزالَ شعرَهُ لذلكَ: فدَى.

ومَنْ حُلقَ رأسُهُ بإذنِهِ، أوْ سكتَ ولمْ ينهَهُ: فدَى.

ويُباحُ للمُحْرِمِ غَسلُ شعرِهِ بسدرٍ ونحوِهِ.

(فمنْ حلَقَ):

  • شعرةً واحدةً أوْ بعضَهَا: فعليهِ طعامُ مسكينٍ،
٦٠٨
  • وشعرتَيْنِ أوْ بعضَ شعرتَيْنِ: فطعامَا(١) مسكينٍ(٢)،
  • وثلاثَ شعراتٍ: فعليهِ دمٌ.

(أوْ قلّمَ):

  • ظُفرًا: فطعامُ مسكينٍ،
  • وظفرَيْنِ: فطعامَا مسكينٍ(٣)،
  • و(ثلاثةً: فعليهِ دمٌ)؛ أيْ: شاةٌ، أوْ إطعامُ ستَّةِ مساكِينَ، أوْ صيامُ ثلاثَةِ أيّامٍ،

وإنْ خلّلَ شعرَهُ وشكَّ فِي سقوطِ شيءٍ بهِ: استُحِبَّتْ.

ڭ ڭ ڭ

الثّالثُ: تغطيَةُ رأسِ الذّكَرِ، وأشارَ إليهِ بقولِهِ: (ومَنْ غطَّى رأسَهُ بمُلاصِقٍ فدَى)، سواءٌ كانَ:

  • معتادًا كعمامَةٍ، وبُرْنُسٍ،
  • أمْ لَا، كقِرطاسٍ وطينٍ ونُورَةٍ وحِنّاءٍ،
  • أوْ عصَبَهُ بِسَيرٍ(٤)،
  • أوِ استظلَّ فِي مِحْمَلٍ راكبًا أوْ لَا، ولوْ لمْ يلاصقْهُ،
  • ويحرُمُ ذلكَ بلَا عذرٍ،
٦٠٩
  • لَا إنْ حَمَلَ عليهِ، أوِ استظلَّ بخيمةٍ، أوْ شجرةٍ، أوْ بيتٍ.

ڭ ڭ ڭ

الرّابعُ: لُبسُهُ(٥) المَخِيطَ، وإلَيْهِ الإشارَةُ بقولِهِ: (وإنْ لبِسَ ذكرٌ مَخِيطًا: فدَى).

ولَا يعقِدُ عليهِ رداءً ولَا غيرَهُ، إلَّا إزارَهُ، ومِنْطَقَةً، وهِمْيانًا فِيهِمَا نفقةٌ معَ حاجةٍ لعقدٍ.

وإنْ لمْ يجدْ نعلَيْنِ لبِسَ خفّيْنَ،

أوْ لمْ يجدْ إزارًا لبِسَ سراويلَ إلَى أنْ يجدَ،

  • ولَا فديَةَ.

ڭ ڭ ڭ

الخامسُ: الطِّيبُ؛ وقدْ ذكرَهُ بقولِهِ:

  • (وإنْ طيّبَ) مُحْرِمٌ: (بدنَهُ، أوْ ثوبَهُ)، أوْ شيئًا مِنْهُمَا،
  • أوِ استعملَهُ فِي أكلٍ، أوْ شُربٍ(٦)،
  • (أوِ ادَّهَنَ(٧))، أوِ اكْتَحَلَ، أوِ اسْتَعَطَ (بمطيَّبٍ،
  • أوْ شمَّ) قصدًا (طِيبًا،
  • أوْ تبخّرَ بعودٍ ونحوِهِ)، أوْ شمَّهُ قصدًا ولوْ بَخورَ الكعبَةِ:
٦١٠
  • أثِمَ و(فدَى)،

ومِنَ الطِّيبِ: مِسكٌ، وكافورٌ، وعَنبرٌ، وزَعفرانٌ، ووَرْسٌ، وورْدٌ، وبَنَفْسَجٌ، ولَيْنَوْفَرٌ، وياسمينٌ، وبانٌ، وماءُ وردٍ.

  • وإنْ شمَّهَا بلَا قصدٍ،
  • أوْ مسَّ مَا لَا يَعْلَقُ؛ كقِطَعِ كافورٍ،
  • أوْ شمَّ فواكهَ،
  • أوْ عودًا،
  • أوْ شِيحًا،
  • أوْ رَيحانًا فارسيًّا،
  • أوْ نَمَّامًا(٨)،
  • أوِ ادّهنَ(٩) بدُهنٍ غيرِ مطيَّبٍ:
  • فلَا فديَةَ.

ڭ ڭ ڭ

السّادسُ: قتلُ صيدِ البرِّ واصطيادُهُ، وقدْ أشارَ إليهِ بقولِهِ: (وإنْ قتَلَ صيدًا:

  • مأكولًا،
  • برِّيًّا أصلًا)؛ كحمامٍ وبطٍّ ولوِ استأنسَ،
٦١١
  • بخلافِ إبِلٍ وبقرٍ أهليَّةٍ -ولوْ توحّشَتْ-،
  • (ولوْ تولّدَ مِنهُ)؛ أيْ: مِنَ الصّيدِ المذكورِ (ومِنْ غيرِهِ)؛ كالمتولِّدِ بينَ المأكولِ وغيرِهِ، أوْ بينَ الوحشيِّ وغيرِهِ؛ تغليبًا للحظرِ.

(أوْ تلِفَ) الصّيدُ المذكورُ (فِي يدِهِ):

  • بمباشرةٍ،
  • أوْ سببٍ؛ كإشارَةٍ، ودلالةٍ، وإعانةٍ ولوْ بمناولَةِ آلةٍ، أوْ جنايَةِ دابَّةٍ هوَ متصرِّفٌ فِيهَا:
  • (فعليهِ جزاؤُهُ)،
  • وإنْ دلَّ -ونحوُهُ- مُحْرِمٌ مُحْرِمًا: فالجزاءُ بينَهُمَا.

ويَحرُمُ علَى المُحْرِمِ:

  • أكلُهُ ممّا صادَهُ،
  • أوْ كانَ لهُ أثرٌ فِي صيدِهِ،
  • أوْ ذُبحَ أوْ صِيدَ لأجلِهِ،
  • ومَا حرُمَ عليهِ لنحوِ دلالةٍ،
  • أوْ صِيدَ لهُ:
  • لَا يَحرُمُ علَى مُحرِمٍ غيرِهِ.

ويُضمنُ بيضُ صَيدٍ ولبنُهُ إذَا حلَبَهُ بقيمتِهِ.

ولَا يملكُ المُحْرِمُ ابتداءً صيدًا بغيرِ إرثٍ.

٦١٢

وإنْ أحرمَ وبمِلْكِهِ صيدٌ:

  • لمْ يُزَلْ،
  • ولَا يدُهُ الحُكميَّةُ،
  • بلْ تُزالُ يدُهُ المشاهَدَةُ بإرسالِهِ.

(ولَا يَحرُمُ) بإحرامٍ أوْ حرمٍ:

  • (حيوانٌ إنسيٌّ)؛ كالدِّجاجِ وبهيمَةِ الأنعامِ؛
  • لأنَّهُ ليسَ بصيدٍ،
  • وقدْ كانَ النَّبِيُّ چ يذبحُ البُدْنَ فِي إحرامِهِ بالحرمِ(١٠)،
  • (ولَا) يحرُمُ (صيدُ البحرِ) إنْ لمْ يكنْ بالحرمِ؛ لقولِهِ تعالَى: {أُحِلَّ لَكُمۡ صَيۡدُ ٱلۡبَحۡرِ وَطَعَامُهُ} [المائدة:٩٦]،
  • وطيرُ الماءِ: برّيٌّ.
  • (ولَا) يَحرُمُ بحرمٍ ولَا إحرامٍ (قتلُ محرَّمِ الأكلِ)؛ كالأسدِ والنّمرِ والكلبِ،
  • إلَّا المتولِّدَ؛ كمَا تقدَّمَ(١١).
  • (ولَا) يحرُمُ قتلُ الصّيدِ (الصّائلِ)؛ دفعًا عنْ نفسِهِ أوْ مالِهِ، سواءٌ
٦١٣

خشِيَ التّلفَ، أوِ الضّررَ بجَرحِهِ(١٢)؛ لأنَّهُ التحقَ بالمؤذِياتِ؛ فصارَ كالكلبِ العقورِ.

ويُسنُّ مطلقًا: قتلُ كلِّ مؤذٍ، غيرِ آدمِيٍّ.

ويحرُمُ بإحرامٍ قتلُ قملٍ وصِئبانِهِ، ولوْ برميِهِ، ولَا جزاءَ فيهِ.

  • لَا بَراغيثَ، وقُرَادٍ، ونحوِهِمَا.

ويُضمنُ جرادٌ بقيمتِهِ.

ولمُحْرِمٍ احتاجَ لفعلِ محظورٍ فِعْلُهُ ويفدِي،

وكذَا لوِ اضْطُرَّ إلَى أكلِ صيدٍ فلهُ ذبحُهُ وأكلُهُ، كمَنْ بالحرمِ،

  • ولَا يُباحُ إلَّا لمَنْ لهُ أكلُ الميْتَةِ.

ڭ ڭ ڭ

السّابعُ - عَقدُ النِّكاحِ؛ وقدْ ذكرَهُ بقولِهِ: (ويحرُمُ عَقدُ نكاحٍ)، فلوْ تزوّجَ المُحْرِمُ، أوْ زوّجَ مُحْرِمةً، أوْ كانَ وليًّا، أوْ وكيلًا فِي النِّكاحِ: حَرُمَ، (ولَا يصحُّ)؛ لمَا روَى مسلمٌ عنْ عثمانَ ڔ مرفوعًا: «لَا يَنكِحُ المُحْرِمُ ولَا يُنكِحُ»(١٣)،

(ولَا فديَةَ) فِي عَقدِ النِّكاحِ؛ كشراءِ الصّيدِ.

ولَا فرقَ بينَ الإحرامِ الصّحيحِ والفاسدِ.

٦١٤

ويُكرهُ للمُحْرِمِ:

  • أنْ يخطِبَ امرأةً،
  • كخُطبَةِ عَقدِهِ،
  • وحضورُهُ،
  • وشهادتُهُ فيهِ،

(وتصحُّ الرّجعَةُ)؛ أيْ: لوْ راجعَ المُحْرِمُ امرأتَهُ صحّتْ بلَا كراهَةٍ؛ لأنَّهُ إمساكٌ،

  • وكذَا شراءُ أمةٍ للوطءِ.

ڭ ڭ ڭ

الثّامنُ: الوطءُ، وإلَيْهِ الإشارَةُ بقولِهِ: (وإنْ جامعَ) المُحْرِمُ بأنْ غيّبَ الحشَفَةَ فِي قُبُلٍ أوْ دُبُرٍ مِنْ آدمِيٍّ أوْ غيرِهِ حَرُمَ(١٤)؛ لقولِهِ تعالَى: {فَمَن فَرَضَ فِيهِنَّ ٱلۡحَجَّ فَلَا رَفَثَ} [البقرة:١٩٧] قالَ ابنُ عبّاسٍ ژ : «هوَ الجماعُ»(١٥)،

وإنْ كانَ الوطءُ (قبلَ التّحلُّلِ الأوّلِ:

  • فسدَ نُسكُهُمَا)، ولوْ بعدَ الوقوفِ بعرفَةَ،
  • ولَا فرقَ بينَ العامدِ والسّاهِي؛ لقضاءِ بعضِ الصّحابَةِ بفسادِ الحجِّ ولمْ يستفصلْ،
  • (ويَمضيانِ فيهِ)؛ أيْ: يجبُ علَى الواطئِ والموطوءَةِ المُضِيُّ فِي
٦١٥
  • النُّسكِ الفاسدِ، ولَا يخرجانِ مِنهُ بالوطءِ؛ رُويَ عنْ عمرَ(١٦)، وعلِيٍّ، وأبِي هريرةَ(١٧)، وابنِ عبّاسٍ ښ (١٨)، فحكمُهُ كالإحرامِ الصّحيحِ؛ لقولِهِ تعالَى: {وَأَتِمُّواْ ٱلۡحَجَّ وَٱلۡعُمۡرَةَ لِلَّهِ} [البقرة:١٩٦]،
  • (ويقضيانِهِ) وجوبًا (ثانِيَ عامٍ)؛ رُوِيَ عنِ ابنِ عبّاسٍ وابنِ عمرَ وابنِ عمرٍو ښ (١٩).
٦١٦
  • وغيرُ المكلّفِ يقضِي بعدَ تكليفِهِ وحجَّةِ الإسلامِ فورًا، مِنْ حيثُ أحرمَ أوّلًا إنْ كانَ قبلَ ميقاتٍ، وإلَّا فمِنْهُ.

وسُنَّ تفرُّقُهُمَا فِي قضاءٍ مِنْ موضعِ وطءٍ إلَى أنْ يَحِلَّا.

والوطءُ بعدَ التّحلُّلِ الأوّلِ:

  • لَا يُفسدُ النُّسكَ،
  • وعليهِ شاةٌ.

ولَا فديَةَ علَى مُكرَهَةٍ،

  • ونفقَةُ حجَّةِ قضائِهَا عليهِ؛ لأنَّهُ المفسِدُ لنُسكِهَا.

ڭ ڭ ڭ

التّاسعُ: المباشرَةُ دونَ الفرجِ، وذكرَهَا بقولِهِ: (وتحرُمُ المباشرَةُ)؛ أيْ: مباشرَةُ الرّجلِ المرأَةَ، (فإنْ فعلَ)؛ أيْ: باشرَهَا (فأنزلَ:

  • لمْ يفسُدْ حجُّهُ) كمَا لوْ لمْ يُنْزِلْ، ولَا يصحُّ قياسُهَا علَى الوطءِ؛ لأنَّهُ يجبُ بهِ الحدُّ دونَهَا،
  • (وعليهِ بدنةٌ) إنْ أنزلَ بمباشرةٍ، أوْ قُبلةٍ، أوْ تَكرارِ نظرٍ، أوْ لمسٍ لشهوةٍ، أوْ أمنَى باستمناءٍ؛ قياسًا علَى بدنَةِ الوطءِ.

وإنْ لمْ يُنْزِلْ فشاةٌ كفديَةِ أذًى.

وخطأٌ فِي ذلكَ كعمدٍ.

وامرأةٌ معَ شهوةٍ كرجلٍ فِي ذلكَ.

(لكنْ يُحرِمُ) بعدَ أنْ يَخرُجَ (مِنَ الحِلِّ)؛ ليجمعَ فِي إحرامِهِ بينَ الحِلِّ

٦١٧

والحرمِ؛ (لطوافِ الفرضِ)؛ أيْ: ليطوفَ طوافَ الزِّيارَةِ مُحرِمًا،

  • وظاهرُ كلامِهِ أنَّ هذَا فِي المباشِرِ(٢٠) دونَ الفرجِ إذَا أنزلَ؛ وهوَ غيرُ متّجهٍ؛ لأنَّهُ لمْ يُفسدْ إحرامَهُ حتَّى يحتاجَ لتجديدِهِ، فالمباشرَةُ كسائرِ المحرَّماتِ غيرِ الوطءِ، هذَا مقتضَى كلامِهِ فِي الإقناعِ، كالمنتهَى، والمقنعِ، والتّنقيحِ، والإنصافِ، والمبدعِ(٢١) وغيرِهَا، وإنَّمَا ذكرُوا هذَا الحكمَ فيمَنْ وطِئَ بعدَ التّحلُّلِ الأوّلِ.
  • إلَّا أنْ يكونَ علَى وجهِ الاحتياطِ؛ مراعاةً للقولِ بالإفسادِ.

ڭ ڭ ڭ

(وإحرامُ المرأَةِ) فيمَا تقدَّمَ (كالرّجلِ، إلَّا:

  • فِي اللِّباسِ)؛ أيْ: لباسِ المَخِيطِ؛ فلَا يحرُمُ عَلَيْهَا،
  • ولَا تغطيَةُ الرّأسِ.

(وتجتنبُ البُرقَعَ والقُفّازَيْنِ)؛ لقولِهِ چ : «لَا تنتقبِ المرأَةُ ولَا تلبَسِ القفّازَيْنِ»، رواهُ البخارِيُّ وغيرُهُ(٢٢)،

  • والقُفّازانِ: شيءٌ يُعملُ لليدَيْنِ، يُدخَلانِ فيهِ، يسترُهُمَا مِنَ الحَرِّ، كمَا يُعملُ للبُزَاةِ،
  • ويفدِي الرّجلُ والمرأَةُ بلُبسِهِمَا.
٦١٨

(و) تجتنبُ أيضًا (تغطيَةَ وجهِهَا)؛ لقولِهِ چ : «إحرامُ الرّجلِ فِي رأسِهِ، وإحرامُ المرأَةِ فِي وجهِهَا»(٢٣)

  • فتضعُ الثّوبَ فوقَ رأسِهَا وتسدلُهُ علَى وجهِهَا؛ لمرورِ الرِّجالِ قريبًا مِنْهَا.

(ويُباحُ لهَا التّحلِّي) بالخَلْخَالِ، والسِّوارِ، والدُّمْلُجِ، ونحوِهَا.

ويُسنُّ لهَا خِضابٌ عندَ إحرامٍ، وكُرهَ بعدَهُ.

وكُرهَ لهمَا اكتحالٌ بإثمِدٍ؛ لزينةٍ.

ولَهُمَا:

  • لُبسُ مُعَصْفَرٍ، وكُحليٍّ،
  • وقطعُ رائحةٍ كريهةٍ بغيرِ طِيبٍ،
  • واتِّجارٌ وعملُ صنعةٍ مَا لمْ يَشغلَا عنْ واجبٍ أوْ مستحَبٍّ.

ولهُ لُبسُ خاتمٍ.

ويجتنبانِ الرّفَثَ، والفسوقَ، والجدالَ.

وتُسنُّ قلَّةُ الكلامِ إلَّا فيمَا ينفعُ.

(١) في (ز): «فطعام».

(٢) في (د، ز، م): «مسكينين».

(٣) في (د، ز، م): «فطعام مسكينين»، وفي (ض، ل): «طعاما».

(٤) في (م): «ولو بسير» وهو الموافق لما في المنتهى (٢/٩٩).

(٥) في (ز): «لبس».

(٦) «أو استعمله في أكلٍ، أو شربٍ» ليست في (س)، وأشار في هامشها أنها في نسخة أخرى.

(٧) في (الأصل، س): «أو الدهن».

(٨) في (د): بالتشديد، وفي (ب، س، ض، ع، ل، م): بالتخفيف.

(٩) في (ز، س): «أو الدهن»، وفي (م، ل): «دهن».

(١٠) ورد من حديث جابر في صفة حج النبي چ عند مسلم (١٢١٨): (ثم انصرف إلى المنحر فنحر ثلاثًا وستين بيده).

(١١) أي عند قوله: «(ولوْ تولّدَ مِنهُ)؛ أيْ: مِنَ الصّيدِ المذكورِ ...» في (ص٦١١).

(١٢) في (د، ز، ع، م): «بجرحه أو لا».

(١٣) أخرجه أحمد (١/٦٤)، ومسلم (١٤٠٩).

(١٤) «حرم» ليست في (الأصل، س) وإنما ألحقت فيهما في الهامش من دون تصحيح، وهي مثبتة في بقية النسخ.

(١٥) أخرجه سعيد بن منصور في التفسير (٣٣٩)، وابن أبي شيبة (٤/١٥٧)، والبيهقي (٥/٦٧).

(١٦) أخرجه ابن أبي شيبة (الجزء المفرد(ص١٣٥)، وسعيد بن منصور (انظر: تنقيح التحقيق لابن عبد الهادي ٣/٤٧٧)، والبيهقي (٥/١٦٧) من طريقين عن عمر ڔ ، لفظ أحدهما: (يقضيان حجهما، وعليهما الحج من قابل...).

قال ابن عبد الهادي وابن الملقن في البدر المنير (٦/٣٨٥): (منقطع).

(١٧) ذكره مالك في الموطأ (١١٢٦) أنه بلغه أن عمر بن الخطاب وعلي بن أبي طالب وأبا هريرة ښ سئلوا: عن رجل أصاب أهله وهو محرم بالحج؟ فقالوا: (ينفذان لوجههما حتى يقضيا حجهما، ثم عليهما حج قابل والهدي).

(١٨) أخرجه إسماعيل بن جعفر في حديثه (١١٤)، ومن طريقه البيهقي (٥/١٦٧ - ١٦٨)، أنه قال في رجل وقع على امرأته وهو محرم: (اقضيا نسكَكُما وارجعا إلى بلدكما، فإذا كان عام قابل ...) وذكر تتمته.

(١٩) «ابن عمرو» ليس في (ب، د، ز، ع، ل، م).

والرواية عنهم أخرجها ابن أبي شيبة في الجزء المفرد (ص١٣٦)، والدارقطني (٣٠٠٠)، وعنه الحاكم (٢/٦٥)، وعنه البيهقي (٥/١٦٧) عن عمرو بن شعيب عن أبيه أن رجلًا أتى عبدالله بن عمرو يسأله عن مُحْرم وقع بامرأة، فأشار إلى عبدالله بن عمر فقال: (اذهب إلى ذلك فَسَلْهُ) فذهب إليه فقال: (بَطَلَ حجُّكَ) فقال الرجل: فما أصنع؟ قال: (اخرج مع الناس واصنع ما يصنعون، فإذا أدركت قابلًا فحُجَّ وأَهْدِ) ثم ذكر تتمة القصة وموافقة ابن عباس وابن عمرو لابن عمر ښ .

قال الحاكم: (هذا حديثُ ثقاتٍ، رواته حفاظٌ)، وقال البيهقي: (هذا إسناد صحيح).

(٢٠) في (ب، د، ز، ع، م): «المباشرة».

(٢١) انظر: الإقناع (١/٥٨٧)، المنتهى (٢/١١٣)، المقنع (٨/٣٥١)، التنقيح (ص١٨٢)، الإنصاف (٨/٣٥٣)، المبدع (٣/١٥٣).

(٢٢) أخرجه أحمد (٢/١١٩)، والبخاري (١٨٣٨) من حديث عبد الله بن عمر ژ .

(٢٣) أخرجه الطبراني الأوسط (٦١٢٢)، والدارقطني (٢٧٦٠)، والبيهقي (٥/٤٧) من حديث ابن عمر ژ به مرفوعًا. ورُويَ موقوفًا على ابن عمر ژ .

ورجَّح وقفه: العقيلي في الضعفاء (ترجمة أيوب بن محمد ١/٣٤١ - ٣٤٢) والطبراني، وابن عدي في الكامل (٢/٢١٥).
