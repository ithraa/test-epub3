٤٧

(بابُ السّواكِ وسُننِ الوُضوءِ)

ومَا أُلحقَ بذلكَ منَ: الادِّهانِ، والاكتحالِ، والاختتانِ، والاستحدادِ؛ ونحوِهَا.

السّواكُ والمسواكُ:

  • اسمٌ للعودِ الَّذِي يُستاكُ بهِ،
  • ويطلقُ السّواكُ علَى الفعلِ؛ أيْ: دلْكُ الفمِ بالعودِ لإزالةِ نحوِ تغيّرٍ؛ كالتّسوُّكِ.

(التّسوُّكُ بعودٍ

  • ليّنٍ) سواءٌ كانَ رطبًا، أوْ يابسًا مندًّى، منْ أراكٍ، أوْ زيتونٍ، أوْ عرجونٍ، أوْ غيرِهَا،
  • (مُنْقٍ) للفمِ
  • (غيرِ مضرٍّ)، احترازٌ(١) عنِ الرّمّانِ والآسِ، وكلِّ مَا لَهُ رائحةٌ طيّبةٌ
  • (لَا يتفتّتُ) ولَا يجرحُ،
  • ويُكرهُ بعودٍ: يجرحُ أوْ يضرُّ أوْ يتفتّتُ.

و(لَا) يصيبُ السُّنَّةَ مَنِ استاكَ (بإصبَعِهِ وخرقةٍ(٢))؛ ونحوِهَا؛ لأنَّ الشّرعَ لمْ يَرِدْ بهِ، ولَا يحصلُ بهِ الإنقاءُ كالعودِ.

٤٨

(مسنونٌ كلَّ وقتٍ): خبرُ قولِهِ: التّسوُّكُ؛ أيْ: يُسنُّ كلَّ وقتٍ؛ لحديثِ: «السّواكُ مطهرةٌ للفمِ مرضاةٌ للرّبِّ»، رواهُ الشّافعِيُّ، وأحمدُ وغيرُهمَا(٣)،

  • (لغيرِ صائمٍ بعدَ الزّوالِ) فيُكرَهُ؛ فرضًا كانَ الصّومُ أوْ نفلًا، وقبلَ الزّوالِ: يُستحبُّ لهُ بيابسٍ، ويُباحُ برطبٍ؛ لحديثِ: «إذَا صُمتمْ فاستاكُوا بالغداةِ، ولَا تستاكُوا بالعشِيِّ»، أخرجَهُ البيهقِيُّ عنْ عليٍّ ڔ (٤).

(متأكِّدٌ) -خبرٌ ثانٍ للتّسوُّكِ-:

  • (عندَ: صلاةٍ)؛ فرضًا كانتْ أوْ نفلًا،
  • (و) عندَ (انتباهٍ) منْ نومِ ليلٍ أوْ نهارٍ،
  • (و) عندَ (تغيّرِ) رائحةِ (فمٍ) بمأكولٍ أوْ غيرِهِ،
  • وعندَ: وُضوءٍ،
  • وقراءةٍ.
  • زادَ الزّركشِيُّ والمصنِّفُ فِي الإقناعِ:
  • «ودخولِ: منزلٍ ومسجدٍ،
  • وإطالةِ سكوتٍ،
٤٩
  • وخلوِّ المعدةِ مِنَ الطّعامِ،
  • واصفرارِ الأسنانِ»(٥).

(ويستاكُ:

  • عرضًا) -استحبابًا- بالنّسبةِ إلَى الأسنانِ،
  • بيدِهِ اليُسرَى،
  • علَى أسنانِهِ ولِثَتِهِ ولسانِهِ.

ويغسلُ السّواكَ.

ولَا بأسَ أنْ يستاكَ بهِ اثنانِ فأكثرُ.

قالَ فِي الرّعايةِ: ويقولُ إذَا استاكَ: «اللهُمَّ طهّرْ قلبِي، ومَحِّصْ ذنوبِي»(٦).

قالَ بعضُ الشّافعيّةِ: «وينوِي بهِ الإتيانَ بالسُّنّةِ».

(مبتدئًا بجانبِ فمِهِ الأيمنِ)؛

  • فتُسنُّ البداءةُ بالأيمنِ فِي: سواكٍ، وطُهُورٍ، وشأنِهِ كلِّهِ، غيرَ مَا يُستقذَرُ.

ڭ ڭ ڭ

(ويدّهنُ) استحبابًا (غِبًّا): يومًا يدّهنُ ويومًا لَا يدّهنُ؛ لأنَّهُ چ نهَى عنِ التّرجّلِ إلَّا غبًّا، رواهُ النّسائِيُّ والتّرمذِيُّ وصحَّحَهُ(٧)، والتّرجّلُ:

٥٠

تسريحُ الشّعرِ، ودهنُهُ.

(ويكتحلُ)

  • فِي كلِّ عينٍ
  • (وترًا) ثلاثًا،
  • بالإثمِدِ المطيَّبِ،
  • كلَّ ليلةٍ، قبلَ أنْ ينامَ؛
  • لفعلِهِ چ ؛ رواهُ أحمدُ وغيرُهُ، عنِ ابنِ عبّاسٍ(٨).

ويُسنُّ:

  • نظرٌ(٩) فِي مرآةٍ،
٥١
  • وتطيُّبٌ(١٠).

(وتجبُ التّسميةُ فِي الوُضوءِ معَ الذُّكرِ)؛ أيْ: أنْ يقولَ: بسمِ اللهِ، لَا يقومُ غيرُهَا مقامَهَا؛ لخبرِ أبِي هريرةَ ڔ مرفوعًا: «لَا صلاةَ لمَنْ لَا وُضوءَ لهُ، ولَا وُضوءَ لمَنْ لمْ يذكرِ اسمَ اللهِ عليهِ»، رواهُ أحمدُ وغيرُهُ(١١).

  • وتسقطُ معَ السّهوِ،

وكذَا: غُسلٌ وتيمّمٌ.

(ويجبُ الخِتانُ): عندَ البلوغِ (مَا لمْ يَخَفْ علَى نفسِهِ)، ذكرًا كانَ، أوْ خنثَى، أوْ أنثَى;

  • فالذّكرُ: بأخذِ جلدةِ الحشفةِ،
  • والأنثَى: بأخذِ جلدةٍ فوقَ محَلِّ الإيلاجِ تشبهُ عُرفَ الدّيكِ، ويُستحبُّ أنْ لَا تؤخذَ كلُّهَا،
  • والخنثَى: بأخذِهمَا.

وفعلُهُ زمنَ صغرٍ: أفضلُ.

وكُرهَ:

  • فِي سابعِ يومٍ،
  • ومنَ الولادةِ إليهِ.
٥٢

(ويُكرهُ القزَعُ): وهوَ حلقُ بعضِ الرّأسِ وتركُ بعضٍ.

وكذَا: حلقُ القفَا لغيرِ حجامةٍ؛ ونحوِهَا.

ويُسنُّ إبقاءُ شعرِ الرّأسِ، قالَ أحمدُ: «هوَ سُنَّةٌ، لوْ نقوَى عليهِ اتّخذناهُ ولكنْ لهُ كلفةٌ ومؤنةٌ»(١٢).

  • ويسرِّحُهُ،
  • ويَفْرُقُهُ،
  • ويكونُ إلَى أذنيْهِ وينتهِي إلَى منكبيْهِ؛
  • كشعرِهِ چ (١٣)،
  • ولَا بأسَ: بزيادةٍ، وجعلِهِ ذؤابةً.

ويُعفِي لحيتَهُ، ويحرُمُ حلقُهَا، ذكرَهُ الشّيخُ تقِيُّ الدّينِ (١٤).

٥٣

ولَا يُكرهُ أخذُ: مَا زادَ علَى القبضةِ، ومَا(١٥) تحتَ حلقِهِ.

ويحفُّ شاربَهُ، وهوَ أوْلَى منْ قصِّهِ.

ويقلّمُ أظفارَهُ مخالفًا.

وينتفُ إبْطَهُ،

ويحلقُ عانتَهُ،

  • ولهُ إزالتُهُ بمَا شاءَ، والتّنويرُ فعلَهُ أحمدُ فِي العورةِ وغيرِهَا(١٦)،
  • ويدفنُ مَا يزيلُهُ منْ شعرِهِ، وظفرِهِ; ونحوِهِ.

ويفعلُهُ:

  • كلَّ أسبوعٍ،
  • يومَ الجمعةِ،
  • قبلَ الزّوالِ،

ولَا يتركُهُ فوقَ أربعينَ يومًا،

  • وأمَا الشّاربُ: ففِي كلِّ جمعةٍ.

ڭ ڭ ڭ

(ومنْ سُننِ الوُضوءِ):

  • وهيَ: جمعُ سُنَّةٍ،
٥٤
  • وهيَ فِي اللُّغةِ: الطّريقةُ.
  • وفِي الاصطلاحِ:
  • مَا يُثابُ علَى فعلِهِ، ولَا يُعاقَبُ علَى تركِهِ،
  • وتُطلَقُ أيضًا علَى: أقوالِهِ، وأفعالِهِ، وتقريراتِهِ چ .
  • وسُمِّيَ غَسلُ الأعضاءِ علَى الوجهِ المخصوصِ: وُضوءًا؛ لتنظيفِهِ المتوضِّئَ وتحسينِهِ.

(السّواكُ)، وتقدَّمَ أنَّهُ يتأكّدُ فيهِ(١٧)،

  • ومحَلُّهُ: عندَ المضمضةِ.

(وغَسلُ الكفَّيْنِ ثلاثًا) فِي أوّلِ الوُضوءِ، ولوْ تحقّقَ طهارتَهمَا،

  • (ويجبُ): غَسلُهمَا ثلاثًا بنيَّةٍ وتسميةٍ، (منْ نومِ ليلٍ ناقضٍ لوُضوءٍ)؛ لمَا تقدَّمَ فِي أقسامِ الماءِ(١٨).
  • ويسقطُ غَسلُهمَا والتّسميةُ: سهوًا.
  • وغَسلُهمَا: لمعنًى فيهمَا؛ فلوِ استعملَ الماءَ، ولمْ يُدخلْ يدَهُ فِي الإناءِ: لمْ يصحَّ وُضوؤُهُ، وفسدَ الماءُ.

(و) منْ سُننِ الوُضوءِ: (البداءةُ) قبلَ غَسلِ الوجهِ (بمضمضةٍ ثمَّ

٥٥

استنشاقٍ): ثلاثًا ثلاثًا، بيمينِهِ، واستنثارٌ(١٩) بيسارِهِ.

(و) منْ سُننِهِ: (مبالغةٌ(٢٠) فيهمَا)؛ أيْ: فِي المضمضةِ والاستنشاقِ

  • (لغيرِ صائمٍ): فتُكرَهُ.

والمبالغةُ:

  • فِي مضمضةٍ:
  • إدارةُ الماءِ بجميعِ فمِهِ،
  • وفِي استنشاقٍ:
  • جذبُهُ بنفَسٍ إلَى أقصَى أنفٍ،
  • وفِي بقيَّةِ الأعضاءِ:
  • دلْكُ مَا ينبُو عنهُ الماءُ، للصّائمِ وغيرِهِ.

(و) منْ سُننِهِ: (تخليلُ اللّحيةِ الكثيفةِ): -بالثّاءِ المثلّثةِ- وهيَ الَّتِي تسترُ البشَرةَ؛

  • فيأخذُ كفًّا منْ ماءٍ يضعُهُ: منْ تحتِهَا بأصابعِهِ مشتبكةً، أوْ منْ جانبيْهَا ويعرُكُهَا.

وكذَا: عنفقةٌ، وباقِي شعورِ الوجهِ.

(و) منْ سُننِهِ: تخليلُ (الأصابعِ)؛ أيْ: أصابعِ اليدَيْنِ والرّجلَيْنِ، قالَ فِي الشّرحِ: «وهوَ فِي الرّجلَيْنِ آكدُ»(٢١)،

٥٦
  • ويخلِّلُ:
  • أصابعَ رجليْهِ: بخنصِرِ يدِهِ اليُسرَى، منْ باطنِ رجلِهِ(٢٢)، منْ خنصِرِهَا إلَى إبهامِهَا، وفِي اليُسرَى بالعكسِ.
  • وأصابعَ يديْهِ: إحداهُمَا بالأخرَى.
  • فإنْ كانتْ أوْ بعضُهَا ملتصقةً: سقطَ.

(و) منْ سُننِهِ: (التّيامُنُ) بلَا خلافٍ.

(وأخذُ ماءٍ جديدٍ للأذنَيْنِ) بعدَ مسحِ رأسِهِ.

ومجاوزةُ محَلِّ فرضٍ.

(و) منْ سُننِهِ: (الغَسلةُ الثّانيةُ والثّالثةُ)،

  • وتُكرَهُ الزّيادةُ عليهَا،
  • ويعملُ فِي عددِ الغَسلاتِ بالأقلِّ،

ويجوزُ الاقتصارُ علَى الغَسلةِ الواحدةِ، والثّنتانِ أفضلُ مِنهَا، والثّلاثةُ(٢٣) أفضلُ منهُمَا.

  • ولوْ غسلَ بعضَ أعضاءِ الوُضوءِ أكثرَ منْ بعضٍ: لمْ يُكرهْ.

ولَا يُسنُّ:

  • مسحُ العنقِ،
  • ولَا الكلامُ علَى الوُضوءِ.

(١) في (د، ب): «احترازًا».

(٢) في (د): «بإصبع أو خرقة»، وفي (ب، ع): «بإصبع وخرقة».

(٣) أخرجه الشافعي (انظر: مسند الشافعي بترتيب السندي ١/٣٠)، وأحمد (٦/٤٧)، والنسائي (١/١٠).

وصححه ابن خزيمة (١٣٥) وابن حبان (١٠٦٧)، وقال ابن دقيق العيد في الإمام (١/٣٣٣): (الحديث جيد).

(٤) أخرجه البيهقي (٤/٢٧٤) عن علي ڔ موقوفًا، والدارقطني في السنن (٢٣٧٣) وضعفه.

(٥) شرح الزركشي (١/١٦٦)، الإقناع (١/٣١).

(٦) نقله في: المبدع، للبرهان ابن مفلح (١/٨٢).

(٧) أخرجه النسائي (٨/١٣٢)، والترمذي (١٧٥٦) وصححه، وأحمد (٤/٨٦) وأبو داود (٤١٥٩) من حديث هشام بن حسان عن الحسن عن عبد الله بن مغفل ڔ .

صححه ابن حبان (٥٤٨٤)، واختلف في رفعه ووقفه، وفي وصله وإرساله؛ فرواه عن قتادة عن الحسن مرسلًا، ورواه يونس بن عبيد عن الحسن وابن سيرين موقوفًا عليهما، قال المنذري في مختصر سنن أبي داود (٦/٨٣): (في إسناده اضطراب)، وأعلّه الذهبي في السير (٦/٣٦٣)، وتكلم ابن عليَّة وابن المديني وأبو داود في رواية هشام عن الحسن.

(٨) أخرجه أحمد (١/٣٥٤) وابن ماجه (٣٤٩٩) والترمذي (١٧٥٧) وحسنه، عن ابن عباس مرفوعًا قال: (كانت لرسول الله چ مكحلة، يكتحل بها عند النوم ثلاثًا في كل عين).

نقل الترمذي في العلل الكبير (٥٢٨) عن البخاري قوله: (هو حديث محفوظ)، وقال البيهقي في السنن الكبير (٤/٢٦١): (أصح ما روي في اكتحال النبي چ ).

وأعله يحيى بن سعيد، وعلي بن المديني، والعقيلي (انظر: الضعفاء للعقيلي ٤/١٠٢)، وأبو حاتم الرازي (انظر: العلل لابنه س٢٤٦٣) وغيرهم.

(٩) في (د، ز، ل، ع): «نظره».

(١٠) في (د، م) حاشية: «ويتفطن إلى نعم الله تعالى ويقول: اللهم كما حسَّنت خَلقي فحسن خُلقي وحرم وجهي على النار؛ لحديث أبي هريرة من رواية ابن مردويه» أثبتها (د) في الصلب ثم رمز إلى أنها حاشية.

(١١) أخرجه أحمد (٢/٤١٨)، وأبو داود (١٠١)، وابن ماجه (٣٩٩) عنه بلفظه.

ضعفه أحمد (انظر: مسائل ابنه صالح ٣٠٢)، والبخاري في التاريخ الكبير (٤/٧٦ رقم ٢٠٠٦).

وقال أحمد (انظر: الترمذي عقب ح٢٥): (لا أعلم في هذا الباب حديثا له إسناد جيِّد).

(١٢) الترجُّل للخلال (٢٩ - ٣٢) (ص١١٨)، وانظر: الفروع للشمس ابن مفلح (١/ ١٥١).

(١٣) سُئل أحمد عن: صفة شعر النبي چ ؟ فقال: (في بعض الحديث: أنه كان إلى شحمة أذنيه [المسند (٤/٢٨١) والبخاري (٣٥٥١) ومسلم (٢٣٣٧) عن البراء بن عازب] وفي بعض الحديث: إلى منكبيه [المسند (٣/٢٤٥) والبخاري (٥٩٠٣) ومسلم (٢٣٣٨) عن أنس بن مالك] وفي بعض الحديث: أنه فرق [المسند (١/٢٦١) والبخاري (٣٥٥٨) ومسلم (٢٣٣٦) عن ابن عباس] وإنما يكون الفرق إذا كان له شعر). الترجل (٢٢) من الجامع للخلال.

وعن عائشة أنها كانت ترجل رأس رسول الله چ (المسند ٦/٥٠ والبخاري ٢٩٥ ومسلم ٢٩٧).

(١٤) انظر: الأخبار العلمية (اختيارات شيخ الإسلام)، لابن اللحام (ص١٩).

(١٥) في (د): «ولا ما».

(١٦) نقله في: الآداب الشرعية للشمس ابن مفلح (٣/ ٤٩٩) من رواية أبي عبدالله النيسابوري.

(١٧) أي عند قوله: «(متأكد).. (عند).. وضوء» في (ص٤٨).

(١٨) أي عند قوله: «(أو غمس فيه)؛ أيْ: فِي الماءِ القليلِ...؛ لحديث: «إذا استيقظ أحدكم...» في (ص٢٣).

(١٩) في (ب، د، ز، ع، م): «واستنثاره».

(٢٠) في (د، ب): «المبالغة».

(٢١) الشرح الكبير (١/٢٨٦).

(٢٢) في (د، ز): «رجله اليمنى»، وفي (ع، ل) أُلحقت في الهامش من غير تصحيح.

(٢٣) في (د): «والثلاث»، وفي (ب): «والثالثة».
