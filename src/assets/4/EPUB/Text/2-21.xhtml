٥٩٧

(بابُ الإحرامِ)

لغةً: نيَّةُ الدُّخولِ فِي التّحريمِ؛ لأنَّهُ يحرِّمُ علَى نفسِهِ بنيّتِهِ مَا كانَ مباحًا لهُ قبلَ الإحرامِ مِنَ النِّكاحِ والطِّيبِ ونحوِهِمَا.

وشرعًا: (نيَّةُ النُّسكِ)؛ أيْ: نيَّةُ الدُّخولِ فيهِ،

  • لَا نيّتُهُ أنْ يحُجَّ أوْ يعتمرَ.

(سُنَّ لمُريدِهِ)؛ أيْ: مُريدِ الدُّخولِ فِي النُّسكِ مِنْ ذكرٍ وأنثَى،

  • (غُسْلٌ) ولوْ حائضًا ونُفَساءَ؛ لأنَّ النَّبِيَّ چ «أمرَ أسماءَ بنْتَ عُميسٍ وهيَ نُفَساءُ أنْ تغتسلَ»، رواهُ مسلمٌ(١)، وأمرَ عائشَةَ ږ أنْ تغتسلَ لإهلالِ الحجِّ وهيَ حائضٌ(٢).
  • (أوْ تيمُّمٌ لعدمٍ)؛ أيْ:
  • عدمِ الماءِ،
  • أوْ تعذُّرِ استعمالِهِ لنحوِ مرضٍ،
  • (و) سُنَّ لهُ أيضًا (تنظُّفٌ) بأخذِ شعرٍ، وظُفرٍ، وقطعِ رائحةٍ كريهةٍ؛ لئلَّا يَحتاجَ إليهِ فِي إحرامِهِ فلَا يتمكّنَ مِنهُ.
٥٩٨
  • (و) سُنَّ لهُ أيضًا (تطيُّبٌ) فِي بدنِهِ بمسكٍ، أوْ بَخورٍ، أوْ ماءِ وردٍ ونحوِهَا؛ لقولِ عائشةَ ږ : «كنْتُ أطيِّبُ رسولَ اللهِ چ لإحرامِهِ قبلَ أنْ يُحْرِمَ، ولحِلِّهِ قبلَ أنْ يطوفَ بالبيْتِ»(٣)، وقالَتْ: «كأنِّي أنظرُ إلَى وَبِيصِ المسكِ فِي مفارقِ رسولِ اللهِ چ وهوَ مُحْرِمٌ»، متَّفقٌ عليهِ(٤).
  • وكُرهَ أنْ يتطيّبَ فِي ثوبِهِ،
  • ولهُ استدامَةُ لُبسِهِ مَا لمْ ينزِعْهُ،
  • فإنْ نزَعَهُ فليسَ لهُ أنْ يلبَسَهُ قبلَ غَسلِ الطِّيبِ مِنهُ.

ومتَى:

  • تعمّدَ مسَّ مَا علَى بدنِهِ مِنَ الطِّيبِ،
  • أوْ نحّاهُ عن موضعِهِ ثمَّ ردَّهُ إليهِ،
  • أوْ نقلَهُ إلَى موضعٍ آخرَ:
  • فدَى،
  • لَا إنْ سالَ بعرَقٍ أوْ شمسٍ.
  • (و) سُنَّ لهُ أيضًا (تجرُّدٌ مِنْ مَخِيطٍ)، وهوَ: كلُّ مَا يُخاطُ علَى قدرِ الملبوسِ عليهِ؛ كالقميصِ والسّراويلِ؛ «لأنَّهُ چ تجرّدَ لإهلالِهِ»، رواهُ التّرمذِيُّ(٥)،
٥٩٩
  • وسُنَّ أيضًا أنْ يُحْرِمَ (فِي إزارٍ ورداءٍ أبيضَيْنِ) نظيفَيْنِ، ونعلَيْنِ؛ لقولِهِ چ : «ولْيُحْرِمْ أحدُكُمْ فِي إزارٍ ورداءٍ ونعلَيْنِ»، رواهُ أحمدُ(٦)،
  • والمرادُ بالنّعلَيْنِ: التّاسومَةُ،
  • ولَا يجوزُ لهُ لُبسُ السَّرموزَةِ والجُمْجُمِ، قالَهُ فِي الفروعِ(٧).
  • (و) سُنَّ (إحرامٌ عَقِبَ ركعتيْنِ) نفلًا، أوْ عَقِبَ فريضَةٍ؛ لأنَّهُ چ أهلَّ دُبرَ صلَاةٍ، رواهُ النَّسائِيُّ(٨).

(ونيّتُهُ شرطٌ) فلَا يصيرُ مُحْرِمًا بمجرَّدِ التّجرُّدِ أوِ التّلبيَةِ مِنْ غيرِ نيَّةِ الدُّخولِ فِي النُّسكِ؛ لحديثِ: «إنَّمَا الأعمالُ بالنِّيّاتِ»(٩).

(ويُستحبُّ:

  • قولُهُ: اللّهمَّ إنِّي أريدُ نُسكَ كذَا)؛ أيْ: أنْ يعيِّنَ مَا يُحْرِمُ بهِ، ويلفِظَ بهِ،
٦٠٠
  • وأنْ يقولَ: (فيسِّرْهُ لِي) وتقبّلْهُ منِّي،
  • وأنْ يَشترطَ فيقولَ: (وإنْ حبسَنِي حابسٌ فمحِلِّي حيثُ حبستَنِي)؛ لقولِهِ چ لضُبَاعَةَ بنْتِ الزُّبَيْرِ حينَ قالتْ لهُ: إنِّي أريدُ الحجَّ وأجدُنِي وجِعَةً، فقالَ: «حُجِّي واشترطِي وقولِي: اللّهمَّ محِلِّي حيثُ حبستَنِي»، متَّفقٌ عليهِ(١٠). زادَ النَّسائِيُّ فِي روايَةٍ إسنادُهَا جيِّدٌ: «فإنَّ لكِ علَى ربِّكِ مَا استثنيْتِ»(١١)؛
  • فمتَى حُبسَ بمرضٍ، أوْ عدوٍّ، أوْ ضلَّ الطّريقَ حلّ ولَا شيءَ عليهِ،
  • ولوْ شرَطَ أنْ يَحِلَّ متَى شاءَ،
  • أوْ إنْ أفسدَهُ لمْ يقضِهِ،
  • لمْ يصحَّ الشّرطُ.

ولَا يبطلُ الإحرامُ:

  • بجنونٍ،
  • أوْ إغماءٍ،
  • أوْ سُكرٍ،
  • كموتٍ،
٦٠١
  • ولَا ينعقدُ معَ وجودِ أحدِهَا.

ڭ ڭ ڭ

والأنساكُ:

  • تمتُّعٌ،
  • وإفرادٌ،
  • وقِرانٌ.
  • (وأفضلُ الأنساكِ التّمتُّعُ) فالإفرادُ، فالقِرانُ، قالَ أحمدُ: «لَا أشكُّ أنَّهُ چ كانَ قارنًا، والمتعَةُ أحبُّ إليَّ» انتهَى(١٢). وقالَ: «لأنَّهُ آخِرُ مَا أمرَ بهِ النَّبِيُّ چ »(١٣). ففِي الصّحيحَيْنِ: أنَّهُ چ أمرَ أصحابَهُ لمَّا طافُوا وسعَوْا أنْ يجعلُوهَا عمرةً إلَّا مَنْ ساقَ هديًا، وثبتَ علَى إحرامِهِ لسَوقِهِ الهدْيَ، وتأسّفَ بقولِهِ: «لوِ استقبلْتُ مِنْ أمرِي مَا استدبرْتُ مَا سقْتُ الهدْيَ ولأَحللْتُ معَكُمْ»(١٤).

(وصفتُهُ)؛ أيِ: التّمتُّعِ:

  • (أنْ يُحْرِمَ بالعمرَةِ،
٦٠٢
  • فِي أشهُرِ الحجِّ،
  • ويفرغَ مِنْهَا،
  • ثمَّ يُحْرِمَ بالحجِّ فِي عامِهِ) مِنْ مكَّةَ، أوْ قُربِهَا، أوْ بعيدٍ مِنْهَا.

والإفرادُ: أنْ يُحْرِمَ بحجٍّ ثمَّ بعمرةٍ بعدَ فراغِهِ مِنهُ.

والقِرانُ:

  • أنْ يُحْرِمَ بهِمَا معًا،
  • أوْ بِهَا ثمَّ يُدخلَهُ عَلَيْهَا قبلَ شروعٍ فِي طوافِهَا،
  • ومَنْ أحرمَ بهِ ثمَّ أدخلَهَا عليهِ: لمْ يصحَّ إحرامُهُ بِهَا.

(و) يجبُ (علَى الأُفقيِّ) وهوَ: مَنْ كانَ مسافَةَ قصرٍ فأكثرَ مِنَ الحرمِ -إنْ أحرمَ متمتِّعًا أوْ قارنًا- (دمُ) نُسكٍ لَا جبرانٍ، بخلافِ:

  • أهلِ الحرمِ،
  • ومَنْ مِنهُ دونَ المسافَةِ:
  • فلَا شيءَ عليهِ؛ لقولِهِ تعالَى: {ذَٰلِكَ لِمَن لَّمۡ يَكُنۡ أَهۡلُهُۥ حَاضِرِي ٱلۡمَسۡجِدِ ٱلۡحَرَامِ} [البقرة:١٩٦].

ويُشترطُ:

  • أنْ يحرمَ بها مِنْ ميقاتٍ، أوْ مسافَةِ قصرٍ فأكثرَ مِنْ مكَّةَ،
  • وأنْ لَا يسافرَ بينَهُمَا،
  • فإنْ سافرَ مسافَةَ قصرٍ فأحرمَ فلَا دمَ عليهِ،
٦٠٣

وسُنَّ لمفرِدٍ وقارنٍ:

  • فسخُ نيّتِهِمَا بحجٍّ،
  • وينويانِ بإحرامِهِمَا ذلكَ عمرةً مفرَدةً؛ لحديثِ الصّحيحَيْنِ السّابقِ(١٥)،
  • فإذَا حلَّا أحرمَا بهِ ليصيرَا متمتِّعيْنِ؛
  • مَا لمْ يسوقَا هديًا،
  • أوْ يقفَا بعرفَةَ.

وإنْ ساقَهُ متمتِّعٌ لمْ يكنْ لهُ أنْ يَحِلَّ،

  • فيُحْرِمُ بحجٍّ إذَا طافَ وسعَى لعمرتِهِ قبلَ حلقٍ،
  • فإذَا ذبحَهُ يومَ النّحرِ حلَّ مِنْهُمَا.

(وإنْ حاضَتِ المرأَةُ) المتمتِّعَةُ قبلَ طوافِ العمرَةِ (فخشيَتْ فواتَ الحجِّ أحرمَتْ بهِ) وجوبًا، (وصارَتْ قارنةً)؛ لمَا روَى مسلمٌ أنَّ عائشَةَ ږ كانتْ متمتِّعةً فحاضَتْ، فقالَ لهَا النَّبِيُّ چ : «أهِلِّي بالحجِّ»(١٦)،

  • وكذَا لوْ خشيَهُ غيرُهَا.

ومَنْ أحرمَ:

  • وأطلقَ: صحَّ، وصرَفَهُ لمَا شاءَ،
٦٠٤
  • و«بمثلِ مَا أحرمَ فلانٌ»: انعقدَ بمثلِهِ،
  • وإنْ جهِلَ(١٧) جعَلَهُ عمرَةً؛ لأنَّهَا اليقينُ.

ويصحُّ:

  • «أحرمْتُ يومًا»،
  • أوْ «بنصفِ نُسكٍ».

لَا «إنْ أحرمَ فلانٌ فأنَا مُحْرِمٌ»؛ لعدمِ جزمِهِ.

ڭ ڭ ڭ

(وإذَا استوَى علَى راحلتِهِ قالَ):

  • قطعَ بهِ جماعَةٌ،
  • والأصحُّ عَقِبَ إحرامِهِ:
  • (لبّيْكَ اللّهمَّ لبّيْكَ)؛ أيْ: أنَا مقيمٌ علَى طاعتِكَ وإجابَةِ أمرِكَ، (لبّيْكَ لَا شريكَ لكَ لبّيْكَ، إنَّ الحمدَ والنِّعمَةَ لكَ والمُلكَ، لَا شريكَ لكَ)، روَى ذلكَ ابنُ عمرَ ژ عنْ رسولِ اللهِ چ فِي حديثٍ متَّفقٍ عليهِ(١٨).

وسُنَّ:

  • أنْ يذكرَ نُسكَهُ فِيهَا،
  • وأنْ يبدأَ القارنُ بذكرِ عمرتِهِ،
٦٠٥
  • وإكثارُ التّلبيَةِ،
  • وتتأكّدُ:
  • إذَا علَا نَشزًا،
  • أوْ هبطَ واديًا،
  • أوْ صلَّى مكتوبةً،
  • أوْ أقبلَ ليلٌ أوْ نهارٌ،
  • أوِ الْتقَتِ الرِّفاقُ،
  • أوْ سمِعَ ملبِّيًا،
  • أوْ فعَلَ محظورًا ناسيًا،
  • أوْ ركِبَ دابّتَهُ، أوْ نزلَ عنْهَا،
  • أوْ رأَى البيْتَ.
  • (يصوِّتُ بِهَا الرّجلُ)؛ أيْ: يجهرُ بالتّلبيَةِ؛ لخبرِ السّائبِ بنِ خلّادٍ ڔ مرفوعًا: «أتانِي جبريلُ فأمرَنِي أنْ آمرَ أصحابِي أنْ يرفعُوا أصواتَهُمْ بالإهلالِ والتّلبيَةِ»، صحَّحَهُ التّرمذِيُّ(١٩)،
  • وإنَّمَا يُسنُّ الجهرُ بالتّلبيَةِ فِي غيرِ:
٦٠٦
  • مساجدِ الحِلِّ وأمصارِهِ،
  • وفِي غيرِ طوافِ القدومِ والسّعْيِ بعدَهُ،

وتُشرعُ بالعربيَّةِ لقادرٍ، وإلَّا فبِلُغتِهِ،

ويُسنُّ بعدَهَا:

  • دعاءٌ،
  • وصلاةٌ علَى النَّبِيِّ چ .

(وتُخفيهَا المرأَةُ) بقدرِ مَا تُسمِعُ رفيقَتَهَا،

  • ويُكرَهُ جهرُهَا فوقَ ذلكَ؛ مخافَةَ الفتنَةِ،

ولَا تُكرهُ التّلبيَةُ لحلالٍ.

ڭ ڭ ڭ

(١) أخرجه أحمد (٣/٣٢٠)، ومسلم (١٢١٨) من حديث جابر بن عبد الله ژ ، في سياق حجَّة النبي چ .

(٢) أخرجه أحمد (٣/٣٩٤)، ومسلم (١٢١٣) من حديث جابر بن عبد الله ژ . وأخرجه البخاري (١٥٥٦) من حديث عائشة ږ .

(٣) أخرجه أحمد (٦/٣٩)، والبخاري (١٥٣٩)، ومسلم (١١٨٩).

(٤) أخرجه أحمد (٦/٣٨)، والبخاري (١٥٣٨)، ومسلم (١١٩٠).

(٥) أخرجه الترمذي (٨٣٠) من حديث خارجة بن زيد بن ثابت عن أبيه ڔ به مرفوعًا.

وقال: (حديث حسنٌ غريب)، وصححه ابن خزيمة (٢٥٩٥)، وضعَّفه العقيلي في الضعفاء (ترجمة محمد بن موسى أبو غُزَيَّة ٥/٣٩٤).

(٦) أخرجه أحمد (٢/٣٤) من حديث ابن عمر ژ .

صححه ابن خزيمة (٢٦٠١)، وابن المنذر في الإشراف (٣/١٨٤).

(٧) قارن بما في: الفروع (٥/٤٢٥)، وانظر: معونة أولي النهى (٤/٥٣).

(٨) أخرجه أحمد (١/٢٦٠، ٢٨٥)، وأبو داود (١٧٧٠)، والترمذي (٨١٩)، والنسائي (٥/١٦٢) من حديث ابن عباس ژ به.

قال الترمذي: (هذا حديث حسن غريب)، وصححه الحاكم (١/٤٥١)، وضعَّفه البيهقي في معرفة السنن والآثار (٧/١٢٠).

(٩) سبق تخريجه في (ص٥٤٣).

(١٠) أخرجه أحمد (٦/١٦٤)، والبخاري (٥٠٨٩)، ومسلم (١٢٠٧) من حديث عائشة ږ .

(١١) وأخرجه أحمد (١/٣٥٢)، والنسائي (٥/١٦٧) واللفظ له من حديث ابن عباس ژ .

صحح إسناده ابن الملقن في البدر المنير (٦/٤١٤).

(١٢) انظر: المبدع للبرهان ابن مفلح (٣/١١٣).

(١٣) انظر: زاد المسافر (٢/٥٤٠).

(١٤) أخرجه أحمد (٣/٢٦٦)، والبخاري واللفظ له (١٦٥١)، ومسلم (١٢١٦) من حديث جابر بن عبد الله ژ مرفوعًا: «لو استقبلت من أمري ما استدبرت ما أهديت، ولولا أن معي الهدي لأحللت».

(١٥) أي حديث: (أنه چ أمر أصحابه لما طافوا وسعوا أن يجعلوها عمرة)، المتقدم قريبًا.

(١٦) أخرجه أحمد (٦/١٦٤)، والبخاري (١٥٥٦)، ومسلم (١٢١١) من حديث عائشة ږ ، وأخرجه أحمد أيضًا (٣/٣٩٤)، ومسلم (١٢١٣) من حديث جابر بن عبد الله ژ .

(١٧) في (ع، م): «جهله».

(١٨) أخرجه أحمد (٢/٣)، والبخاري (١٥٤٩)، ومسلم (١١٨٤).

(١٩) أخرجه أحمد (٤/٥٦)، وأبو داود (١٨١٤)، والترمذي (٨٢٩)، وابن ماجه (٢٩٢٢)، والنسائي (٥/١٦٢).

صححه ابن خزيمة (٢٦٢٥)، وابن حبان (٣٨٠٢)، وابن الملقن في البدر المنير (٦/١٥٢).
