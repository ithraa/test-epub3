١٥٧٩

(بابُ القطعِ فِي السّرقَةِ)

وهيَ: أخذُ مالٍ علَى وجهِ الاختفاءِ مِنْ مالكِهِ أوْ نائبِهِ.

(إذَا أخذَ):

  • المكلّفُ،
  • (الملتزمُ) مسلمًا كانَ أوْ ذمِّيًا، بخلافِ المستأمِنِ ونحوِهِ،
  • (نصابًا،
  • مِنْ حرزِ مثلِهِ،
  • مِنْ مالِ معصومٍ)، بخلافِ حربيٍّ،
  • (لَا شُبهَةَ لهُ فيهِ،
  • علَى وجهِ الاختفاءِ:
  • قُطِعَ)؛ لقولِهِ تعالَى: {وَٱلسَّارِقُ وَٱلسَّارِقَةُ فَٱقۡطَعُوٓاْ أَيۡدِيَهُمَا} [المائدة:٣٨]، ولحديثِ عائشةَ ږ : «تُقطعُ اليدُ فِي ربعِ دينارٍ فصاعدًا»(١).

(فلَا قَطْعَ علَى(٢)):

١٥٨٠
  • (مُنْتَهِبٍ)، وهوَ: الَّذِي يأخذُ المالَ علَى وجهِ الغنيمَةِ،
  • (ولَا مُخْتَلِسٍ)، وهوَ: الَّذِي يختطفُ(٣)الشّيءَ ويمرُّ بهِ ،
  • (ولَا غاصبٍ،
  • ولَا خائنٍ فِي وديعَةٍ، أوْ عاريةٍ أوْ غيرِهَا)؛ لأنَّ ذلكَ ليسَ بسرقةٍ،
  • لكنِ الأصحُّ: أنَّ جاحدَ العاريَّةِ يُقطَعُ إنْ بلغَتْ نصابًا؛ لقولِ ابنِ عمرَ ژ : «كانتْ مخزوميَّةٌ تستعيرُ المتاعَ وتجحدُهُ، فأمرَ النَّبِيُّ چ بقطعِ يدِهَا»، رواهُ أحمدُ والنَّسائِيُّ وأبُو داودَ(٤)، قالَ أحمدُ: «لَا أعرفُ شيئًا يدفعُهُ»(٥).

(ويُقطَعُ الطَّرَّارُ)؛ وهوَ: (الَّذِي يَبُطُّ الجيبَ أوْ غيرَهُ ويأخذُ مِنهُ)، أوْ بعدَ سقوطِهِ نصابًا؛ لأنَّهُ سرقَ مِنْ حِرْزٍ.

ڭ ڭ ڭ

(ويُشتَرَطُ) للقطعِ فِي السّرِقَةِ ستَّةُ شُرُوطٍ:

أحدُهَا: (أنْ يكونَ المسروقُ مالًا محتَرَمًا)؛ لأنَّ مَا ليسَ بمالٍ لَا حُرمةَ ؛

١٥٨١

لهُ، ومالُ الحربيِّ تجوزُ سرقتُهُ بكلِّ حالٍ

  • (فلَا قطعَ بسرقَةِ:
  • آلةِ لهوٍ)؛ لعدمِ الاحترامِ،
  • (ولَا) بِسَرِقَةِ (مُحَرَّمٍ؛ كالخمرِ)، وصليبٍ، وآنيةٍ فِيهَا خمرٌ،
  • ولَا بسرقَةِ ماءٍ، أوْ إناءٍ فيهِ ماءٌ،
  • ولَا بسرقةِ مُكَاتَبٍ، وأمِّ ولدٍ،
  • ومُصحَفٍ، وحُرٍّ ولوْ صغيرًا، ولَا بمَا عَلَيْهِمَا.

ڭ ڭ ڭ

الشّرطُ الثَّانِي: مَا أشارَ إليهِ بقولِهِ: (ويُشْتَرَطُ) أيضًا (أنْ يكونَ) المسروقُ (نصابًا، وهوَ)؛ أيْ: نصابُ السَّرقَةِ:

  • (ثلاثةُ دراهمَ) خالصةٍ، أوْ تخلصُ مِنْ مَغْشُوشَةٍ،
  • (أوْ ربعُ دينارٍ)؛ أيْ: مِثقالٌ وإنْ لمْ يُضرَبْ،
  • (أوْ عَرْضٌ قيمتُهُ كأحدِهِمَا)؛ أيْ: ثلاثَةُ دراهمَ، أوْ ربعُ دينارٍ.
  • فلَا قطعَ بسرقةِ مَا دونَ ذلكَ؛ لقولِهِ چ : «لَا تُقطَعُ اليدُ إلَّا فِي ربعِ دينارٍ فصاعدًا»، رواهُ أحمدُ ومسلمٌ وغيرُهُمَا(٦)، وكانَ ربعُ الدِّينارِ يومئذٍ: ثلاثةَ دراهمَ، والدِّينارُ: اثْنَا عَشَرَ دِرْهَمًا. رواهُ أحمدُ(٧).
١٥٨٢

(وإذَا(٨):

  • نقصَتْ قيمةُ المسروقِ) بعدَ إخراجِهِ: لمْ يَسْقُطِ القَطْعُ؛ لأنَّ النُّقصانَ وُجِدَ فِي العَيْنِ بعدَ سرقتِهَا،
  • (أوْ مَلَكَهَا) أيِ: العينَ المسروقةَ (السّارقُ) ببيعٍ، أوْ هبةٍ، أوْ غيرِهِمَا: (لمْ يسقطِ القَطْعُ) بعدَ التَّرافعِ إلَى الحاكمِ.

(وتعتبرُ قيمتُهَا) أيْ: قيمةُ العينِ المَسروقَةِ: (وقْتَ إخراجِهَا مِنَ الحرزِ)؛ لأنَّهُ وقْتُ السّرقَةِ الَّتِي بِهَا وجبَ القطعُ،

  • (فلوْ:
  • ذَبَحَ فيهِ)؛ أيْ: فِي الحرزِ (كبشًا) فنقصَتْ قيمتُهُ،
  • (أوْ شقَّ فيهِ ثوبًا فنَقَصَتْ قيمتُهُ عنْ نصابِ) السّرقَةِ،
  • (ثمَّ أخرجَهُ) مِنَ الحرزِ: فلَا قطعَ؛ لأنَّهُ لمْ يُخرِجْ مِنَ الحرزِ نصابًا،
  • (أوْ أتلفَ فيهِ)؛ أيْ: فِي الحرزِ (المالَ: لمْ يُقطَعْ)؛ لأنَّهُ لمْ يُخرِجْ مِنهُ شيئًا.

ڭ ڭ ڭ

(و) الشّرطُ الثّالثُ: (أنْ يخرجَهُ مِنَ الحرزِ،

  • فإنْ سرقَهُ مِنْ غيرِ حرزٍ)؛ كمَا لوْ وَجَدَ بابًا مفتوحًا، أوْ حِرْزًا مهتوكًا: (فلَا قَطعَ) عليهِ.
١٥٨٣

(وحِرْزُ المالِ: مَا العادةُ حفظُهُ فيهِ)؛ إذِ الحرزُ معناهُ الحفظُ، ومنْهُ: احْتَرَزَ؛ أيْ: تحفّظَ.

  • (ويختلفُ) الحرزُ (باختلافِ الأموالِ والبُلْدَانِ، وعدلِ السُّلطانِ وجَوْرِهِ، وقُوّتِهِ وضعفِهِ)؛ لاختلافِ الأحوالِ باختلافِ المذكوراتِ؛
  • (فحِرزُ الأموالِ)؛ أيِ: النُّقوَدِ (والجواهرِ والقُمَاشِ فِي الدُّورِ والدّكاكينِ والعُمرانِ)؛ أيِ: الأبنيَةِ الحَصِينةِ والمَحالِّ المسكونَةِ مِنَ البلدِ: (وراءَ الأبوابِ والأغلاقِ الوثيقَةِ)، والغَلَقُ: اسمٌ للقُفلِ خَشَبًا كانَ أوْ حديدًا.
  • وصندوقٌ بسُوقٍ وثَمَّ حارسٌ: حِرْزٌ.
  • (وحِرْزُ البقلِ، وقُدُورِ البَاقِلَاءِ، ونحوِهِمَا)؛ كقدورِ طَبِيخٍ، وخَزَفٍ: (وراءَ الشّرَائِجِ)؛ وهيَ: مَا يُعمَلُ مِنْ قَصَبٍ أوْ نحوِهِ، يُضَمُّ بعضُهُ إلَى بعضٍ بِحَبْلٍ أوْ غيرِهِ، (إذَا كانَ فِي السُّوقِ حارسٌ)؛ لجريانِ العَادَةِ بذلكَ.
  • (وحِرْزُ الحَطَبِ والخَشَبِ: الحَظَائِرُ) جمعُ حَظِيرَةٍ -بالحاءِ المُهْمَلَةِ والظّاءِ المُعْجَمَةِ-: مَا يُعملُ للإبلِ والغَنَمِ مِنَ الشّجرِ تأوِي إليهِ فَيَعْبُرُ(٩) بعضُهُ فِي بعضٍ ويُربَطُ.
  • (وحِرْزُ المواشِي: الصِّيَرُ) جمعُ صِيرَةٍ، وهيَ: الحظيرَةُ.
١٥٨٤
  • (وحرزُهَا)؛ أيِ: المواشِي: (فِي المرعَى بالرّاعِي ونظرِهِ إلَيْهَا غالبًا)؛ فمَا غابَ عنْ مشاهدتِهِ غالبًا فقدْ خرجَ عنِ الحرزِ.
  • وحِرْزُ سُفُنٍ فِي شَطٍّ: بربطِهَا.
  • وإبلٍ باركةٍ معقولَةٍ: بحافظٍ حتَّى نائمٍ.
  • وحُمُولتِها:
  • بتقطيرِهَا: معَ قائدٍ يراهَا،
  • ومعَ عدمِ تقطيرٍ: بسائقٍ يراهَا.
  • وحرزُ ثيابٍ فِي حمّامٍ ونحوِهِ: بحافظٍ؛ كقعودِهِ علَى متاعٍ.
  • وإنْ فرّطَ حافظُ حمّامٍ بنومٍ أوْ تشاغُلٍ: ضَمِنَ، ولَا قَطْعَ علَى سارقٍ إذًا.
  • وحرزُ بابٍ ونحوِهِ: تركيبُهُ بموضعِهِ.

ڭ ڭ ڭ

(و) الشّرطُ الرّابعُ: (أنْ تنتفِيَ الشُّبهةُ) عنِ السّارقِ؛ لحديثِ «ادرؤوا الحدودَ بالشُبُهاتِ مَا استطعْتُمْ»(١٠)؛

  • (فلَا يُقْطَعُ) سارقٌ:
  • (بالسّرقةِ مِنْ مالِ أبيهِ وإنْ علَا،
  • ولَا) بسرقةٍ (مَنْ مالِ ولدِهِ وإنْ سَفَلَ)؛
١٥٨٥
  • لأنَّ نفقَةَ كلٍّ مِنْهُمَا تجبُ فِي مالِ الآخرِ.
  • (والأبُ والأمُّ فِي هذَا سواءٌ)؛ لمَا ذُكرَ.

(ويُقْطَعُ:

  • الأخُ) بسرقةِ مالِ أخيهِ،
  • (و) يُقطَعُ (كلُّ قريبٍ بسرقةِ مالِ(١١) قريبِهِ)؛
  • لأنَّ القرابَةَ هنَا لَا تمنعُ قبولَ الشّهادَةِ مِنْ أحدِهِمَا للآخرِ، فلمْ تمنعِ القَطْعَ.

(ولَا يُقْطَعُ أحدٌ مِنَ الزّوجَيْنِ بسرقتِهِ مِنْ مالِ الآخرِ، ولوْ كانَ مُحْرَزًا عنهُ)؛ روَى ذلكَ سعيدٌ عنْ عُمَرَ ڔ بإسنادٍ جَيِّدٍ(١٢).

(وإذَا سَرَقَ:

  • عبدٌ) ولوْ مُكاتَبًا (مَنْ مالِ سيِّدِهِ،
  • أوْ سيِّدٌ مِنْ مالِ مُكَاتَبِهِ):
١٥٨٦
  • فلَا قَطْعَ.

(أوْ) سرقَ:

  • (حُرٌّ) أوْ قنٌّ (مسلمٌ مِنْ:
  • بيتِ المالِ): فلَا قطعَ.
  • (أوْ) سرقَ (مِنْ غنيمةٍ لمْ تُخَمَّسْ): فلَا قطعَ؛ لأنَّ لبيْتِ المالِ فِيهَا خُمُسَ الخُمُسِ.
  • (أوْ سرقَ فقيرٌ مِنْ غَلَّةِ وقفٍ علَى الفقراءِ): فلَا قطعَ؛ لدخولِهِ فيهِمْ.
  • (أوْ) سرقَ (شخصٌ مِنْ مالٍ فيهِ شِرْكَةٌ:
  • لهُ،
  • أوْ لأحدٍ ممَّنْ لَا يُقْطَعُ بالسّرِقَةِ مِنهُ)؛ كأبيهِ، وابنِهِ، وزوجِهِ، ومكاتَبِهِ:
  • (لمْ يُقطَعْ)؛ للشُّبهةِ.

ڭ ڭ ڭ

الشّرطُ الخامسُ: ثبوتُ السّرِقَةِ، وقدْ ذكَرَهُ بقولِهِ: (ولَا يُقْطَعُ إلَّا:

  • بشهادةِ عدلَيْنِ) يصفانِهَا بعدَ الدّعوَى مِنْ مالكٍ، أوْ مَنْ يقومُ مقامَهُ،
  • (أوْ) بـ(ـإقرارِ) السّارقِ:
  • (مرّتَيْنِ) بالسّرقَةِ ويصفُهَا فِي كلِّ مرَّةٍ؛ لاحتمالِ ظنِّهِ القطعَ فِي
١٥٨٧
  • حالٍ لَا قطعَ فِيهَا،
  • (ولَا يَنْزِعُ)؛ أيْ: يرجعُ (عنْ إقرارِهِ حتَّى يُقْطَعَ)،
  • ولَا بأسَ بتلقينِهِ الإنكارَ.

ڭ ڭ ڭ

(و) الشّرطُ السّادسُ: (أنْ يطالِبَ المسروقُ مِنهُ) السّارقَ (بمالِهِ)؛ فلوْ:

  • أقرَّ بسرقةٍ مِنْ مالِ غائبٍ،
  • أوْ قامَتْ بِهَا بيِّنةٌ:
  • انتُظِرَ حضورُهُ ودعواهُ، فيُحبَسُ وتعادُ الشّهادَةُ.

ڭ ڭ ڭ

(وإذَا وجبَ القطعُ) لاجتماعِ شُرُوطِهِ:

  • (قُطِعَتْ يدُهُ اليُمنَى)؛ لقراءَةِ ابنِ مسعودٍ ڔ : «فاقطعُوا أيمانَهُمَا»(١٣)، ولأنَّهُ قولُ أبِي بكرٍ وعمرَ ژ ولَا مخالفَ لهمَا مِنَ الصّحابَةِ.
  • (مَنْ مَفْصِلِ الكفِّ)؛ لقولِ أبِي بكرٍ وعمرَ ژ ولَا مُخالِفَ لهمَا مِنَ الصّحابَةِ(١٤).
١٥٨٨
  • (وحُسِمَتْ) وجوبًا بغَمْسِهَا فِي زيتٍ مغليٍّ؛ لتستدَّ أفواهُ العُرُوقِ فينقطعَ الدّمُ،
  • فإنْ عادَ: قُطِعَتْ رجلُهُ اليُسرَى مِنْ مَفْصِلِ كعبِهِ بتركِ عقِبِهِ وحُسِمَتْ،
  • فإنْ عادَ: حُبِسَ حتَّى يتوبَ، وحَرُمَ أنْ يُقْطَعَ.

ڭ ڭ ڭ

(ومَنْ سرقَ شيئًا مِنْ غيرِ حرزٍ:

  • ثمرًا كانَ،
  • أوْ كُثَرًا) -بضمِّ الكافِ وفتحِ المثلَّثَةِ- طَلْعُ الفحَّالِ،
  • (أوْ غيرَهُمَا) مِنْ جُمَّارٍ أوْ غيرِهِ:
  • (أُضْعِفَتْ عليهِ القيمَةُ) أيْ: ضَمِنَهُ بعِوَضِهِ مرَّتَيْنِ، قالَهُ القاضِي، واختارَهُ الزَّرْكشِيُّ(١٥).
  • وقدّمَ فِي التّنقيحِ: أنَّ التّضعيفَ خاصٌّ بالثَّمَرِ، والطَّلعِ، والجُمَّارِ، والماشيَةِ، وقطعَ بهِ فِي المنتهَى(١٦)وغيرِهِ؛ لأنَّ التّضعيفَ وردَ فِي هذهِ الأشياءِ علَى خلافِ القياسِ، فلَا يُتَجَاوَزُ بهِ مَحَلُّ النّصِّ.
  • (ولَا قطعَ)؛ لفواتِ شرطِهِ، وهوَ الحرزُ.

ڭ ڭ ڭ

(١) أخرجه أحمد (٦/٣٦)، والبخاري (٦٧٨٩)، ومسلم (١٦٨٤).

(٢) في (الأصل، ز، س، ع، ل) «على» من الشرح، وقد اختلفت نسخ زاد المستقنع على وجهين: «فلا يقطع منتهب»، و«فلا قطع على منتهب»، والمثبت هو الموافق لما في زاد المستقنع (ص٤٢٣ت: القاسم)، ولكن ليس في نسخ الروض المعتمدة لدينا «يقطع».

(٣) في (ب، د، ل): «يخطف».

(٤) أخرجه أحمد (٢/١٥١)، وأبو داود (٤٣٩٥)، والنسائي (٨/٧٠ - ٧١) من حديث ابن عمر ژ ، ورُوي عن نافع مرسلًا.

ورجَّح المرسل أبو حاتم في العلل لابنه (س١٣٦١)، والدارقطني في العلل (س٢٧٥٨).

وأخرجه مسلم (١٦٨٨) من حديث عائشة ږ .

(٥) انظر: مسائل الكوسج (٢٠٥٠)، مسائل عبد الله (ص٤٢٩) برقم (١٥٤٩).

(٦) سبق تخريجه في (ص١٥٧٩).

(٧) أخرجه أحمد (٦/٨٠ - ٨١) من حديث عائشة ږ .

(٨) في (د): «فإذا»، وفي (ز): «وإن».

(٩) هكذا في النسخ المعتمدة لدينا، وهو الموافق لما في كشاف القناع (١٤/١٤٧)، وشرح ابن النجار على المنتهى (١٠/٤٨٧)، أما شرح المنتهى للبهوتي (٦/٢٤٥ ط. التركي) ففيه: «فيصير».

(١٠) سبق تخريجه في (ص١٥٦٠).

(١١) في (د، ل): «من مال»، وكذلك في الموضع الذي قبله.

(١٢) لم نقف عليه، واحتج في المغني (١٢/٤٦١) بقول عمر ڔ لعبدالله بن عمرو الحضرمي حين قال: إن غلامي سرق مرآة امرأتي: (أرسِلْه، لا قطع عليه، خادمكم أخذ متاعكم)، قال الموفق: (وإذا لم يقطع عبده بسرقة مالها، فهو أولى).

أخرجه مالك (٢٤٣٣)، وعبدالرزاق (١٠/٢١٠)، وابن أبي شيبة (١٠/٢١)، وابن المنذر في الأوسط (١٢/٣٥٣)، والبيهقي (٨/٢٨٢)، وصحَّحه ابن الملقن في البدر المنير (٨/٦٧٧).

(١٣) أخرجه الطبري في تفسيره (٨/٤٠٧ - ٤٠٨)، والبيهقي (٨/٢٧١) من طرق عن ابن مسعود ڔ . قال البيهقي: (منقطع).

(١٤) غريب عنهما، قاله ابن الملقن في البدر المنير (٨/٦٨٥)، ولم نقف على من أخرجه من قولهما؛ لكن أخرج ابن أبي شيبة (١٠/٣٠)، والبيهقي (٨/٢٧١) عن عمر ڔ : أنه قطع من المفصل.

(١٥) قارن بما في: الأحكام السلطانية (ص٢٨١)، شرح الزركشي (٦/٣٣٦).

(١٦) انظر: التنقيح المشبع (ص٤٤٩)، المنتهى (٥/١٥٣).
