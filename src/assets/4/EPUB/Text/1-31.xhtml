(٣) أخرجه أحمد (٢/٤٣٠)، والبخاري (٨٩١)، ومسلم (٨٨٠).

(٤) انظر: المبدع (٢/١٦٩).

(٥) في (د): «إعادتهما».

(٦) أخرجه أحمد (٢/٦، ١١)، والبخاري (٩٣٧)، ومسلم (٨٨٢) زاد أحمد مسلم في رواية: في بيته.

(٧) أخرجه أبو داود (١١٣٠) من حديث يزيد بن أبي حبيب عن عطاء بن أبي رباح عن ابن عمر مرفوعًا، بنحوه.

وأخرجه أبو داود (١١٣٣)، والترمذي (٥٢٣م) من حديث ابن جريج عن عطاء عن ابن عمر من فِعلِه موقوفًا.

وتابع ابن جريج: أبو إسحاق السبيعي والزبير بن عدي، عند عبدالرزاق (٣/٢٤٦-٢٤٧)، وابن أبي شيبة (٢/١٣٢)، وابن المنذر في الأوسط (٤/١٣٩) عن أبي إسحاق وحده، وعبدالملك بن أبي سليمان عند ابن أبي شيبة (٢/١٣٩) كلهم عن عطاء عن ابن عمر موقوفًا.

وسُئل أحمد عن عطاء عن ابن عمر فقال: (رأى ابن عمر ولم يسمع منه).

(٨) في (ز، ع): «فببيته أفضل».

(٩) انظر: الفروع، للشمس ابن مفلح (٣/١٩٠).

(١٠) أخرجه أحمد (٦/٦٢-٦٣)، والبخاري (٩٠٣)، ومسلم (٨٤٧) واللفظ لهما.

(١١) أي عند قوله في كتاب الطهارة: «(وإن استعمل) قليل (في طهارة مستحبة: كتجديدِ وُضُوءٍ، وغُسْلِ جمعةٍ)» في (ص١٦).

(١٢) في (د، ض): «أن يتنظف ويتطيب»، وفي (ل): «تنظيف وتطييب».

(١٣) أخرجه أحمد (٥/٤٣٨)، والبخاري (٨٨٣) من حديث سلمان الفارسي ڔ .

وحديث أبي سعيد أخرجه أحمد (٣/٦٩)، والبخاري (٨٨٠)، ومسلم (٨٤٦) مرفوعًا ولفظه: «الغسل يوم الجمعة واجب على كل محتلم، وأن يستنَّ، وأن يمس طيبا إن وجد».

(١٤) جاء من حديث أبي سعيد الخدري وأبي هريرة ژ ، أخرجه أحمد (٣/٨١)، وأبو داود (٣٤٣)، وصححه ابن خزيمة (١٧٦٢)، وابن حبان (٢٧٧٨)، والحاكم (١/٢٨٣).

ومن حديث أبي ذر ڔ ، أخرجه أحمد (٥/١٧٧)، وابن ماجه (١٠٩٧)، وصححه ابن خزيمة (١٧٦٣)، والحاكم (١/٢٩٠)، واختلف في إسناده؛ فقيل عن سلمان الفارسي ڔ ، انظر: فتح الباري لابن حجر (٢/٣٧١).

ومن حديث أبي أيوب الأنصاري ڔ ، أخرجه أحمد (٥/٤٢٠)، وصححه ابن خزيمة (١٧٧٥).

ومن حديث عبدالله بن عمرو بن العاص ژ ، أخرجه أبو داود (٣٤٧)، وصححه ابن خزيمة (١٨١٠) ولفظه: «ولبس من صالح ثيابه» وبهذا اللفظ جاء في بعض طرق حديث أبي سعيد الخدري ڔ عند أحمد (٣/٦٥).

(١٥) انظر ما بعده.

(١٦) أخرجه أحمد (٤/٩-١٠) وأبو داود (٣٤٥) والترمذي (٤٩٦) وابن ماجه (١٠٨٧) والنسائي (٣/٩٥-٩٦) من حديث أوس بن أوس الثقفي ڔ ، وأكثر الروايات: «... عملُ سنةٍ، أجرُ صيامها وقيامها». صححه: ابن خزيمة (١٧٥٨) وابن حبان (٥٥٩) والحاكم (١/٢٨٢) وحسَّنه الترمذي والبغوي (١٠٦٥) والنووي في الخلاصة (٢٧١٧).

(١٧) في (ز): «بالصلاة على النبي».

(١٨) في (ز): «في يوم».

(١٩) أخرجه البيهقي (٣/٢٤٩)، والنسائي في السنن الكبرى (١٠٨٩٩-١٠٩٠١) بنحوه، والحاكم (٢/٣٦٨) وصححه، وردَّه الذهبي في تلخيصه.

ورُوي موقوفًا على أبي سعيد ڔ بنحوه، أخرجه النسائي في الكبرى، والبيهقي في الجامع لشعب الإيمان (٢٢٢٠)، قال النسائي: (وقفه أصح)، وكذا قال الذهبي في المهذب (٣/١١٨١)، وقال البيهقي: (وهذا هو المحفوظ).

(٢٠) أخرجه أحمد (٤/٨)، وأبو داود (١٠٤٧)، وابن ماجه (١٠٨٥)، والنسائي (٣/٩١-٩٢) من حديث حسين بن علي الجعفي عن عبد الرحمن بن يزيد بن جابر عن أوس بن أوس الثقفي ڔ .

قال ابن رجب في شرح علل الترمذي (٢/٦٨١-٦٨٤): (قالت طائفة: هو حديث منكر، وحسين الجعفي سمع من عبدالرحمن بن يزيد بن تميم الشامي، وروى عنه أحاديث منكرة فغلط في نسبته، وممن ذكر ذلك البخاري وأبو زرعة وأبو حاتم وأبو داود وابن حبان وغيرهم، وأنكر ذلك آخرون وقالوا: الذي سمع منه هو ابن جابر)، وانظر: جلاء الأفهام لابن القيم (ص١٢٦)١٣٢ ح٧٤).

(٢١) أخرجه أحمد (٤/١٨٨)، وأبو داود (١١١٨)، والنسائي (٣/١٠٣) من حديث عبد الله بن بسر المازني ڔ ، وأخرجه ابن ماجه (١١١٥) من حديث جابر بن عبد الله ڔ .

زاد أحمد وابن ماجه: (وآنيت)، صححه ابن خزيمة (١٨١١)، وابن حبان (٢٧٩٠)، وابن الملقن في البدر المنير (٤/٦٨٠) وغيرهم.

قال ابن حجر في التلخيص (٣/١٠٤٤): (وضعَّفه ابن حزم في المحلى (٥/٧٠) بما لا يقدح).

(٢٢) انظر: الغنية (٢/١٠٢).

(٢٣) أخرجه أحمد (٢/٢٢)، والبخاري (٩١١)، ومسلم (٢١٧٧).

(٢٤) نقله في: المبدع، للبرهان ابن مفلح (٢/١٧٥).

(٢٥) انظر: الشرح الكبير (٥/٢٩١).

(٢٦) نقله في: المبدع، للبرهان ابن مفلح (٢/١٧٦).

(٢٧) أخرجه أحمد (٢/٢٦٣)، ومسلم (٢١٧٩) من حديث أبي هريرة ڔ .

(٢٨) أخرجه أحمد (٣/٢٩٧، ٣١٦-٣١٧)، والبخاري (١١٦٦)، ومسلم (٨٧٥) من حديث جابر بن عبدالله ڔ .

(٢٩) أخرجه أحمد (١/٩٣)، وأبو داود (١٠٥١) من حديث عطاء الخرساني عن مولى امرأته عن علي بن أبي طالب ڔ .

(٣٠) أخرجه أحمد (٣/٢٩٧، ٣١٦-٣١٧)، والبخاري (١١٦٦)، ومسلم (٨٧٥) من حديث جابر بن عبدالله ڔ ، وفيه قال جابر: جاء سليك الغطفاني يوم الجمعة ورسول الله چ يخطب فجلس قبل أن يصلي فقال له النبي چ : «أركعت ركعتين؟» قال: لا. فقال: «قم فاركعهما».
