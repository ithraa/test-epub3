٤٣٣

(فصلٌ) فِي الصّلاةِ علَى الميّتِ

تسقطُ بمكلّفٍ.

وتُسنُّ:

  • جماعةً،
  • وأنْ لَا تنقصَ الصُّفوفُ عنْ ثلاثَةٍ.

و(السُّنَّةُ أنْ يقومَ الإمامُ:

  • عندَ صدرِهِ)؛ أيْ: صدرِ ذكرٍ،
  • (وعندَ وسَطِهَا)؛ أيْ: وسَطِ أنثَى،
  • والخنثَى بينَ ذلكَ.

والأَوْلَى بِهَا:

  • وصيُّهُ العدلُ،
  • فسيِّدٌ برقيقِهِ،
  • فالسُّلطانُ،
  • فنائبُهُ الأميرُ،
  • فالحاكمُ،
  • فالأَوْلَى بغَسلِ رجلٍ،
٤٣٤
  • فزوجٌ بعدَ ذوِي الأرحامِ.

ومنْ قدّمَهُ وليٌّ بمنزلتِهِ، لَا منْ قدّمَهُ وصيٌّ.

وإذَا اجتمعَتْ جنائزُ قُدِّمَ إلَى الإمامِ:

  • أفضلُهُمْ وتقدّمَ(١)،
  • فأسنُّ،
  • فأسبقُ،
  • ويُقرَعُ معَ التّساوِي.

وجمعُهُمْ بصلَاةٍ أفضلُ، ويُجعلُ وسَطُ أنثَى حذاءَ صدرِ ذكرٍ، وخنثَى بينَهُمَا.

(ويُكبِّرُ أربعًا)؛ «لتكبيرِ النَّبِيِّ چ علَى النّجاشِيِّ أربعًا»، متّفقٌ عليهِ(٢):

  • (يقرأُ فِي الأُولَى)؛ أيْ بعدَ التَّكبيرَةِ الأُولَى وهيَ تكبيرَةُ الإحرامِ (بعدَ التّعوُّذِ) والبسملَةِ: (الفاتحَةَ) سرًّا ولوْ ليلًا؛ لمَا روَى ابنُ ماجهْ عنْ أمِّ شريكٍ الأنصاريَّةِ ږ قالتْ: «أمرَنَا رسولُ اللهِ چ أنْ نقرأَ علَى الجنازَةِ بفاتحَةِ الكتابِ»(٣)،
٤٣٥
  • ولَا يستفتحُ ولَا يقرأُ سورَةً معهَا.
  • (ويصلِّي علَى النَّبِيِّ چ فِي)؛ أيْ بعدَ التّكبيرَةِ (الثّانيَةِ كـ)ـالصّلَاةِ فِي (التّشهُّدِ) الأخيرِ؛ لمَا روَى الشّافعيُّ عنْ أبِي أمامَةَ بنِ سهلٍ أنَّهُ أخبرَهُ رجلٌ منْ أصحابِ النَّبِيِّ چ : «أنَّ السُّنَّةَ فِي الصّلاةِ علَى الجنازَةِ أنْ يكبِّرَ الإمامُ ثمَّ يقرأَ بفاتحَةِ الكتابِ بعدَ التّكبيرَةِ الأُولَى سرًّا فِي نفسِهِ، ثمَّ يصلِّيَ علَى النَّبِيِّ چ ، ويُخلصَ الدّعاءَ للميِّتِ، ثمَّ يسلِّمَ»(٤).
  • (ويدعُو فِي الثّالثَةِ)؛ لمَا تقدَّمَ، (فيقولُ:
  • «اللّهمَّ اغفرْ لحيِّنَا وميِّتِنَا، وشاهدِنَا وغائبِنَا، وصغيرِنَا وكبيرِنَا، وذكرِنَا وأنثانَا. إنَّكَ تعلمُ متقلّبَنَا ومثوانَا وأنْتَ علَى كلِّ شيءٍ قديرٌ، اللّهمَّ منْ أحييتَهُ منّا فأحيِهِ علَى الإسلامِ والسُّنَّةِ، ومنْ توفّيتَهُ منّا فتوفَّهُ عَلَيْهِمَا») رواهُ أحمدُ والتّرمذيُّ وابنُ ماجهْ منْ حديثِ أبِي هريرةَ ڔ (٥)، لكنْ زادَ فيهِ الموفَّقُ: «وأنْتَ
٤٣٦
  • علَى كلِّ شيءٍ قديرٌ»، ولفظَةَ: «السُّنَّةِ»(٦)،
  • («اللّهمَّ اغفرْ لهُ وارحمْهُ، وعافِهِ واعفُ عنهُ، وأكرِمْ نزُلَهُ») بضمِّ الزّايِ وقدْ تُسكَّنُ وهوَ: القِرَى، («وأوسِعْ مدخلَهُ») بفتحِ الميمِ: مكانُ الدّخولِ، وبضمِّهَا: الإدخالِ، («واغسلْهُ بالماءِ والثّلجِ والبرَدِ، ونقِّهِ مِنَ الذُّنوبِ والخطايَا كمَا يُنقَّى الثّوبُ الأبيضُ مِنَ الدّنسِ، وأبدِلْهُ دارًا خيرًا منْ دارِهِ، وزوجًا خيرًا منْ زوجِهِ، وأدخِلْهُ الجنَّةَ، وأعِذْهُ منْ عذابِ القبرِ وعذابِ النّارِ») رواهُ مسلمٌ عنْ عوفِ بنِ مالكٍ ڔ أنَّهُ سمِعَ النَّبِيَّ چ يقولُ ذلكَ علَى جنازَةٍ حتَّى تمنَّى أنْ يكونَ ذلكَ الميِّتَ، وَفِيهِ: «وأبدِلْهُ أهلًا خيرًا منْ أهلِهِ وأدخِلْهُ الجنَّةَ». وزادَ الموفَّقُ لفظَ: «منَ الذُّنوبِ»(٧)،
٤٣٧
  • (وافسحْ لهُ فِي قبرِهِ ونوِّرْ لهُ فيهِ)؛ لأنَّهُ لائقٌ بالمحَلِّ،
  • وإنْ كانَ الميِّتُ أنثَى أنّثَ الضّميرَ، وإنْ كانَ خنثَى قالَ: هذَا الميِّتَ ونحوَهُ،
  • ولَا بأسَ بالإشارَةِ بالإصبَعِ حالَ الدّعاءِ للميِّتِ.
  • (وإنْ كانَ) الميِّتُ (صغيرًا) -ذكرًا أوْ أنثَى- أوْ بلغَ مجنونًا واستمرَّ (قالَ) بعدَ: «ومنْ توفّيتَهُ منَّا فتوفَّهُ عَلَيْهِمَا»، (اللّهمَّ اجعلْهُ ذخرًا لوالديْهِ وفَرَطًا)؛ أيْ: سابقًا مهيِّئًا لمصالحِ أبويْهِ فِي الآخرَةِ سواءٌ ماتَ فِي حيَاةِ أبويْهِ أوْ بعدَهُمَا، (وشفيعًا(٨) مجابًا. اللّهمَّ ثقِّلْ بهِ موازينَهمَا، وأعظِمْ بهِ أجورَهمَا، وألحِقْهُ بصالحِ سلفِ المؤمنِينَ، واجعلْهُ فِي كفالَةِ إبراهيمَ، وقِهِ برحمتِكَ عذابَ الجحيمِ)، ولَا يستغفرُ لهُ؛ لأنَّهُ شافعٌ غيرُ مشفوعٍ فيهِ، ولَا جرَى عليهِ قلمٌ، وإذَا لمْ يُعرفْ إسلامُ والديْهِ دعَا لموالِيهِ.
  • (ويقفُ بعدَ الرّابعَةِ قليلًا) ولَا يدعُو ولَا يتشهّدُ ولَا يسبِّحُ،
  • (ويسلِّمُ) تسليمةً (واحدَةً عنْ يمينِهِ)؛ روَى الجُوزَجانِيُّ عنْ عطاءِ بنِ السّائبِ: «أنَّ النَّبِيَّ چ سلّمَ علَى الجنازَةِ تسليمَةً»(٩)،
٤٣٨
  • ويجوزُ:
  • تلقاءَ وجهِهِ،
  • وثانيَةً.

وسُنَّ وقوفُهُ حتَّى تُرفعَ.

(ويرفعُ يديْهِ) ندبًا (معَ كلِّ تكبيرَةٍ)؛ لمَا تقدَّمَ فِي صلَاةِ العيدَيْنِ(١٠).

(وواجبُهَا)؛ أيِ: الواجبُ فِي صلَاةِ الجنازَةِ ممّا تقدَّمَ:

  • (قيامٌ) فِي فرضِهَا،
  • (وتكبيراتٌ) أربعٌ،
  • (والفاتحَةُ) ويتحمّلُهَا الإمامُ عنِ المأمومِ،
  • (والصّلَاةُ علَى النَّبِيِّ چ ،
  • ودعوَةٌ للميِّتِ،
  • والسّلامُ).
٤٣٩

ويُشترطُ لهَا:

  • النِّيَّةُ؛ فينوِي الصّلاةَ علَى الميِّتِ.
  • ولَا يضرُّ جهلُهُ بالذَّكَرِ وغيرِهِ، فإنْ جهِلَهُ: نوَى علَى مَنْ يصلِّي عليهِ الإمامُ،
  • وإنْ نوَى أحدَ الموتَى: اعتُبِرَ تعيينُهُ،
  • وإنْ نوَى علَى هذَا الرّجلِ فبانَ امرأَةً أوْ بالعكسِ: أجزأَ؛ لقوَّةِ التّعيينِ، قالَهُ أبُو المعالِي(١١).
  • وإسلامُ الميِّتِ،
  • وطهارتُهُ مِنَ الحدثِ والنّجِسِ معَ القدرَةِ وإلَّا صُلِّيَ عليهِ.
  • والاستقبالُ، والسُّترَةُ؛ كمكتوبَةٍ.
  • وحضورُ الميِّتِ بينَ يديْهِ، فلَا تصحُّ علَى جنازَةٍ محمولَةٍ ولَا مِنْ وراءِ جدارٍ.

(ومنْ فاتَهُ شيءٌ مِنَ التّكبيرِ: قضاهُ) ندبًا (علَى صفتِهِ)؛ لأنَّ القضاءَ يحكِي الأداءَ كسائرِ الصلواتِ،

  • والمقضيُّ: أوّلُ صلاتِهِ يأتِي فيهِ بحسَبِ ذلكَ.
  • وإنْ خشِيَ رفْعَهَا: تابعَ التّكبيرَ رُفعَتْ أمْ لَا.
٤٤٠
  • وإنْ سلّمَ معَ الإمامِ ولمْ يقضِهِ: صحّتْ؛ لقولِهِ چ لعائشَةَ ږ : «مَا فاتَكِ لَا قضاءَ عليكِ»(١٢).

(ومنْ فاتتْهُ الصّلَاةُ عليهِ)؛ أيْ علَى الميِّتِ: (صلَّى علَى القبرِ) إلَى شهرٍ منْ دفنِهِ؛

  • لمَا فِي الصّحيحَيْنِ منْ حديثِ أبِي هريرةَ وابنِ عبّاسٍ ښ : «أنَّ النَّبِيَّ چ صلَّى علَى قبرٍ»(١٣).
  • وعنْ سعيدِ بنِ المسيِّبِ: «أنَّ أمَّ سعدٍ ماتَتْ والنَّبِيُّ چ غائبٌ؛ فلمّا قدِمَ صلَّى عَلَيْهَا وقدْ مضَى لذلكَ شهرٌ»، رواهُ التّرمذِيُّ ورواتُهُ ثقاتٌ(١٤). قالَ أحمدُ: أكثرُ مَا سمِعْتُ هذَا(١٥)،
٤٤١
  • وتحرُمُ بعدَهُ، مَا لمْ تكنْ زيادَةً يسيرَةً.

(و) يُصلِّي (علَى غائبٍ) عنِ البلدِ ولوْ دونَ مسافَةِ قصرٍ، فيجوزُ صلَاةُ الإمامِ والآحادِ عليهِ (بالنِّيَّةِ إلَى شهرٍ)؛ «لصلاتِهِ چ علَى النّجاشِيِّ»، كمَا فِي المتّفقِ عليهِ عنْ جابرٍ ڔ (١٦).

وكذَا:

  • غريقٌ،
  • وأسيرٌ؛ ونحوُهمَا.

وإنْ وُجدَ بعضُ ميِّتٍ لمْ يصلَّ عليهِ:

  • فَكَكُلِّهِ،
  • إلَّا:
  • الشّعرَ،
  • والظُّفرَ،
  • والسِّنَّ.
  • فيُغسَّلُ، ويُكفَّنُ، ويُصلَّى عليهِ،
  • ثمَّ إنْ وُجدَ الباقِي: فكذلكَ،
  • ويُدفنُ بجنبِهِ.

ولَا يُصلَّى علَى:

  • مأكولٍ ببطنِ آكلٍ،
٤٤٢
  • ولَا مستحيلٍ بإحراقٍ ونحوِهِ،
  • ولَا علَى بعضِ حيٍّ مدَّةَ حياتِهِ.

(ولَا) يُسنُّ أنْ (يصلِّيَ الإمامُ) الأعظمُ، ولَا إمامُ كلِّ قريَةٍ وهوَ: واليهَا فِي القضاءِ:

  • (علَى الغالِّ) وهوَ: مَنْ كتمَ شيئًا ممّا غَنِمَهُ؛ لمَا روَى زيدُ بنُ خالدٍ ڔ قالَ: تُوفِّيَ رجلٌ منْ جهينَةَ يومَ خيبرَ، فذُكرَ ذلكَ لرسولِ اللهِ چ فقالَ: «صلُّوا علَى صاحبِكُمْ». فتغيّرَتْ وجوهُ القومِ، فلمّا رأَى مَا بهِمْ قالَ: «إنَّ صاحبَكُمْ غلَّ فِي سبيلِ اللهِ»؛ ففتّشْنَا متاعَهُ فوجدْنَا فيهِ خَرَزًا منْ خَرَزِ اليهودِ مَا يساوِي درهمَيْنِ. رواهُ الخمسَةُ إلَّا التّرمذِيَّ، واحتجَّ بهِ أحمدُ(١٧).
  • (ولَا علَى قاتلِ نفسِهِ) عمدًا؛ لمَا روَى جابرُ بنُ سَمُرَةَ ڔ : «أنَّ النَّبِيَّ چ جاؤُوهُ برجلٍ قدْ قتلَ نفسَهُ بِمَشَاقِصَ فلمْ يصلِّ عليهِ» رواهُ مسلمٌ وغيرُهُ(١٨).
  • والمشَاقِصُ جمعُ مِشْقَصٍ كمِنْبَرٍ: نصلٌ عريضٌ أوْ سهمٌ فيهِ ذلكَ، أوْ نصلٌ طويلٌ أوْ سهمٌ فيهِ ذلكَ يُرمَى بهِ الوحشُ.
٤٤٣

(ولَا بأسَ بالصّلاةِ عليهِ)؛ أيْ علَى الميِّتِ (فِي المسجدِ) إنْ أُمنَ تلويثُهُ؛

  • لقولِ عائشةَ ږ : «صلَّى رسولُ اللهِ چ علَى سهيلِ بنِ بيضاءَ فِي المسجدِ»، رواهُ مسلمٌ(١٩)،
  • وصُلِّيَ علَى أبِي بكرٍ وعمرَ ښ فيهِ، رواهُ سعيدٌ(٢٠).

وللمصلِّي: قيراطٌ وهوَ أمرٌ معلومٌ عندَ اللهِ تعالَى،

ولهُ بتمامِ دفنِهَا: آخَرُ، بشرطِ أنْ لَا يُفارقَهَا مِنَ الصّلَاةِ حتَّى تُدفنَ.

ڭ ڭ ڭ

(١) أي عند قوله: «(ويليهِ)؛ أيْ: الإمامَ مِنَ المأمومِين..» في (ص٣٠٨).

(٢) أخرجه أحمد (٣/٣٦١)، والبخاري (١٣٣٤)، ومسلم (٩٥٢) من حديث جابر بن عبدالله ژ .

(٣) أخرجه ابن ماجه (١٤٩٦).

أشار النووي إلى ضعفه في الخلاصة (٣٤٨٦)، وحسَّنه البوصيري في مصباح الزجاجة (٥٣٨)، وقال ابن حجر في التلخيص (٣/١٢٠١): (وفي إسناده ضعف يسير).

(٤) أخرجه الشافعي (مسنده ١/٢١٠)، والحاكم (١/٣٦٠) بنحوه عن أبي أمامة بن سهل عن رجل من أصحاب رسول الله چ ، وأخرجه النسائي (٤/٧٥)، وعبدالرزاق (٣/٤٨٩)، وابن أبي شيبة (٣/٢٩٦)، وابن الجارود في المنتقى (٥٤٧) عن أبي أمامة قوله: السنة في الصلاة ...إلخ. قال الدارقطني في العلل (س٢٦٨٩): (الصواب قول من رواه عن الزهري عن أبي أمامة بن سهل).

(٥) أخرجه أحمد (٢/٣٦٨)، وأبو داود (٣٢٠١)، والترمذي (١٠٢٤)، وابن ماجه (١٤٩٨) من حديث يحيى بن أبي كثير عن أبي سلمة عن أبي هريرة، وأخرجه الترمذي وأحمد (٤/١٧٠) من حديث يحيى عن أبي إبراهيم الأشهلي عن أبيه مرفوعًا، وليس في شيء من طرق الحديث قوله: (إنك تعلم متقلبنا ومثوانا وأنت على كل شيء قدير)، وعند جميعهم لفظه: «اللهم من أحييته منا فأحيه على الإسلام ومن توفيته منا فتوفه على الإيمان».

صححه الترمذي وابن حبان (٣٠٧٠)، والحاكم (١/٣٥٨).

اختُلفَ في هذا الحديث في وصله وإرساله وتعيين صحابيه؛ صححه مرفوعًا من تقدم، ورجَّح أبو حاتم كما في العلل لابنه (س١٠٤٧، ١٠٥٨) والدارقطني في العلل (س١٧٩٤) إرساله، وقال البخاري فيما نقله الترمذي والدارقطني: (أصح الروايات في هذا حديث يحيى بن أبي كثير عن أبي إبراهيم الأشهلي عن أبيه).

(٦) انظر: المقنع (٦/١٥٠).

(٧) أخرجه مسلم (٩٦٣)، وأما الزيادة التي نسبها للموفق فهي في المقنع (٦/١٥١) وأخرجها ابن ماجه (١٥٠٠).

(٨) في (د، ز): «وأجرًا وشفيعًا»، وفي (ز) «وأجرًا» من الشرح.

(٩) أخرجه أبو داود في المراسيل (٤١٨) من طريق أبي إسحاق الفزاري عن عطاء به مرسلًا، وأخرجه أبو داود في مسائله (١٠٣١)، وأبو نعيم في الحلية (٩/٥٧) عن عطاء عن عبدالله بن أبي أوفى موقوفًا عليه من فعله.

قال أحمد كما في زاد المعاد (١/٤٩٢): عن ستة من الصحابة أنهم كانوا يسلِّمون تسليمة واحدة خفيفة عن يمينه، فذكر ابن عمر وابن عباس وأبا هريرة وواثلة بن الأسقع وابن أبي أوفى وزيد بن ثابت.

(١٠) أي عند قوله: «وعنْ عمرَ ڔ : أنَّهُ كانَ يرفعُ يديْهِ في كلِّ تكبيرَةٍ فِي الجنازَةِ والعيدِ» في (ص٣٧٨).

(١١) نقله في: المبدع، للبرهان ابن مفلح (٢/٢٥٦).

(١٢) أورده ابن الجوزي في التحقيق (٢/٦٦١ مع التنقيح) غير مسند، وأخرجه بنحوه ابن شاهين في ناسخ الحديث ومنسوخه (ح ٢٩٣) عن عائشة قالت: يا رسول الله إنك تصلي على الجنازة وأنا في بيتي يخفى عليَّ كثير من التكبير فقال النبي چ : «لا عدد، ما فهمت فكبري» وفي إسناده الحكم بن عبدالله بن سعد، قال أبو حاتم عنه في الجرح والتعديل (١/١٢/١٢١): (كان يكذب).

(١٣) أخرجه أحمد (٢/٣٥٣)، والبخاري (١٣٣٧)، ومسلم (٩٥٦) من حديث أبي هريرة ڔ ، وأخرجه أحمد (١/٢٢٤) والبخاري (١٣١٩) ومسلم (٩٥٤) من حديث ابن عباس ژ .

(١٤) أخرجه الترمذي (١٠٣٨) والبيهقي (٤/٤٩) من طريقين عن قتادة عن سعيد بن المسيِّب به.

قال البيهقي: (وهو مرسل صحيح)، وكذا قال ابن عبدالهادي في تنقيح التحقيق (٢/٦٦٤).

(١٥) انظر: مسائل الكوسج (٤٥٧)، ومسائل أبي داود (١٠٤٧).

(١٦) سبق تخريجه في (ص٤٣٤).

(١٧) أخرجه أحمد (٤/١١٤)، وأبو داود (٢٧١٠)، وابن ماجه (٢٨٤٨)، والنسائي (٤/٦٤).

صححه ابن حبان (٤٨٥٣)، والحاكم (٢/١٢٧)، وأبو نعيم في الحلية (٨/٢٦٢) وقال: (صحيح متفق عليه من حديث يحيى بن سعيد رواه الناس عنه)، وانظر احتجاج أحمد به في: زاد المسافر (٢/٢٨٣).

(١٨) أخرجه أحمد (٥/٨٧)، ومسلم (٩٧٨).

(١٩) أخرجه أحمد (٦/٧٩)، ومسلم (٩٧٣)

(٢٠) أما الصلاة على أبي بكر فأخرج عبدالرزاق (٣/٥٢٦)، وابن أبي شيبة (٣/٣٦٤)، والبيهقي (٤/٥٢) من حديث عروة بن الزبير قال: (ما صُلِّيَ على أبي بكر إلا في المسجد)، وصححه ابن حزم في المحلَّى (٥/١٦٣).

وأما عمر فأخرج مالك في الموطأ (٦١٥) وعنه عبدالرزاق (٣/٥٢٦)، وابن أبي شيبة (٣/٣٦٤)، والبيهقي (٤/٥٢) عن ابن عمر قال: (صُلِّيَ على عمر في المسجد). صححه الحاكم (٣/٩٩)، ابن حزم في المحلى (٥/١٦٣).
