١٦١٧

(بابُ الذّكَاةِ)

يُقالُ: ذكَّى الشّاةَ ونحوَهَا تذكيةً؛ أيْ: ذَبَحَهَا.

فهِيَ:

  • ذبحُ، أوْ نحرُ، الحيوانِ المأكولِ البرِّيِّ بقطعِ حُلْقُومِهِ ومريئِهِ،
  • أوْ عَقْرُ ممتنِعٍ.

و (لَا يُباحُ شيءٌ مِنَ الحيوانِ المقدورِ عليهِ بغيرِ ذكاةٍ)؛ لأنَّ غيرَ المُذكَّى مَيتةٌ، وقالَ تعالَى: {حُرِّمَتۡ عَلَيۡكُمُ ٱلۡمَيۡتَةُ} [المائدة:٣]،

  • (إلَّا:
  • الجرادُ،
  • والسّمكُ،
  • وكلُّ مَا لَا يعيشُ إلَّا فِي الماءِ)؛
  • فيحِلُّ بدونِ ذكاةٍ؛ لحلِّ مَيتتِهِ؛ لحديثِ ابنِ عمرَ ژ يرفعُهُ: «أُحلَّ لنَا ميتتانِ ودمانِ، فأمَّا المَيتتانِ: الحوتُ والجرادُ، وأمَّا الدَّمانِ: فالكبِدُ والطِّحالُ»، رواهُ أحمدُ وغيرُهُ(١).
١٦١٨

ومَا يعيشُ فِي البرِّ والبحرِ؛ كالسُّلَحْفَاةِ وكلبِ الماءِ: لَا يحِلُّ إلَّا بالذّكَاةِ.

وحَرُمَ بلعُ سمكٍ حيًّا.

وكُرِهَ شَيُّهُ حيًّا،

  • لَا جرادٍ؛ لأنَّهُ لَا دمَ لهُ.

ڭ ڭ ڭ

(ويُشتَرَطُ للذَّكاةِ أربعةُ شُرُوطٍ)،

أَحَدُهَا: (أهليَّةُ المذكِّي؛ بأنْ يكونَ:

  • عاقلًا)، فلَا يُباحُ مَا ذكَّاهُ مجنونٌ، أوْ سكرانُ، أوْ طفلٌ لمْ يميِّزْ؛ لأنَّهُ لَا يصحُّ مِنهُ قصدُ التّذكيةِ،
  • (مُسلمًا) كانَ (أوْ كتابيًّا) أبواهُ كتابيّانِ؛ لقولِهِ تعالَى: {وَطَعَامُ ٱلَّذِينَ أُوتُواْ ٱلۡكِتَٰبَ حِلّٞ لَّكُمۡ} [المائدة:٥]، قالَ البخاريُّ: قالَ ابنُ عبّاسٍ ژ : طعامُهُمْ ذبائِحُهُمْ(٢)،
  • (ولو(٣)) كانَ المذكِّي مميِّزًا، أوْ (مراهِقًا، أوِ امرأةً، أوْ أَقْلَفَ)
١٦١٩
  • لمْ يُختنْ(٤) ولوْ بلَا عُذرٍ، (أوْ أعمَى)، أوْ حائضًا أوْ جُنُبًا.

(ولَا تُباحُ ذكاةُ:

  • سكرانَ،
  • ومجنونٍ)؛
  • لمَا تقدَّمَ.

(و) لَا ذكاةُ: (وثَنيٍّ،

  • ومجُوسيٍّ،
  • ومُرتدٍّ)؛
  • لمفهومِ قولِهِ تعالَى: {وَطَعَامُ ٱلَّذِينَ أُوتُواْ ٱلۡكِتَٰبَ حِلّٞ لَّكُمۡ} [المائدة:٥].

ڭ ڭ ڭ

الشّرطُ (الثّاني: الآلَةُ، فتُباحُ الذَّكاةُ بكلِّ مُحَدَّدٍ) يَنْهَرُ الدّمَ بِحَدِّهِ، (ولوْ) كانَ (مغصوبًا، مِنْ حديدٍ وحجَرٍ وقصَبٍ وغيرِهِ)؛ كخشبٍ لهُ حَدٌّ وذهبٍ وفضَّةٍ وعَظْمٍ،

  • (إلَّا:
  • السِّنَّ،
  • والظُّفرَ)؛
  • لقولِهِ چ : «ومَا(٥) أنهرَ الدّمَ فَكُلْ، ليسَ السِّنَّ والظُّفرَ»، متَّفقٌ عليهِ(٦).

ڭ ڭ ڭ

١٦٢٠

الشّرطُ (الثَّالثُ: قطعُ:

  • الحُلْقُومِ)، وهوَ مجرَى النّفَسِ،
  • (و) قَطعُ (المَرِيءِ) بالمدِّ، وهوَ مجرَى الطّعامِ والشّرابِ،
  • ولَا يُشترطُ:
  • إبانتُهُمَا،
  • ولَا قَطعُ الوَدْجَينِ،
  • ولَا يضرُّ رفعُ يدِ الذابحِ إنْ أتمَّ الذَّكاةَ علَى الفورِ.

والسُّنَّةُ:

  • نحرُ إبِلٍ بطَعنٍ بمحدَّدٍ فِي لَبَّتِهَا،
  • وذَبحُ غيرِهَا(٧).

(وذكاةُ مَا عَجِزَ عنهُ مِنَ:

  • الصّيدِ والنَّعَمِ المتوحِّشَةِ و) النَّعَمِ (الواقعَةِ فِي بئرٍ ونحوِهَا:
  • بجَرْحِهِ فِي أيِّ موضعٍ كانَ مِنْ بدنِهِ)؛ رُوِيَ عنْ عليٍّ وابنِ مسعودٍ وابنِ عمرَ وابنِ عباسٍ وعائشةَ ښ (٨)،
١٦٢١
  • (إلَّا أنْ يكونَ رأسُهُ فِي الماءِ ونحوِهِ) ممّا يقتُلُهُ لوِ انفردَ؛
  • (فلَا يُباحُ) أكلُهُ؛ لحصولِ قتلِهِ بمُبيحٍ وحاظِرٍ، فغُلِّبَ جانبُ الحَظْرِ.

ومَا ذُبِحَ مِنْ قفاهُ ولوْ عمدًا إنْ أتَتِ الآلَةُ علَى محلِّ ذبحِهِ وَفِيهِ حياةٌ مستقِرَّةٌ: حلَّ،

  • وإلَّا فلَا.
  • ولوْ أبانَ رأسَهُ: حلَّ مطلقًا(٩).

والنَّطيحةُ ونحوُهَا،

  • إنْ ذكّاهَا وحياتُهَا تُمْكِنُ زيادتُهَا علَى حركَةِ مذبوحٍ: حَلّتْ،
  • والاحتياطُ معَ تحرُّكٍ ولوْ بيدٍ أوْ رجلٍ،
  • ومَا قُطِعَ حلقومُهُ، أوْ أُبِينَتْ حَشْوَتُهُ: فوجودُ حياتِهِ كعدمِهَا.

ڭ ڭ ڭ

الشّرطُ (الرّابعُ: أنْ يقولَ) الذّابحُ (عندَ) حركَةِ يدِهِ بـ(ـالذَّبحِ: بسمِ الله)؛ لقولِهِ تعالَى: {وَلَا تَأۡكُلُواْ مِمَّا لَمۡ يُذۡكَرِ ٱسۡمُ ٱللَّهِ عَلَيۡهِ وَإِنَّهُۥ لَفِسۡقٞ}(١٠) [الأنعام:١٢١]،

١٦٢٢
  • (لَا يجزئُهُ غيرُهَا)؛ كقولِ بِاسمِ الخالقِ ونحوِهِ؛ لأنَّ إطلاقَ التّسميَةِ ينصرفُ إلَى بسمِ الله.

وتجزئُ بغيرِ عربيَّةٍ، ولوْ أحسنَهَا.(فإنْ تركَهَا)؛ أيِ: التّسميةَ (سهوًا: أُبيحَتِ) الذّبيحةُ؛ لقولِهِ چ : «ذبيحَةُ المسلمِ حلالٌ، وإنْ لمْ يسمِّ إذَا لمْ يتعمّدْ»، رواهُ سعيدٌ(١١)،

  • (لَا) إنْ تركَ التّسميَةَ (عمدًا) ولوْ جهلًا: فلَا تحلُّ الذّبيحةُ؛ لمَا تقدَّمَ.

ومَنْ بدَا لهُ ذبحُ غيرِ مَا سمَّى عليهِ: أعادَ التَّسميَةَ.

ويُسَنُّ معَ التَّسميةِ: التَّكبيرُ،

  • لَا الصَّلَاةُ علَى النَّبِيِّ چ .

وَمَنْ ذكرَ معَ اسمِ اللهِ اسمَ غيرِهِ:

  • حَرُمَ؛
  • ولمْ يَحِلَّ المذبوحُ.

ڭ ڭ ڭ

(ويُكرَهُ:

  • أنْ يذبحَ بآلةٍ كالَّةٍ)؛ لحديثِ: «إنَّ اللهَ كتبَ الإحسانَ علَى كلِّ
١٦٢٣
  • شيءٍ، فإذَا قتلْتُمْ فأحسِنُوا القِتْلَةَ، وإذَا ذبحْتُمْ فأحسِنُوا الذِّبْحَةَ، وليُحِدَّ أحدُكُمْ شفرتَهُ وليُرِحْ ذَبِيحَتَهُ»، رواهُ الشّافعيُّ وغيرُهُ(١٢).
  • (و) يُكرَهُ أيضًا: (أنْ يُحِدَّهَا)؛ أيِ: الآلَةَ (والحَيْوانُ يُبْصِرُهُ)؛ لقولِ ابنِ عمرَ ژ : «إنَّ رسولَ اللهِ چ أمرَ أنْ تُحَدَّ الشِّفَارُ وأنْ تُوَارَى عنِ البَهَائِمِ»، رواهُ أحمدُ وغيرُهُ(١٣)،
  • (و) يُكرَهُ أيضًا: (أنْ يوجِّهَهُ)؛ أيِ: الحَيْوانَ (إلَى غيرِ القبلَةِ)؛
  • لأنَّ السُّنَّةَ:
  • توجيهُهُ إلَى القبلَةِ علَى شِقِّهِ الأيسرِ،
  • والرِّفقُ بهِ،
  • والحَمْلُ علَى الآلَةِ بقوَّةٍ.
  • (و) يُكرَهُ أيضًا: (أنْ يَكْسِرَ عُنُقَهُ)؛ أيْ: عُنُقَ مَا ذَبَحَ، (أوْ يَسْلُخَهُ قبلَ أنْ يَبْرُدَ)؛ أيْ: قبلَ زُهُوقِ نَفْسِهِ؛
  • لحديثِ أبِي هريرةَ ڔ : «بعثَ رسولُ اللهِ چ بُدَيْلَ بنَ وَرْقَاءَ
١٦٢٤
  • الخُزاعِيَّ علَى جملٍ أَوْرَقَ يصيحُ فِي فِجاجِ منى بكلماتٍ مِنْهَا: لَا تُعْجِلُوا الأنفسَ قبلَ أنْ تَزْهَقَ»، رواهُ الدّارقطنِيُّ(١٤).

وإنْ ذبحَ كتابيٌّ مَا يَحْرُمُ عليهِ: حلَّ لنَا إنْ ذُكِرَ اسمُ اللهِ عليهِ.

وذكاةُ جنينٍ مباحٍ بذكاةِ أُمِّهِ إنْ:

  • خرجَ مَيِّتًا،
  • أوْ متحرِّكًا كمذبوحٍ.

ڭ ڭ ڭ

(١) أخرجه أحمد (٢/٩٧)، وابن ماجه (٣٣١٤) من حديث عبدالرحمن بن زيد بن أسلم عن أبيه عن ابن عمر ژ به مرفوعًا.

قال الإمام أحمد في العلل لعبدالله (٢/١٣٦ برقم ١٧٩٥): (حديث منكر)، ورواه سليمان بن بلال عن زيد عن ابن عمر ژ من قوله، أخرجه البيهقي (١/٢٥٤)، ورجَّح وقفه أبو زرعة في العلل لابن أبي حاتم (س١٥٢٤)، والدارقطني في العلل (س٢٢٧٧)، والبيهقي، وابن عبدالهادي في تنقيح التحقيق (٤/٦٤٣) وغيرهم، قال ابن القيِّم في زاد المعاد (٣/٣٤٥): (وهذا الموقوف في حكم المرفوع).

(٢) علَّقه البخاري (٧/٩٣) ووصله الطبري (تفسيره ٨/١٣٦) والبيهقي (٩/٢٨٢).

(٣) في (الأصل، س، ع): «لو» من الشرح، والمثبت هو الموافق لما في زاد المستقنع (ص٤٣٤ت: القاسم).

(٤) في (ز، س، ض، م): «يختتن»، وفي (ز) من المتن.

(٥) في (ب، د، ز، ل): «ما» بدون الواو.

(٦) أخرجه أحمد (٣/٤٦٣)، والبخاري (٢٤٨٨)، ومسلم (١٩٦٨) من حديث رافع بن خديج ڔ .

(٧) في (د، ل) بزيادة: «فإن أبان الرأس بالذبح: لم يحرم المذبوح».

(٨) علَّقه عنهم البخاري (٧/٩٣)، ووصله عنهم -دون عائشة- عبدالرزاق (٤/٤٦٤ - ٤٦٦)، وابن أبي شيبة (٥/٣٨٥ - ٣٨٦ و٣٩٤)، والبيهقي (٩/٢٤٥ - ٢٤٧).

وأما أثر عائشة فلم نقف على من أخرجه، وكذا قال ابن حجر في فتح الباري (٩/٦٣٩).

(٩) في (ب): «حَلًَ مطلقاً وكُره»، ونقل في المبدع (٨/٢٧) القول بالكراهة عن صاحب الرعاية، فقال: (إذا أبان رأسه بالذبح: لم يحرم به المذبوح؛ قدمه في المحرر، وأكلُهُ مباح؛ قاله في المستوعب، وفي الرعاية: يكره ويحل...).

(١٠) في (ز): «أي: حرام».

(١١) أخرجه سعيد بن منصور كما في المحلى (٧/٤١٣)، والحارث بن أسامة في مسنده كما المطالب العالية (٢٣١٧) من مرسل راشد بن سعد.

ورُوي عن ابن عباس k بنحوه مرفوعًا وموقوفًا؛ أخرجه موقوفًا عبد الرزاق (٤/٤٧٩)، وسعيد بن منصور في التفسير (٩١٤)، والدارقطني (٤٨٠٦)، والبيهقي في معرفة السنن والآثار (١٣/٤٤٧)، وقال: (وهو المحفوظ)، وصحح وقفَهُ ابن عبدالهادي في تنقيح التحقيق (٤/٦٣٧)، وابن حجر في فتح الباري (٩/٦٢٤).

(١٢) وأخرجه أحمد (٤/١٢٣)، ومسلم (١٩٥٥) من حديث شداد بن أوس ڔ ، وأخرجه البيهقي في معرفة السنن والآثار (١٤/٤٠) من طريق الشافعي به.

(١٣) أخرجه أحمد (٢/١٠٨)، وابن ماجه (٣١٧٢) عن الزهري عن سالم بن عبدالله بن عمر عن أبيه.

وأخرجه البيهقي (٩/٢٨٠) عن الزهري عن ابن عمر به، بدون ذكر سالم بن عبدالله، وبذلك أعلَّهُ أبو حاتم في العلل لابنه (س ١٦١٧)، وعبدالحق في الأحكام الوسطى (٤/١٣١)، وقال ابن حجر في الدراية (٢/٢٠٨): (صوَّب الحفَّاظ إرساله).

(١٤) أخرجه الدارقطني (٤٧٥٤).

وضعَّفه البيهقي (٩/٢٧٨)، وابن عبدالهادي في تنقيح التحقيق (٤/٦٤٠)، وابن كثير في تفسيره (سورة المائدة/٤) وقال: (ورُوي عن عمر موقوفًا وهو أصح: «ألا إن الذكاة في الحلق واللبَّة، ولا تعجلوا الأنفس أن تزهق»).

والموقوف أخرجه عبدالرزاق (٤/٤٩٥)، وابن أبي شيبة (٥/٣٩٢)، والبيهقي (٩/٢٧٨).
