٦٢٧

(بابُ جزاءِ الصّيدِ)

أيْ: مِثلُهُ فِي الجملَةِ إنْ كانَ، وإلَّا فقيمتُهُ.

فيجبُ المِثلُ مِنَ النَّعَمِ فيمَا لهُ مِثلٌ؛

  • لقولِهِ تعالَى: {فَجَزَآءٞ مِّثۡلُ مَا قَتَلَ مِنَ ٱلنَّعَمِ} [المائدة:٩٥]،
  • «وجعَلَ النَّبِيُّ چ فِي الضّبُعِ كبشًا»(١).

ويُرجَعُ فيمَا قضَتْ فيه الصّحابَةُ إلَى مَا قضَوْا بهِ، فلَا يُحتاجُ أنْ يُحكمَ عليهِ مرَّةً أخرَى؛

  • لأنَّهُمْ أعرفُ، وقولَهُمْ أقربُ إلَى الصّوابِ،
  • ولقولِهِ چ : «أصحابِي كالنُّجومِ، بأيِّهِمُ اقتديْتُمُ اهتديْتُمْ»(٢).
٦٢٨

ومنْهُ:

  • (فِي النّعامَةِ بدنةٌ)؛ رُويَ عنْ عمرَ، وعثمانَ، وعليٍّ، وزيدٍ، وابنِ عبّاسٍ، ومعاويَةَ ښ (٣)؛ لأنَّهَا تُشبهُهَا.
  • (و) فِي (حمارِ الوحشِ) بقرةٌ؛ رُويَ عنْ عمرَ ڔ (٤)،
  • (و) فِي (بقرتِهِ(٥))؛ أيِ: الواحدَةِ مِنْ بقرِ الوحشِ: بقرَةٌ؛ رُويَ عنِ ابنِ مسعودٍ ڔ (٦)،
  • (و) فِي (الأيِّلِ) -علَى وزنِ قِنَّبٍ، وخُلَّبٍ، وسَيِّدٍ-: بقرَةٌ؛ رُويَ عنِ ابنِ عبّاسٍ ژ (٧).
٦٢٩
  • (و) فِي (التّيْتَلِ(٨)) بقرةٌ،
  • قالَ الجوهرِيُّ: «التّيْتَلُ(٩): الوعلُ المُسنُّ»(١٠).
  • (و) فِي (الوعلِ بقرةٌ)، يُروَى عنِ ابنِ عمرَ ژ أنَّهُ قالَ: «فِي الأرْوَى بقرَةٌ»(١١).
  • قالَ فِي الصِّحاحِ: «الوعلُ هيَ الأروَى»(١٢).
  • وفِي القاموسِ: الوَعَلُ -بفتحِ الواوِ معَ فتحِ العينِ وكسرِهَا وسكونِهَا-: تيسُ الجبلِ(١٣).
  • (و) فِي (الضّبُعِ كبشٌ)، قالَ الإمامُ: «حكمَ فِيهَا رسولُ اللهِ چ بكبشٍ»(١٤).
٦٣٠
  • (و) فِي (الغزالَةِ عنْزٌ)، روَى(١٥) جابرٌ ڔ عنهُ چ أنَّهُ قالَ: «فِي الظّبْيِ شاةٌ»(١٦).
  • (و) فِي (الوَبْرِ) -وهوَ: دُوَيْبَةٌ كَحْلَاءُ دونَ السِّنَّوْرِ لَا ذَنَبَ لهَا-: جَدْيٌ.
  • (و) فِي (الضّبِّ جدْيٌ)؛ قضَى بهِ عمرُ، وأَرْبَدُ ژ (١٧).
  • والجَدْيُ: الذَّكرُ مِنْ أولادِ المعزِ لهُ ستَّةُ أشهُرٍ.
  • (و) فِي (اليَربُوعِ جَفْرَةٌ) لهَا أربعَةُ أشهُرٍ؛ رُويَ عنْ عمرَ(١٨)، وابنِ مسعودٍ ژ (١٩)،
٦٣١
  • (و) فِي (الأرنبِ عَناقٌ)؛ رُويَ عنْ عمرَ ڔ (٢٠)،
  • والعَنَاقُ: الأنثَى مِنْ أولادِ المعزِ أصغرُ مِنَ الجَفرَةِ.
  • (و) فِي (الحمامَةِ شاةٌ)، حكمَ بهِ عمرُ(٢١)، وعثمانُ(٢٢)، وابنُ عمرَ(٢٣)، وابنُ عبّاسٍ(٢٤)، ونافعُ بنُ عبدِ الحارثِ ښ فِي حَمامِ الحرمِ، وقِيسَ عليهِ حَمَامُ الإحرامِ.
  • والحمامُ: كلُّ مَا عبَّ الماءَ وهدَرَ، فيدخلُ فيهِ الفَوَاخِتُ، والوَرَاشِينُ، والقَطَا، والقُمْرِيُّ، والدُّبسِيُّ.
٦٣٢

ومَا لمْ تقضِ فيهِ الصّحابَةُ: يُرجَعُ فيهِ إلَى قولِ عدلَيْنِ خبيرَيْنِ.

ومَا لَا مِثلَ لهُ كباقِي الطّيرِ -ولوْ أكبرَ مِنَ الحمامِ-: فيهِ القيمَةُ.

وعلَى جماعَةٍ اشتركُوا فِي قتلِ صيدٍ جزاءٌ واحدٌ.

ڭ ڭ ڭ

(١) أخرجه أبو داود (٣٨٠١)، وابن ماجه (٣٠٨٥) من حديث جابر بن عبد الله ژ .

صححه ابن خزيمة (٢٦٤٦)، وابن حبان (٣٩٦٤)، والحاكم (١/٤٥٢).

وأنكره يحيى القطان وجعله موقوفًا على عمر ڔ ، حكاه الطحاوي في مشكل الآثار (٩/٩٥)، ورجَّح وقفه البيهقي في السنن الكبير (٥/١٨٣)، وابن الملقن في البدر المنير (٦/٣٦٠ - ٣٦١) وحكاه عن الدارقطني وابن القطان.

(٢) روي من حديث عدة من الصحابة: عمر وابنه وابن عباس وجابر وأبي هريرة والضحاك بن مزاحم وأنس ښ . وانظر لطرقها تخريج أحاديث الكشاف (٢/٢٢٩وما بعده)، والبدر المنير (٩/٥٨٤وما بعده).

والحديث ضعَّفه الإمام أحمد كما في المنتخب من علل الخلال (٦٩)، والبيهقي في المدخل إلى السنن (١/١٤٩)، وابن عبد البر في بيان العلم وفضله (١٧٥٨)، وغيرهم.

(٣) أخرجه عنهم الشافعي في الأم (٣/٤٨٨)، وعبد الرزاق (٤/٣٩٨)، وابن أبي شيبة في الجزء المفرد (ص٣٣٢)، والبيهقي (٥/١٨٢) من طريق عطاء الخراساني أنهم قالوا ذلك.

قال الشافعي: (هذا غير ثابت عند أهل العلم بالحديث)، قال البيهقي: (وجه ضعفه كونه مرسلًا)، وقال في معرفة السنن والآثار (٧/٤٠٢): (وروي من وجه آخر عن عطاء بن أبي رباح عن ابن عباس ژ ، وإسناده حسن)، أخرجه في السنن الكبير (٥/١٨٢).

(٤) لم نقف على من أخرجه عن عمر ڔ ، وأخرجه الدارقطني (٢٥٤٨)، والبيهقي (٥/١٨٢ و١٨٧) من طريقين عن ابن عباس ژ ، وضعَّفه ابن حزم في المحلى (٧/٢٢٨).

(٥) عُدلت في (الأصل، س) إلى: «بقره»، والمثبت من بقية النسخ، وهو الموافق لما في زاد المستقنع (ص١٢١) والإقناع (١/ ٥٩٩).

(٦) أخرجه عبد الرزاق (٤/٤٠٠) عن الضحاك بن مزاحم عن ابن مسعود ڔ .

(٧) أخرجه ابن أبي حاتم في تفسيره (٤/١٢٠٨)، والبيهقي (٥/١٨٧ - ١٨٨) من حديث علي بن أبي طلحة عن ابن عباس ژ .

(٨) في (ب، د، س، ض، ل، م): «الثيتل»، وجاء في هامش (س) قوله: (الذي في أصلها المقروءة على المؤلف ... «التيتل» بتاء فياء فتاء، وكلها مثناة. وما صححت عليه في نسختي هذه هو ما قدمه في المطلع ..) [وانظر: المطلع، للبعلي(ص٢١٥).

(٩) في (ب، د، س، ض، ل، م): «الثيتل».

(١٠) الصحاح (٤/١٦٤٥).

(١١) لم نقف على من أخرجه عن ابن عمر ژ ، وأخرجه عبد الرزاق (٤/٣٩٨ و ٤٠٠) عن عطاء ومجاهد به، وصححه ابن حزم عنهما في المحلى (٧/٢٢٨).

(١٢) انظر: (٥/١٨٤٣).

(١٣) قارن بما في: القاموس (ص١٠٦٨).

(١٤) سبق تخريجه في (ص٦٢٧)، والخلاف في وقفه ورفعه.

وأما قول الإمام أحمد، فانظر: زاد المسافر (٢/٥٦١).

(١٥) في (د، ز، ع، ل، م): «روي عن».

(١٦) أخرجه الدارقطني (٢٥٤٦)، والبيهقي (٥/١٨٣) عن جابر بن عبد الله ژ به مرفوعًا. وأخرجه مالك في الموطأ (١٢٤٤الزهري)، والشافعي في الأم (٣/٤٩٥)، والبيهقي (٥/١٨٣) عن جابر عن عمر موقوفًا عليه: (أنه قضى في الغزال بعنز).

وأخرجه عبد الرزاق (٤/٤٠٣) وقال: (شاة).

ورجّح وقفه البيهقي، وعبد الحق في الأحكام الوسطى (٢/٣٣٠)، وابن الملقن في البدر المنير (٦/٣٩٥)، وانظر ما سبق في (ص٦٢٧).

(١٧) أخرجه الشافعي في الأم (٣/٤٩٩)، وعبد الرزاق (٤/٤٠٢ - ٤٠٣)، وابن أبي شيبة (٤/٧٦)، والبيهقي (٥/١٨٢و١٨٥) من حديث طارق بن شهاب عن عمر وأربد ژ به، وفيه قصة.

صححه ابن الملقن في البدر المنير (٦/٤٠٠ - ٤٠١) وابن حجر في التلخيص الحبير (٤/١٦٩٢).

(١٨) أخرجه مالك في الموطأ (١٢٣٩)، والشافعي في الأم (٣/٤٩٧)، وعبد الرزاق (٤/٤٠١و٤٠٣)، والبيهقي (٥/١٨٤) من حديث جابر بن عبد الله عن عمر ښ .

(١٩) أخرجه الشافعي في الأم (٣/٤٩٨و٥٣١)، وعنه البيهقي (٥/١٨٥)، وعبد الرزاق (٤/٤٠١) من حديث أبي عبيدة بن عبد الله بن مسعود عن أبيه ڔ .

وأخرجه البيهقي عقبه، من حديث مجاهد عن ابن مسعود ڔ ، وقال: (وهاتان الروايتان عن ابن مسعود ڔ مرسلتان، وإحداهما تؤكد الأخرى).

(٢٠) أخرجه مالك في الموطأ (١٢٣٩)، والشافعي في الأم (٣/٤٩٦)، وعبد الرزاق (٤/٤٠٣)، والبيهقي (٥/١٨٤). صححه ابن الملقن (البدر المنير ٦/٣٩٥)، وابن حجر (التلخيص ٤/١٦٨٩)، وانظر ما سبق في رواية الغزال واليربوع.

(٢١) أخرجه الشافعي في الأم (٣/٥٠٢ - ٥٠٣)، ومن طريقه البيهقي (٥/٢٠٥)، وبنحوه أخرجه عبد الرزاق (٤/٤١٥)، وابن أبي شيبة الجزء المفرد (ص١٥٦) عن عمر ونافع بن عبد الحارث ژ .

(٢٢) أخرجه الشافعي في الأم (٣/٥٠٢ - ٥٠٣)، والبيهقي (٥/٢٠٥)، ومن وجه آخر عبد الرزاق (٤/٤١٨)، وابن أبي شيبة في الجزء المفرد (ص١٥٦).

(٢٣) أخرجه عبد الرزاق (٤/٤١٦)، وابن أبي شيبة في الجزء المفرد (ص١٥٥)، والبيهقي (٥/٢٠٦).

(٢٤) أخرجه الشافعي في الأم (٣/٥٠٣)، وعبد الرزاق (٤/٤١٤)، وابن أبي شيبة في الجزء المفرد (ص١٥٦)، والبيهقي (٥/٢٠٥).
