٧٧

(بابُ نواقضِ الوُضوءِ)

أيْ: مفسداتِهِ.

وهيَ ثمانيةٌ،

أحدُهَا: الخارجُ منْ سبيلٍ، وأشارَ إليهِ بقولِهِ: (ينقضُ) الوُضوءَ (مَا خرجَ منْ سبيلٍ)؛ أيْ: مخرجِ بولٍ أوْ غائطٍ، ولوْ: نادرًا، أوْ طاهرًا: كولدٍ بلَا دمٍ، أوْ مُقَطَّرًا فِي إحليلِهِ، أوْ مُحتشًى وابتلَّ.

  • لَا الدّائمُ؛ كالسّلسِ والاستحاضةِ: فلَا ينقضُ؛ للضّرورةِ.

(و) الثّانِي: (خارجٌ منْ بقيَّةِ البدنِ)، سوَى السّبيلِ:

  • (إنْ كانَ بولًا أوْ غائطًا)، قليلًا كانَ، أوْ كثيرًا.
  • (أوْ) كانَ (كثيرًا نجِسًا غيرَهمَا)؛ أيْ: غيرَ البولِ والغائطِ، كقيءٍ ولوْ بحالِهِ؛ لمَا روَى التّرمذِيُّ: «أنَّهُ چ قاءَ فتوضّأَ»(١).
٧٨
  • والكثيرُ: مَا فحُشَ فِي نفسِ كلِّ أحدٍ بحسَبِهِ.

وإذَا استدَّ(٢) المخرجُ وانفتحَ غيرُهُ: لمْ يثبتْ لهُ أحكامُ المعتادِ.

(و) الثّالثُ:

  • (زوالُ العقلِ)؛
  • أوْ(٣): تغطيتُهُ،
  • قالَ أبُو الخطّابِ وغيرُهُ: «ولوْ تلجّمَ ولمْ يخرجْ شيءٌ; إلحاقًا بالغالبِ»(٤).
  • (إلَّا يسيرَ نومٍ منْ قاعدٍ وقائمٍ) غيرِ محتبٍ أوْ متّكئٍ أوْ مستنِدٍ.

وعُلمَ منْ كلامِهِ:

  • أنَّ الجنونَ والإغماءَ والسُّكرَ: ينقضُ كثيرُهَا ويسيرُهَا، ذكرَهُ فِي المبدعِ إجماعًا(٥).
  • وينقضُ أيضًا: النّومُ منْ:
  • مضطجعٍ، وراكعٍ، وساجدٍ مطلقًا؛
  • كمحتبٍ، ومتّكئٍ، ومستنِدٍ،
٧٩
  • والكثيرُ منْ قائمٍ وقاعدٍ؛
  • لحديثِ: «العينُ وكاءُ السّهِ؛ فمَنْ نامَ فليتوضّأْ»، رواهُ أحمدُ وغيرُهُ(٦)، والسّهُ: حلقةُ الدُّبرِ.

(و) الرّابعُ:

  • (مسُّ ذكَرِ) آدمِيٍّ، -تعمّدَهُ أوْ لَا-، (متّصلٍ) ولوْ: أشلَّ، أوْ قُلْفَةً، أوْ منْ ميتٍ.
  • لَا:
  • الأنثيَيْنِ،
  • ولَا بائنٍ،
  • أوْ محَلِّهِ.
  • (أوْ) مسُّ (قُبلٍ) مِنِ امرأةٍ؛ وهوَ فرجُهَا الَّذي(٧) بينَ إسْكَتَيْهَا،
  • لقولِهِ چ : «مَنْ مسَّ ذكرَهُ فليتوضّأْ»، رواهُ مالكٌ والشّافعِيُّ ،
٨٠

وغيرُهمَا، وصحَّحَهُ أحمدُ والتّرمذِيُّ(٨)

  • وفِي لفظٍ: «مَنْ مسَّ فرجَهُ فليتوضّأْ»، صحّحَهُ أحمدُ(٩).
  • ولَا ينقضُ مسُّ شُفْرَيْهَا، وهمَا: حافَتَا فرجِهَا.

وينقضُ المسُّ بيدٍ بلَا حائلٍ، ولوْ كانتْ زائدةً، سواءٌ كانَ (بظهرِ كفِّهِ أوْ بطنِهِ) أوْ حرفِهِ، منْ رؤوسِ الأصابعِ إلَى الكوعِ؛ لعمومِ حديثِ: «مَنْ أفضَى بيدِهِ إلَى ذكَرِهِ ليسَ دونَهُ سترٌ فقدْ وجبَ عليهِ الوُضوءُ»، رواهُ أحمدُ(١٠).

٨١
  • لكنْ لَا ينقضُ مسُّهُ بالظّفرِ.

(و) ينقضُ:

  • (لمسُهمَا)؛ أيْ: لمسُ الذّكَرِ والقُبلِ معًا (منْ خُنثَى مُشكلٍ)، لشهوةٍ أوْ لَا؛ إذْ أحدُهمَا أصلِيٌّ قطعًا.
  • (و) ينقضُ أيضًا (لمسُ ذكَرٍ ذكرَهُ)؛ -أيْ: ذكرَ الخنثَى المشكلِ- لشهوةٍ؛ لأنَّهُ إنْ كانَ ذكرًا فقدْ مسَّ ذكرَهُ، وإنْ كانَ امرأةً فقدْ لمسَهَا لشهوةٍ.
  • فإنْ لمْ يمسَّهُ لشهوةٍ، أوْ مسَّ قُبلَهُ: لمْ ينتقضْ(١١).
  • (أوْ أنثَى قُبُلَهُ)؛ أيْ: وينقضُ لمسُ أنثَى قُبلَ الخنثَى المُشكلِ، (لشهوةٍ فيهمَا)؛ -أيْ: فِي هذهِ والّتِي قبلَهَا-؛ لأنَّهُ إنْ كانَ أنثَى فقدْ مسّتْ فرجَهَا، وإنْ كانَ ذكرًا فقدْ لمستْهُ لشهوةٍ.
  • فإنْ كانَ المسُّ(١٢) لغيرِهَا، أوْ مسّتْ ذكرَهُ: لمْ ينقضْ(١٣) وضوءَهَا.

(و) الخامسُ: (مسُّهُ)؛ أيِ: الذّكرِ (امرأةً بشهوةٍ)؛ لأنَّهَا الَّتِي تدعُو إلَى الحدثِ.

  • والباءُ للمصاحبةِ.
٨٢
  • والمرأةُ شاملةٌ:
  • للأجنبيّةِ،
  • وذاتِ المحرمِ،
  • والميتةِ،
  • والكبيرةِ،
  • والصّغيرةِ المميِّزةِ(١٤).
  • وسواءٌ كانَ المسُّ: باليدِ، أوْ غيرِهَا، ولوْ بزائدٍ لزائدٍ، أوْ أشلَّ.

(أوْ تمسُّهُ بهَا)؛ أيْ: ينقضُ مسُّهَا لرجلٍ بشهوةٍ؛ كعكسِهِ السّابقِ.

(و) ينقضُ (مسُّ حلْقَةِ دبرٍ)؛ لأنَّهُ فرجٌ، سواءٌ كانَ مِنهُ، أوْ منْ غيرِهِ.

  • (لَا مسُّ شعرٍ وسِنٍّ وظفرٍ) مِنهُ أوْ مِنهَا،
  • ولَا المسُّ بهَا.
  • (و) لَا مَسُّ رجلٍ لـ(أمردَ)، ولوْ بشهوةٍ.
  • (ولَا) المسُّ (معَ حائلٍ)؛ لأنَّهُ لمْ يمسَّ البشَرةَ.

(ولَا) ينتقضُ وُضوءُ (ملموسٍ بدنُهُ، ولوْ وجدَ مِنهُ شهوةً)، ذكرًا كانَ أوْ أنثَى.

  • وكذَا: لَا ينتقضُ وُضوءُ ملموسٍ فرجُهُ.
٨٣

(وينقضُ غَسلُ ميتٍ) مسلمًا كانَ أوْ كافرًا، ذكرًا كانَ أوْ أنثَى، صغيرًا أوْ كبيرًا؛ رُويَ عنِ ابنِ عمرَ وابنِ عبّاسٍ ښ : «أنّهمَا كانَا يأمرانِ غاسلَ الميتِ بالوُضوءِ»(١٥).

  • والغاسلُ هوَ: مَنْ يقلّبُهُ ويباشِرُهُ، ولوْ مرَّةً،
  • لَا: مَنْ يصبُّ عليهِ الماءَ، ولَا مَنْ يُيَمّمُهُ.

وهذَا هوَ السّادسُ.

(و) السّابعُ: (أكلُ اللّحمِ خاصّةً مِنَ الجزورِ)؛ أيِ: الإبلِ،

  • فلَا تنقضُ: بقيَّةُ أجزائِهَا؛ كالكبدِ، وشربُ لبنِهَا، ومرقِ لحمِهَا سواءٌ كانَ نِيئًا أوْ مطبوخًا.
  • قالَ أحمدُ: فيهِ حديثانِ صحيحانِ: حديثُ البراءِ ڔ (١٦) وحديثُ جابرِ بنِ سمرةَ ڔ (١٧).

(و) الثّامنُ المشارُ إليهِ بقولِهِ: (كلُّ مَا أوجبَ غُسلًا): كإسلامٍ، أوِ

٨٤

انتقالِ منِيٍّ؛ ونحوِهمَا: (أوجبَ وُضوءًا،

  • إلَّا الموتَ)؛ فيوجبُ الغُسلَ دونَ الوُضوءِ.

ولَا نقضَ بغيرِ مَا مرَّ:

  • كالقذفِ، والكذبِ، والغيبةِ؛ ونحوِهَا،
  • والقهقهةِ -ولوْ فِي الصّلاةِ-،
  • وأكلِ مَا مسّتِ النّارُ، غيرَ لحمِ الإبلِ.
  • ولَا يُسنُّ الوُضوءُ مِنهُما(١٨).

(ومَنْ تيقّنَ الطّهارةَ وشكَّ)؛ أيْ: تردّدَ (فِي الحدثِ)،

(أوْ بالعكسِ)؛ بأنْ تيقّنَ الحدثَ وشكَّ فِي الطّهارةِ:

  • (بنَى علَى اليقينِ) سواءٌ كانَ: فِي الصّلاةِ أوْ خارجِهَا، تساوَى عندَهُ الأمرانِ، أوْ غلبَ علَى ظنِّهِ أحدُهمَا؛ لقولِهِ چ : «لَا ينصرفْ حتَّى يسمعَ صوتًا أوْ يجدَ ريحًا»، متّفقٌ عليهِ(١٩).

(فإنْ تيقّنَهمَا)؛ أيْ: تيقّنَ الطّهارةَ والحدثَ، (وجهِلَ السّابقَ) منهُمَا:

  • (فهوَ بضدِّ حالِهِ قبلَهمَا) إنْ علمَها;
  • فإنْ كانَ قبلَهمَا متطهّرًا: فهوَ الآنَ محْدِثٌ،
  • وإنْ كانَ محْدِثًا: فهوَ الآنَ متطهّرٌ؛
٨٥
  • لأنَّهُ قدْ تيقَّنَ زوالَ تلكَ الحالةِ إلَى ضدِّهَا وشكَّ فِي بقاءِ ضدِّهَا، وهوَ الأصلُ.
  • وإنْ لمْ يعلمْ حالَهُ قبلَهمَا: تطهّرَ.

وإذَا سمِعَ اثنانِ صوتًا، أوْ شمّا ريحًا منْ أحدِهمَا لَا بعينِهِ:

  • فلَا وُضوءَ عليهِمَا،
  • ولَا يأتمُّ أحدُهمَا بصاحبِهِ،
  • ولَا يصاففُهُ فِي الصّلاةِ وحدَهُ،
  • وإنْ كانَ أحدُهمَا إمامًا: أعادَا صلاتَهمَا.

ڭ ڭ ڭ

(ويحرُمُ علَى المحْدِثِ:

  • مسُّ المصحفِ)، أوْ بعضِهِ، حتَّى جلدِهِ وحواشيهِ، بيدٍ وغيرِهَا، بلَا حائلٍ،
  • لَا: حملُهُ بعِلاقتِهِ(٢٠) أوْ فِي كيسٍ أوْ كُمٍّ منْ غيرِ مسٍّ،
  • ولَا تصفّحُهُ بكمِّهِ أوْ عودٍ،
  • ولَا صغيرٌ لوحًا فيهِ قرآنٌ مِنَ الخالِي مِنَ الكتابةِ،
  • ولَا مسُّ تفسيرٍ؛ ونحوِهِ.

ويحرُمُ أيضًا:

  • مسُّ مصحفٍ بعضوٍ متنجِّسٍ،
٨٦
  • وسفرٌ بهِ لدارِ حربٍ،
  • وتوسّدُهُ، وتوسّدُ كتبٍ فيهَا قرآنٌ؛
  • مَا لمْ يخفْ سرقةً.

ويحرُمُ أيضًا: كَتْبُ قرآنٍ بحيثُ يهانُ.

وكُرهَ:

  • مدُّ رِجلٍ إليهِ،
  • واستدبارُهُ،
  • وتخطّيهِ،
  • وتحليتُهُ بذهبٍ أوْ فضةٍ.

وتحرمُ: تحليةُ كتبِ العلمِ.

(و) يحرمُ علَى المحْدِثِ أيضًا: (الصّلاةُ) ولوْ نفلًا، حتَّى صلاةُ جنازةٍ، وسجودُ تلاوةٍ، وشكرٍ.

  • ولَا يكفرُ مَنْ صلَّى محْدِثًا.

(و) يحرمُ علَى المحْدِثِ أيضًا:(الطّوافُ)؛ لقولِهِ چ : «الطّوافُ بالبيتِ صلاةٌ، إلَّا أنَّ اللهَ أباحَ فيهِ الكلامَ»، رواهُ الشّافعِيُّ فِي مسندِهِ(٢١).

ڭ ڭ ڭ

(١) أخرجه الترمذي (٨٧)، وأحمد (٦/٤٤٣)، وأبو داود (٢٣٨١) من حديث معدان بن أبي طلحة عن أبي الدرداء ڔ أن النبي چ : قاء فأفطر، قال: فلقيت ثوبان مولى رسول الله چ (فأخبره الحديث)، فقال: (صدق أنا صببت له وضوءه).

والحديث اختلف في إسناده؛ فصححه أحمد، وروجع فقال: (حسين المعلم يجوِّده) (انظر: سنن الأثرم ١٠٥)، وكذا قال البخاري (انظر: علل الترمذي الكبير ٥٧)، والترمذي.

وصححه ابن خزيمة (١٩٥٦)، وابن حبان (١٠٩٧)، وابن منده (انظر: البدر المنير ٥/٦٦٣)، وتكلَّم فيه البيهقي (١/١٤٤)، وأجابَ عما ذكره ابنُ دقيق العيد في الإمام (٢/٣٤١-٣٤٣).

(٢) في (ب، ز): «انسد».

(٣) في (ب، د، ع): «أي».

(٤) انظر: الانتصار (١/٣١٢).

(٥) انظر: المبدع (١/١٣٤).

(٦) أخرجه أحمد (١/١١١)، وأبو داود (٢٠٣)، وابن ماجه (٤٧٧) من حديث علي بن أبي طالب ڔ .

سُئل أحمد (انظر: التنقيح ١/٢٥٣): عن حديث علي ومعاوية (المسند ٤/٩٦) في ذلك، فقال: (حديث عليٍّ أثبت وأقوى)، وضعَّفهما أبو حاتم وأبو زرعة (انظر: العلل لابن أبي حاتم س١٠٦)، وابن عبد البر في التمهيد (انظر: موسوعة شروح الموطأ ٢/٤٥١)، وابن عبد الهادي في تنقيح التحقيق (١/٢٥٢).

وحسّنه المنذري، وابن الصلاح، والنووي (انظر: التلخيص الحبير ١/٣١٤).

(٧) في (الأصل، س، ل، م): «التي».

(٨) أخرجه مالك (١١١)، والشافعي (١/٣٤)، وأحمد (٦/٤٠٦)، وأبو داود (١٨١)، والترمذي (٨٢)، وابن ماجه (٤٧٩)، والنسائي (١/١٠٠) من حديث بسرة بنت صفوان ږ .

صححه أحمد (انظر: مسائل أبي داود ١٩٦٦)، وابن معين (انظر: التمهيد/ موسوعة شروح الموطأ ٣/٢٣٧)، والترمذي ونقل عن البخاري قال: (أصح شيء في هذا الباب حديث بسرة)، وصححه ابن خزيمة (٣٣)، وابن حبان (١١١٢)، والدارقطني في العلل (س٤٠٦٢).

وضعَّفه علي بن المديني، والطحاوي (انظر: شرح معاني الآثار ١/٧٦)، وعمرو الفلَّاس (انظر: الإمام ٢/٢٧٦)، والنسائي، وإبراهيم الحربي (انظر: التنقيح ١/٢٦٤).

(٩) أخرجه ابن ماجه (٤٨١) من حديث مكحول عن عنبسة بن أبي سفيان عن أم حبيبة ږ به مرفوعًا.

صححه أحمد (انظر: التمهيد/ موسوعة شروح الموطأ ٣/٢٣٧)، وأبو زرعة (انظر: سنن الترمذي ٨٢)، وأعلّه البخاري (انظر: العلل الكبير للترمذي ٥٤)، وأبو حاتم (انظر: العلل لابنه س٨١، ٤٨٨).

وأجاب دحيم عن علته، وقال ابن السكن: (لا أعلم به علّة)، (انظر: التلخيص الحبير ١/٣٣١).

(١٠) أخرجه أحمد (٢/٣٣٣) من حديث أبي هريرة ڔ .

اختلف في رفعه ووقفه؛ فصححه ابن حبان (١١١٨)، والحاكم (١/١٣٨)، وابن السكن، وقوَّاه ابن عبدالبر (انظر: التمهيد/ موسوعة شروح الموطأ ٣/٢٤٢-٢٤٣).

ورجَّح الدارقطني في العلل (س١٤٥٤) وقفه.

(١١) في (ب، د، ز، س، م): «ينقض».

(١٢) في (ز، س، م): «اللمس».

(١٣) في (ع، ل، م): «ينتقض».

(١٤) في (ز، ل): «المميزة التي يوطأ مثلها».

(١٥) أخرجه عبد الرزاق في المصنف (٣/٤٠٥-٤٠٦)، وابن المنذر في الأوسط (٥/٣٧٣).

(١٦) أنه قال: سُئل رسول الله چ عن الوضوء من لحوم الإبل فقال: «توضؤوا منها».

أخرجه أحمد (٤/٢٨٨)، وأبو داود (١٨٤)، وابن ماجه (٤٩٤)، والترمذي (٨١).

نقل الترمذي في سننه عن إسحاق بن راهويه كقول أحمد عند المصنف، وصححه أيضًا ابن خزيمة (٣١) وحكى الاتفاق على صحته.

(١٧) أنه قال: سأل رجلٌ رسول الله چ أأتوضأ من لحوم الإبل قال: «نعم، فتوضأْ من لحوم الإبل»، أخرجه أحمد (٥/٨٦)، ومسلم (٣٦٠).

وتصحيح أحمد المذكور نقله ابن أبي يعلى في الطبقات (٢/٢٨٤) بإسناده إلى الإمام أحمد من رواية الأثرم، وبنحوه في: مسائل عبدالله (٥٩).

(١٨) في (ز، ع): «منها».

(١٩) أخرجه أحمد (٤/٤٠)، والبخاري (١٣٧)، ومسلم (٣٦١)، من حديث عبد الله بن زيد ڔ .

(٢٠) في (د، ع): «بعلاقة».

(٢١) أخرجه الشافعي في مسنده (١/٣٤٨) موقوفًا على ابن عمر ژ بنحوه.

وأخرجه مرفوعًا من حديث طاوس عن ابن عباس ژ الترمذي (٩٦٠)، وأخرجه أحمد (٣/٤١٤) والنسائي (٥/٢٢٢) عن طاوس عن رجل أدرك النبي چ ، قال ابن حجر في التلخيص الحبير (١/٣٤٩): (والظاهر أن المبهم فيها هو ابن عباس).

أومأ أحمد في المسند إلى وقفه، ورجَّح الوقف النسائيُّ في سننه، والبزارُ في مسنده (١١/١٢٨)، والدارقطنيُّ في العلل (س٣٠٤٥) وشيخُ الإسلام ابن تيمية في مجموع الفتاوى (٢١/٢٧٤) وقال: (وأهل المعرفة بالحديث لا يصححونه إلا موقوفًا ويجعلونه من كلام ابن عباس، ولا يثبتون رفعه).

وصححه مرفوعًا ابن السكن، وابن خزيمة (٢٧٣٩)، وابن حبان (٣٨٣٦)، وحسنه ابن حجر في الأربعون العاليات (٤٢ عن الإرواء ١/١٥٥).
