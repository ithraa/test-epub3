٨٧

(بابُ الغُسلِ)

بضمِّ الغَيْنِ: الاغتسالُ؛ أيِ: استعمالُ الماءِ فِي جميعِ بدنِهِ علَى وجهٍ مخصوصٍ.

وبالفتحِ: الماءُ، أوِ الفعلُ.

وبالكسرِ: مَا يُغسلُ بهِ الرّأسُ منْ خطميٍّ وغيرِهِ.

(وموجِبُهُ) ستّةُ أشياءَ:

أحدُهَا: (خروجُ المنِيِّ):

  • منْ مخرجِهِ،
  • (دفقًا بلذّةٍ)،
  • (لَا) إنْ خرجَ (بدونِهمَا منْ غيرِ نائمٍ)؛ ونحوِهِ، فلوْ خرجَ مِنْ يقظانَ لغيرِ ذلكَ -كبَرْدٍ؛ ونحوِهِ- منْ غيرِ شهوةٍ: لمْ يجبْ بهِ ؛
٨٨

غُسلٌ؛ لحديثِ عليٍّ ڔ يرفعُهُ: «إذَا فضخْتَ الماءَ فاغتسلْ، وإنْ لمْ تكنْ فاضخًا فلَاْ تغتسلْ»، رواهُ أحمدُ(١)، والفضخُ: هوَ خروجُهُ بالغلبةِ، قالَهُ إبراهيمُ الحربِيُّ(٢)

  • فعلَى هذَا يكونُ نجِسًا وليسَ بمذْيٍ(٣)، قالَهُ فِي الرّعايةِ(٤).
  • وإنْ خرجَ المنِيُّ منْ غيرِ مخرجِهِ -كمَا لوِ انكسرَ صلبُهُ؛ فخرجَ منهُ-: لمْ يجبِ الغُسلُ، وحكمُهُ: كالنّجاسةِ المعتادةِ.

وإنْ أفاقَ نائمٌ أوْ نحوُهُ يمكنُ بلوغُهُ؛ فوجدَ بللًا:

  • فإنْ تحققَ أنَّهُ منِيٌّ: اغتسلَ فقطْ، ولوْ لمْ يذكرِ احتلامًا.
  • وإنْ لمْ يتحقّقْهُ منيًّا:
  • فإنْ سبقَ نومَهُ ملاعبةٌ أوْ نظرٌ أوْ فكْرٌ أوْ نحوُهُ، أوْ كانَ بهِ إبْرِدَةٌ: لمْ يجبْ غُسلٌ،
  • وإلَّا اغتسلَ وطهَّرَ مَا أصابَهُ؛ احتياطًا.

(وإنِ انتقلَ) المنِيُّ (ولمْ يخرجِ:

  • اغتسلَ لهُ)؛ لأنَّ الماءَ قدْ باعدَ محَلَّهُ؛ فصدقَ عليهِ اسمُ الجُنُبِ.
  • ويحصلُ بهِ البلوغُ ونحوُهُ ممَّا يترتّبُ علَى خروجِهِ.
٨٩
  • (فإنْ خرجَ) المنِيُّ (بعدَهُ)، أيْ: بعدَ غُسلِهِ لانتقالِهِ: (لمْ يُعِدْهُ)؛ لأنَّهُ منِيٌّ واحدٌ فلَا يوجبُ غُسلَيْنِ.

(و) الثّانِي:

  • (تغييبُ حشفةٍ أصليّةٍ) أوْ قدرِهَا إنْ فُقدَتْ -وإنْ لمْ يُنزلْ-،
  • (فِي فرجٍ أصلِيٍّ، قُبلًا كانَ أوْ دبرًا)، -وإنْ لمْ يجدْ حرارةً-.
  • فإنْ أولجَ الخنثَى المُشكلُ حشفتَهُ فِي فرجٍ أصلِيٍّ ولمْ يُنزلْ،
  • أوْ أولجَ غيرُ الخنثَى ذكرَهُ فِي قُبلِ الخنثَى:
  • فلَا غُسلَ علَى واحدٍ منهُمَا إلَّا أنْ يُنزلَ.
  • ولَا غُسلَ إذَا مسَّ الخِتانُ الخِتانَ منْ غيرِ إيلاجٍ،
  • ولَا بإيلاجِ بعضِ الحشفةِ.

(ولوْ) كانَ الفرجُ: (منْ بهيمةٍ، أوْ ميتٍ)، أوْ نائمٍ، أوْ مجنونٍ، أوْ صغيرٍ يُجامَعُ مثلُهُ.

وكذَا: لوِ استدخلتْ ذكرَ نائمٍ أوْ صغيرٍ؛ ونحوِهِ.

(و) الثّالثُ: (إسلامُ كافرٍ) أصليًّا كانَ أوْ مرتدًّا، ولوْ مميِّزًا، أوْ لمْ يوجدْ فِي كفرِهِ مَا يوجبُهُ؛ لأنَّ «قيسَ بنَ عاصمٍ أسلمَ فأمرَهُ النّبيُّ چ أنْ يغتسلَ بماءٍ وسدرٍ»، رواهُ أحمدُ والتّرمذِيُّ وحسّنَهُ(٥).

٩٠
  • ويُستحبُّ لهُ إلقاءُ شعرِهِ،
  • قالَ أحمدُ: «ويغسلُ ثيابَهُ»(٦).

(و) الرّابعُ: (موتٌ)،

  • غيرَ شهيدِ معركةٍ، ومقتولٍ ظلمًا، ويأتِي(٧).

(و) الخامسُ: (حيضٌ)،

(و) السّادسُ: (نِفاسٌ)،

  • ولَا خلافَ فِي وجوبِ الغُسلِ بهِمَا، قالَهُ فِي المغنِي(٨).
  • فيجبُ بالخروجِ، والانقطاعُ شرطٌ.
  • (لَا ولادةٌ عاريةٌ عنْ دمٍ)؛ فلَا غُسلَ بهَا، والولدُ طاهرٌ.

ڭ ڭ ڭ

(ومَنْ لزمَهُ الغُسلُ) لشيءٍ ممَّا تقدَّمَ:

  • (حرمَ عليهِ):
  • الصّلاةُ،
  • والطّوافُ،
٩١
  • ومسُّ المصحفِ،
  • و(قراءةُ القرآنِ)؛ أيْ: قراءةُ آيةٍ فصاعدًا.
  • ولهُ: قولُ مَا وافقَ قرآنًا -إنْ لمْ يقصدْهُ- كالبسملةِ، والحمدلةِ؛ ونحوِهمَا؛ كالذِّكْرِ.
  • ولهُ:
  • تهجّيهِ،
  • والتّفكّرُ فيهِ،
  • وتحريكُ شَفَتيْهِ بهِ مَا لمْ يبيّنِ الحروفَ،
  • وقراءةُ بعضِ آيةٍ مَا لمْ تطُلْ.
  • ولَا يُمنعُ منْ قراءتِهِ متنجِّسُ الفمِ.
  • ويُمنعُ الكافرُ منْ قراءتِهِ، ولوْ رُجِيَ إسلامُهُ.
  • (ويعبرُ المسجدَ)؛ أيْ: يدخلُهُ؛ لقولِهِ تعالَى: {وَلَا جُنُبًا إِلَّا عَابِرِي سَبِيلٍ} [النساء:٤٣]؛ أيْ: طريقٍ (لحاجةٍ)، وغيرِهَا علَى الصّحيحِ؛ كمَا مشَى عليهِ فِي الإقناعِ(٩).
  • وكونُهُ طريقًا قصيرًا: حاجةٌ.
  • وكرهَ أحمدُ اتّخاذَهُ طريقًا(١٠).
  • ومصلَّى العيدِ: مسجدٌ،
٩٢
  • لَا مصلَّى الجنائزِ.
  • (ولَا) يجوزُ أنْ (يلبثَ فيهِ) -أيْ: فِي المسجدِ- مَنْ عليهِ غُسلٌ (بغيرِ وُضوءٍ)؛
  • فإنْ توضّأَ: جازَ لهُ اللّبثُ فيهِ.
  • ويُمنعُ مِنهُ: مجنونٌ، وسكرانُ، ومَنْ عليهِ نجاسةٌّ تتعدَّى.
  • ويُباحُ بهِ وُضوءٌ وغُسلٌ إنْ لمْ يُؤذِ بهِمَا.
  • وإذَا كانَ الماءُ فِي المسجدِ: جازَ دخولُهُ بلَا تيمّمٍ.
  • وإنْ أرادَ اللّبثَ فيهِ للاغتسالِ: تيمّمَ.
  • وإنْ تعذّرَ الماءُ واحتاجَ للّبثِ: جازَ بلَا تيمّمٍ.

ڭ ڭ ڭ

(ومَنْ غسّلَ ميتًا) مسلمًا أوْ كافرًا: سُنَّ لهُ الغُسلُ؛ لأمرِ أبِي هريرةَ ڔ بذلكَ، رواهُ أحمدُ وغيرُهُ(١١).

(أوْ أفاقَ منْ جنونٍ أوْ إغماءٍ بلَا حُلْمٍ)؛ أيْ: إنزالٍ: (سُنَّ لهُ الغُسلُ)؛

  • لأنَّ «النّبيَّ چ اغتسلَ مِنَ الإغماءِ»، متّفقٌ عليهِ(١٢)،
٩٣
  • والجنونُ فِي معناهُ، بلْ أوْلَى.

وتأتِي بقيَّةُ الأغسالِ المستحبَّةِ فِي أبوابِ مَا تُستحبُّ لهُ.

ويتيمّمُ للكلِّ(١٣)، ولمَا يُسنُّ لهُ وُضوءٌ، لعذرٍ.

ڭ ڭ ڭ

(و) صفةُ (الغُسلِ الكاملِ)؛ أيِ: المشتملِ علَى الواجباتِ والسُّننِ:

  • (أنْ ينوِيَ) رفعَ الحدثِ أوِ استباحةَ الصّلاةِ أوْ نحوِهَا.
  • (ثمَّ يسمِّيَ)، وهيَ هنَا كوُضوءٍ(١٤): تجبُ معَ الذُّكرِ، وتسقطُ معَ السّهوِ.
  • (ويغسلَ يديْهِ ثلاثًا) كمَا فِي الوُضوءِ، وهوَ هنَا آكدُ؛ لرفعِ الحدثِ عنهمَا بذلكَ.
  • (و) يغسلَ (مَا لوّثَهُ) منْ أذًى،
  • (ويتوضّأَ) كاملًا،
  • (ويَحثِيَ) الماءَ (علَى رأسِهِ ثلاثًا تروِّيهِ(١٥))؛ أيْ: يروِّي فِي كلِّ مرَّةٍ
٩٤
  • أصولَ شعرِهِ؛
  • لحديثِ عائشةَ ږ : «كانَ رسولُ اللهِ چ إذَا اغتسلَ مِنَ الجنابةِ غسلَ يديْهِ ثلاثًا، وتوضّأَ وُضوءَهُ للصّلاةِ، ثمَّ يخلّلُ شعرَهُ بيديْهِ حتَّى إذَا ظنَّ أنَّهُ قدْ روَّى بشَرتَهُ أفاضَ الماءَ عليهِ ثلاثَ مرّاتٍ، ثمَّ غسلَ سائرَ جسدِهِ»، متّفقٌ عليهِ(١٦).
  • (ويعمَّ بدنَهُ غَسلًا) -فلَا يجزئُ المسحُ- (ثلاثًا)، حتَّى مَا يظهرُ منْ فرجِ امرأةٍ عندَ قعودٍ لحاجةٍ، وباطنَ شعرٍ، وتنقضُهُ لحيضٍ ونِفاسٍ(١٧).
  • (ويَدلُكَهُ)؛ أيْ: يدلُكَ بدنَهُ بيديْهِ; ليتيقَّنَ وصولَ الماءِ إلَى مغابنِهِ وجميعِ بدنِهِ.
  • ويتفقّدَ:
  • أصولَ شعرِهِ،
  • وغضاريفَ أذنيْهِ،
  • وتحتَ حلقِهِ وإبْطيْهِ،
  • وعمقَ سُرّتِهِ،
  • وبينَ إليتيْهِ،
  • وطيَّ ركبتيْهِ.
٩٥
  • (ويتيامَنَ)؛ لأنَّهُ چ كانَ يعجبُهُ التّيامنُ فِي طُهورِهِ(١٨).
  • (ويغسلَ قدميْهِ) ثانيًا (مكانًا آخرَ).

ويكفِي الظّنُّ فِي الإسباغِ.

قالَ بعضُهُمْ: «ويحرّكُ خاتمَهُ؛ ليتيقَّنَ وصولَ الماءِ».

(و) الغُسلُ (المجزئُ)؛ أيِ: الكافِي:

  • (أنْ ينوِيَ) كمَا تقدَّمَ(١٩)،
  • (ويسمِّيَ(٢٠)) فيقولَ: بسمِ اللهِ،
  • (ويعمَّ بدنَهُ بالغَسلِ مرَّةً(٢١))؛ أيْ: يغسلَ ظاهرَ جميعِ بدنِهِ، ومَا فِي حكمِهِ، منْ غيرِ ضررٍ:
  • كالفمِ والأنفِ،
  • والبشَرةِ الَّتِي تحتَ الشّعورِ ولوْ كثيفةً،
  • وباطنِ الشّعرِ وظاهرِهِ معَ مسترسِلِهِ،
  • ومَا تحتَ حشفةِ أقلفَ إنْ أمكنَ شمْرُهَا.

ويرتفعُ حدثٌ قبلَ زوالِ حكمِ خَبَثٍ.

٩٦

ويُستحبُّ:

  • سدرٌ فِي غُسلِ:
  • كافرٍ أسلمَ،
  • وحائضٍ،
  • وأخذُهَا مِسكًا تجعلُهُ فِي قطنةٍ أوْ نحوِهَا، وتجعلُهَا فِي فرجِهَا؛
  • فإنْ لمْ تجدْ: فطيبًا؛
  • فإنْ لمْ تجدْ: فطينًا.

ڭ ڭ ڭ

(ويتوضّأُ بمُدٍّ)، استحبابًا.

  • والمُدُّ: رطلٌ وثلثٌ عراقِيٌّ، ورطلٌ وأوقيّتانِ وسُبعَا أوقيّةٍ مصرِيٌّ، وثلاثُ أواقٍ وثلاثةُ أسباعِ أوقيّةٍ دمشقيّةٍ، وأوقيّتانِ وأربعةُ أسباعِ أوقيّةٍ قدسيّةٍ.

(ويغتسلُ بصاعٍ)، وهوَ: أربعةُ أمدادٍ.

  • وإنْ زادَ: جازَ،
  • لكنْ يُكرهُ الإسرافُ، ولوْ علَى نهرٍ جارٍ.

ويحرُمُ: أنْ يغتسلَ عُريانًا بينَ النّاسِ.

  • وكُرهَ: خاليًا فِي الماءِ.

(فإنْ أسبغَ بأقلَّ) ممَّا ذُكرَ فِي الوُضوءِ أوِ الغُسلِ: أجزأَ.

٩٧
  • والإسباغُ: تعميمُ العضوِ بالماءِ؛ بحيثُ يجرِي عليهِ ولَا يكونُ مسحًا.

(أوْ نوَى بغُسلِهِ:

  • الحدثَيْنِ)،
  • أوِ الحدثَ وأطلقَ،
  • أوِ الصّلاةَ؛ ونحوَهَا، ممَّا يحتاجُ لوُضوءٍ وغُسلٍ:
  • (أجزأَ) عنِ الحدثَيْنِ.
  • ولمْ يلزمْهُ ترتيبٌ ولَا موالاةٌ.

(ويُسنُّ لجُنبٍ)، ولوْ أنثَى، وحائضٍ، ونفسَاء انقطعَ دمُهمَا:

  • (غَسلُ فرجِهِ)؛ لإزالةِ مَا عليهِ مِنَ الأذَى.
  • (والوُضوءُ:
  • لأكلٍ) وشربٍ؛ لقولِ عائشةَ ږ : «رخّصَ رسولُ اللهِ چ للجُنبِ إذَا أرادَ أنْ يأكلَ أوْ يشربَ أنْ يتوضّأَ وُضوءَهُ للصّلاةِ»، رواهُ أحمدُ بإسنادٍ صحيحٍ(٢٢).
  • (ونومٍ)؛ لقولِ عائشةَ ږ : «كانَ رسولُ اللهِ چ إذَا أرادَ أنْ ينامَ .
٩٨

وهوَ جُنبٌ غسَلَ فرجَهُ وتوضّأَ وُضوءَهُ للصّلاةِ»، متّفقٌ عليهِ

(٢٣)
  • ويُكرهُ تركُهُ لنومٍ فقطْ.
  • (و)(٢٤) يُسنُّ أيضًا غَسلُ فرجِهِ ووُضوؤُهُ: (لمعاودةِ(٢٥) وطءٍ)؛ لحديثِ: «إذَا أتَى أحدُكُمْ أهلَهُ ثمَّ أرادَ أنْ يعاودَ فليتوضّأْ بينَهُمَا وُضوءًا»، رواهُ مسلمٌ وغيرُهُ، وزادَ الحاكمُ: «فإنَّهُ أنشطُ للعَوْدِ»(٢٦).
  • والغُسلُ أفضلُ.

ڭ ڭ ڭ

وكرهَ الإمامُ أحمدُ: بناءَ الحمّامِ، وبيعَهُ، وإجارتَهُ(٢٧)، وقالَ فِي مَنْ بنَى حمّامًا للنّساءِ: ليسَ بعدلٍ(٢٨).

  • ولرجلٍ دخولُهُ بسترةٍ معَ أمنِ الوقوعِ فِي محرّمٍ،
  • ويحرُمُ علَى المرأةِ بلَا عذرٍ.

ڭ ڭ ڭ

(١) أخرجه أحمد (١/١٠٩)، وأبو داود (٢٠٦)، والنسائي (١/١١١) من حديث علي بن أبي طالب ڔ بنحوه.

وصححه ابن خزيمة (٢٠)، وابن حبان (١١٠٧).

(٢) نقله في: المبدع، للبرهان ابن مفلح (١/١٥٠).

(٣) في (ز، ع، ل): «وليس بمني».

(٤) نقله في: المبدع، للبرهان ابن مفلح (١/١٥٠).

(٥) أخرجه أحمد (٥/٦١)، والترمذي (٦٠٥)، وأبو داود (٣٥٥)، والنسائي (١/١٠٩) من حديث خليفة بن حُصين عن جده قيس بن عاصم ڔ بنحوه.

قال الترمذي: (حديث حسن لا نعرفه إلا من هذا الوجه)، وصححه ابن خزيمة (٢٥٤)، وابن حبان (١٢٤٠)، وابن السكن (انظر: التلخيص الحبير ٢/١٠٣٢)، وحسَّنه البغوي في شرح السنة (٢/١٧١)،وضعَّفه ابن القطان في بيان الوهم والإيهام (٢/٤٢٩).

(٦) أحكام أهل الملل، للخلال (ص٤٦) برقم (١٠٩).

(٧) أي عند قوله: «(ولَا يُغسَّلُ: شهيدُ) معركَةٍ، ومقتولٌ ظلمًا..» في (ص٤٢٢).

(٨) انظر: المغني (١/٢٧٧).

(٩) انظر: الإقناع (١/٦٩).

(١٠) انظر: مسائل الكوسج (٢/٧٣٩)، مسائل ابن هانئ (١/٦٩).

(١١) أخرجه ابن أبي شيبة (٣/٢٦٩)، والبزار في مسنده (٧٩٩٢)، والبيهقي (١/٣٠٢) موقوفًا على أبي هريرة ڔ كما ذكره المصنِّف. ورُويَ مرفوعًا أخرجه أحمد (٢/٢٧٢)، ورجَّح أحمد وقفه فيما نقله عنه ابن عبد الهادي في تنقيح التحقيق (١/٣١٨)، وكذا البخاري في التاريخ الكبير (١/٣٩٧) وأبو حاتم (انظر: العلل لابنه س١٠٣٥)، والبيهقي.

(١٢) أخرجه أحمد (٦/٢٥١)، والبخاري (٦٨٧)، ومسلم (٤١٨) من حديث عائشة ږ .

(١٣) في (ز): «وتيمَّمَ للكلِّ لحاجةٍ، ولمَا يُسنُّ لهُ وضوءٌ، لعذرٍ»، فغايَرَ بين اللفظين (لحاجة ... لعذر)، وهو الموافق لما في المنتهى والإقناع وغيرهما، أما بقية نسخ الروض المعتمدة وغيرها التي بين أيدينا فعلى خلافها، وقد قال البهوتي في حاشية المنتهى [١/٩٤]: (ولعل مغايرته بينهما للتفنن)، فيظهر أن ترك البهوتي لها في الروض مقصودٌ؛ طلبًا للاختصار [وانظر: حاشية الخلوتي على المنتهى ١/١٣٥].

(١٤) أي عند قوله: «وتجبُ التسميةُ فِي الوضوءِ معَ الذُّكرِ» في (ص٥١).

(١٥) في (ز، ع، ل): «يرويه»، والمثبت هو الموافق لما في زاد المستقنع (ص٤٤ت: القاسم).

(١٦) أخرجه البخاري (٢٧٢)، ومسلم (٣١٦) من حديث عائشة ږ . ولفظ: (ثلاثًا) في غسل اليدين لمسلم وحده.

(١٧) «ونفاس» أُلحقت في هامش (الأصل) مصححة، وليست في بقية النسخ.

(١٨) أخرجه أحمد (٦/٢٠٢) واللفظ له، والبخاري (١٦٥)، ومسلم (٢٦٨) عن عائشة ږ قالت: (كان رسول الله چ يحب التيامن في طهوره).

(١٩) أي عند قوله: «(أنْ ينوِيَ) رفعَ الحدثِ أوِ استباحةَ الصلاةِ أوْ نحوِهَا.» في (ص٩٣).

(٢٠) في (ب): «ثم يسمي» وهو الموافق لما في زاد المستقنع (ص٤٥ت: القاسم).

(٢١) في (ز): «ويعم بالماء جميع بدنه بالغسل مرة».

(٢٢) هذا لفظ حديث عمَّار بن ياسر ڔ ، أخرجه أحمد (٤/٣٢٠)، وأبو داود (٢٢٥)، والترمذي (٦١٣).

قال الترمذي: (حسن صحيح)، وحسَّنه البغوي في شرح السنة (٢/٣٥)، وأعلَّه بالانقطاع أبو داود، والدارقطني (انظر: سؤالات البرقاني ٢٩)، والمنذري في مختصر سنن أبي داود (٦/٩٣).

وجاء ذكر (الأكل) من حديث عائشة ږ ، انظره في الآتي بعده.

(٢٣) أخرجه أحمد (٦/١٢٦)، والبخاري (٢٨٨)، ومسلم (٣٠٥).

زاد أحمد ومسلم: «أو يأكل»، وأعلَّها شعبة وأحمد (انظر: فتح الباري لابن رجب ١/٣٥٠)، ويدل عليه صنيع النسائي (١/١٣٨-١٣٩).

(٢٤) في (الأصل) الواو من الشرح، والمثبت من (ب، د، ز، س، ع، ل)، وهو الموافق لما في زاد المستقنع (ص٤٥ت: القاسم).

(٢٥) في (ب، س، ع، ل) اللام من الشرح، وهو الموافق لما في زاد المستقنع (ص٤٥ت: القاسم).

(٢٦) أخرجه أحمد (٣/٧)، ومسلم (٣٠٨)، والحاكم في المستدرك (١/١٥٢) من حديث أبي سعيد الخدري ڔ .

(٢٧) انظر: مسائل أبي داود (ص٢٦٥) برقم (١٢٦٥)، الفروع، للشمس ابن مفلح (١/٢٧٠).

(٢٨) انظر: زاد المسافر (٣/٥٢٤).
