٤١١

(فصلٌ)

(غَسلُ الميّتِ) المسلمِ (وتكفينُهُ): فرضُ كفايَةٍ؛ لقولِ النَّبِيِّ چ فِي الَّذِي وقصتْهُ راحلتُهُ: «اغسِلُوهُ بماءٍ وسدرٍ وكفِّنُوهُ فِي ثوبيْهِ» متّفقٌ عليهِ عنِ ابنِ عبّاسٍ(١).

(والصّلَاةُ عليهِ): فرضُ كفايَةٍ؛ لقولِهِ چ : «صلُّوا علَى مَنْ قالَ: لَا إلهَ إلَّا اللهُ» رواهُ الخلّالُ والدّارقطنِيُّ، وضعّفَهُ ابنُ الجوزيِّ(٢).

(ودفنُهُ فرضُ كفايَةٍ)؛ لقولِهِ تعالَى: {ثُمَّ أَمَاتَهُۥ فَأَقۡبَرَهُ} [عبس:٢١]، قالَ ابنُ عبّاسٍ ڔ : «معناهُ: أكرمَهُ بدفنِهِ»(٣).

وحملُهُ أيضًا فرضُ كفايَةٍ.

واتّباعُهُ سنَّةٌ.

وكَرهَ الإمامُ للغاسلِ والحفّارِ أخْذَ أجرَةٍ علَى عملِهِ إلَّا أنْ يكونَ محتاجًا فيُعطَى منْ بيْتِ المالِ، فإنْ تعذّرَ أُعطِيَ بقدرِ عملِهِ، قالَهُ فِي المبدعِ(٤).

٤١٢

والأفضلُ أنْ يُختارَ لتغسيلِهِ:

  • ثقةٌ،
  • عارفٌ بأحكامِهِ.

(وأَوْلَى النّاسِ بغَسلِهِ:

  • وصيُّهُ) العدلُ؛ لأنَّ أبَا بكرٍ ڔ أوصَى أنْ تغسلَهُ امرأتُهُ أسماءُ ږ (٥)، وأوصَى أنسٌ ڔ أنْ يغسِلَهُ محمَّدُ بنُ سيرينَ(٦)،
  • (ثمَّ أبُوهُ)؛ لاختصاصِهِ بالحُنُوِّ والشّفقةِ،
  • (ثمَّ جدُّهُ) وإنْ علَا؛ لمشاركتِهِ الأبَ فِي المعنَى،
  • (ثمَّ الأقربُ فالأقربُ منْ عصباتِهِ)،
  • فيُقدَّمُ الابنُ،
  • ثمَّ ابنُهُ وإنْ نزلَ،
  • ثمَّ الأخُ لأبوينِ،
  • ثمَّ الأخُ لأبٍ؛
٤١٣
  • علَى ترتيبِ الميراثِ،
  • (ثمَّ ذوُو أرحامِهِ) كالميراثِ،
  • ثمَّ الأجانبُ.
  • وأجنبيٌّ أَوْلَى منْ زوجَةٍ وأمَةٍ،
  • وأجنبيَّةٌ أَوْلَى منْ زوجٍ وسيِّدٍ،
  • وزوجٌ أَوْلَى منْ سيِّدٍ،
  • وزوجَةٌ أَوْلَى منْ أمِّ ولدٍ.

(و) الأَوْلَى (بـ)ـغَسلِ (أنثَى:

  • وصيّتُهَا) العدلُ،
  • (ثمَّ القربَى فالقربَى منْ نسائِهَا)،
  • فتُقدَّمُ أمُّهَا وإنْ علتْ،
  • ثمَّ بنتُهَا وإنْ نزلَتْ،
  • ثمَّ القربَى كالميراثِ،
  • وعمّتُهَا وخالتُهَا سواءٌ، وكذَا بنْتُ أخيهَا وبنْتُ أختِهَا; لاستوائِهمَا فِي القربِ والمَحرميَّةِ.

(ولكلِّ واحدٍ مِنَ الزّوجَيْنِ) إنْ لمْ تكنِ الزّوجَةُ ذمّيَّةً: (غَسلُ صاحبِهِ)؛

٤١٤

  • لمَا تقدَّمَ عنْ أبِي بكرٍ ڔ (٧)، وروَى ابنُ المنذرِ أنَّ عليًّا ڔ غسّلَ فاطمَةَ ږ (٨)؛
  • ولأنَّ آثارَ النِّكاحِ منْ عدَّةِ الوفَاةِ والإرثِ باقيَةٌ، فكذَا الغَسلُ.
  • ويشملُ:
  • مَا قبلَ الدُّخولِ،
  • وأنَّهَا تغسلُهُ وإنْ لمْ تكنْ فِي عدَّةٍ؛ كمَا لوْ ولدَتْ عقِبَ موتِهِ،
  • والمطلّقَةَ الرّجعيَّةَ إذَا أُبيحتْ(٩).

(وكذَا سيِّدٌ معَ سُرِّيَّتِهِ)؛ أيْ: أمتِهِ المباحَةِ لهُ ولوْ أمَّ ولدٍ.

(ولِرجلٍ وامرأةٍ غَسْلُ مَنْ لهُ) دونَ(١٠) (سبعِ سنينَ فقطْ) ذكرًا كانَ أوْ أنثَى؛

٤١٥

  • لأنَّهُ لَا عورَةَ لهُ؛
  • ولأنَّ إبراهيمَ ابْنَ النَّبِيِّ چ غسّلَهُ النِّساءُ(١١)،
  • فتغسلُهُ مجرَّدًا بغيرِ سُترَةٍ،
  • وتَمَسُّ عورتَهُ،
  • وتَنظرُ إلَيْهَا.

(و) إنْ(١٢) (ماتَ رجلٌ بينَ نسوةٍ) ليسَ فيهنَّ زوجةٌ ولَا أمةٌ مباحَةٌ لهُ: يُمِّمَ،

(أوْ عكسُهُ) بأنْ ماتتِ امرأةٌ بينَ رجالٍ ليسَ فيهِمْ زوجٌ ولَا سيِّدٌ لهَا: (يُمِّمَتْ،

كخنثَى مشكلٍ) لمْ تحضُرْهُ أمَةٌ لهُ: فيُمِّمَ(١٣)؛

  • لأنَّهُ لَا يحصلُ بالغَسلِ منْ غيرِ مسٍّ تنظيفٌ، ولَا إزالَةُ نجاسَةٍ بلْ ربَّمَا كثُرَتْ،

وعُلمَ مِنهُ: أنَّهُ لَا مدخلَ للرِّجالِ فِي غَسْلِ الأقاربِ مِنَ النِّساءِ، ولَا بالعكسِ.

(ويحرُمُ:

  • أنْ يُغسِّلَ مسلمٌ كافرًا)،
٤١٦
  • وأنْ يحملَهُ
  • أوْ يُكفِّنَهُ
  • أوْ يتبعَ جنازتَهُ؛
  • كالصّلَاةِ عليهِ؛
  • لقولِهِ تعالَى: {لَا تَتَوَلَّوۡاْ قَوۡمًا غَضِبَ ٱللَّهُ عَلَيۡهِمۡ} [الممتحنة:١٣].
  • (أوْ يدفنَهُ)؛ للآيَةِ.

(بلْ يُوارَى) وجوبًا (لعدمِ) مَنْ يواريهِ؛ لإلقاءِ قتلَى بدرٍ فِي القَليبِ(١٤).

ويُشترطُ لغَسلِهِ:

  • طَهوريَّةُ ماءٍ،
  • وإباحتُهُ،
  • وإسلامُ غاسلٍ؛
  • لَا نائبًا عنْ مسلمٍ نواهُ،
  • وعقلُهُ ولوْ مميِّزًا، أوْ حائضًا أوْ جُنبًا.

ڭ ڭ ڭ

(وإذَا أَخذَ)؛ أيْ شرعَ (فِي غَسلِهِ:

  • ستَرَ عورتَهُ) وجوبًا، وهيَ مَا بينَ سُرّتِهِ وركبتِهِ،
٤١٧
  • (وجرَّدَهُ) ندبًا؛ لأنَّهُ أمكنُ فِي تغسيلِهِ، وأبلغُ فِي تطهيرِهِ،
  • وغُسِّل چ فِي قميصٍ(١٥)؛ لأنَّ فضلاتِهِ طاهرَةٌ فلمْ يُخْشَ تنجُّسُ قميصِهِ،
  • (وستَرَهُ عنِ العيونِ) تحتَ سِترٍ فِي خيمَةٍ أوْ بيتٍ إنْ أمكنَ؛ لأنَّهُ أسترُ لهُ،
  • (ويُكرهُ لغيرِ مُعِينٍ فِي غَسلِهِ حضورُهُ)؛ لأنَّهُ ربَّمَا كانَ فِي الميِّتِ مَا لَا يُحِبُّ اطّلاعَ أحدٍ عليهِ، والحاجَةُ غيرُ داعيَةٍ إلَى حضورِهِ بخلافِ المُعيْنِ.
  • (ثمَّ يرفعُ رأسَهُ)؛ أيْ رأسَ الميِّتِ غيرِ أنثَى حاملٍ (إلَى قربِ جلوسِهِ) بحيثُ يكونُ كالمحتضَنِ فِي صدرِ غيرِهِ، (ويعصِرُ بطنَهُ برفقٍ)؛ ليَخرُجَ مَا هوَ مستعدٌّ للخروجِ، ويكونُ هناكَ بَخُورٌ.
  • (ويُكثِرُ صبَّ الماءِ حينئذٍ)؛ ليدفعَ مَا يخرجُ بالعصرِ،
  • (ثمَّ يلُفُّ) الغاسلُ (علَى يدِهِ خِرقَةً فينجّيهِ)؛ أيْ يمسحُ فرجَهُ بِهَا، (ولَا يحِلُّ مسُّ عورَةِ مَنْ لهُ سبعُ سنينَ) بغيرِ حائلٍ؛ كحالِ الحيَاةِ؛ لأنَّ التّطهيرَ يمكنُ بدونِ ذلكَ.
٤١٨
  • (ويُستحبُّ أنْ لَا يمَسَّ سائرَهُ إلَّا بخِرقَةٍ)؛ لفعلِ عليٍّ ڔ معَ النَّبِيِّ چ (١٦)،
  • فحينئذٍ يُعِدُّ الغاسلُ خِرقتَيْنِ إحداهُمَا للسّبيلَيْنِ، والأخرَى لبقيَّةِ بدنِهِ.
  • (ثمَّ يوضِّئُهُ ندبًا) كوُضوئِهِ للصّلَاةِ؛ لمَا روَتْ أمُّ عطيَّةَ ږ أنَّ النَّبِيَّ چ قالَ فِي غَسلِ ابنتِهِ: «ابدأْنَ بميامِنِهَا ومواضعِ الوُضوءِ منهَا» رواهُ الجماعةُ(١٧)،
  • وكانَ ينبغِي تأخيرُهُ عنْ نيَّةِ الغَسلِ كمَا فِي المنتهَى(١٨) وغيرِهِ.
  • (ولَا يُدخِلُ الماءَ فِي فِيهِ ولَا فِي أنفِهِ)؛ خشيَةَ تحريكِ النّجاسَةِ، (ويُدخِلُ إصبَعيْهِ) إبهامَهُ وسبّابتَهُ (مبلولتَيْنِ)؛ أيْ عَلَيْهِمَا خِرقَةٌ مبلولَةٌ (بالماءِ بينَ شَفتيْهِ فيمسحُ أسنانَهُ، وفِي منخريْهِ فينظّفُهمَا)
٤١٩
  • بعدَ غَسلِ كفَّيِ الميّتِ، فيقومُ المسحُ فِيهِمَا مَقامَ غسلِهمَا؛ خوفَ تحريكِ النّجاسَةِ بدخولِ الماءِ جوفَهُ، (ولَا يُدخلُهمَا)؛ أيِ: الفمَ والأنفَ (الماءَ)؛ لمَا تقدَّمَ.
  • (ثمَّ ينوِي غَسلَهُ)؛ لأنَّهُ طهارَةٌ تعبُّديَّةٌ فاشتُرطَتْ لهُ النّيَّةُ كغُسلِ الجنابَةِ،
  • (ويسمِّي) وجوبًا؛ لمَا تقدَّمَ،
  • (ويغسلُ برغوَةِ السِّدرِ) المضروبِ (رأسَهُ ولحيتَهُ فقطْ)؛ لأنَّ الرّأسَ أشرفُ الأعضاءِ والرّغوَةُ لَا تَعلَقُ بالشّعرِ،
  • (ثمَّ يغسلُ شِقَّهُ الأيمنَ، ثمَّ) شِقَّهُ (الأيسرَ)؛ للحديثِ السّابقِ،
  • (ثمَّ) يغسلُهُ (كلَّهُ)؛ أيْ يُفيضُ الماءَ علَى جميعِ بدنِهِ،

يفعلُ مَا تقدَّمَ: (ثلاثًا)؛

  • إلَّا الوُضوءَ، ففِي المرَّةِ الأُولَى فقطْ.

(يُمِرُّ فِي كلِّ مرَّةٍ) مِنَ الثّلاثِ (يدَهُ علَى بطنِهِ)؛ ليَخرُجَ مَا تخلّفَ،

  • (فإنْ لمْ يَنْقَ بثلاثِ) غسلاتٍ: (زِيدَ حتَّى يَنْقَى ولوْ جاوزَ السّبعَ)،
  • وكُرهَ: اقتصارُهُ فِي غَسلِهِ علَى مرَّةٍ إنْ لمْ يَخرُجْ مِنهُ شيءٌ،
  • فيحرُمُ الاقتصارُ مَا دامَ يخرجُ شيءٌ علَى مَا دونَ السّبعِ،
  • وسُنَّ قطعٌ علَى وترٍ.
  • ولَا تجبُ مباشرَةُ الغَسلِ، فلوْ تُركَ تحتَ ميزابٍ ونحوِهِ، وحضرَ مَنْ يصلحُ لغَسلِهِ، ونوَى وسمَّى، وعمَّهُ الماءُ: كفَى.
٤٢٠

(ويجعلُ فِي الغَسلَةِ الأخيرَةِ) ندبًا: (كافورًا) وسدرًا؛ لأنَّهُ يصلِّبُ الجسدَ ويطردُ عنهُ الهوامَّ برائحتِهِ.

(والماءُ الحارُّ): يُستعملُ إذَا احتيجَ إليهِ،

(والأشنانُ): يُستعملُ إذَا احتيجَ إليهِ،

(والخِلالُ: يُستعملُ إذَا احتيجَ إليهِ)،

  • فإنْ لمْ يُحتجْ إلَيْهَا كُرهَتْ.

(ويقصُّ: شاربَهُ،

ويقلِّمُ أظفارَهُ)،

  • ندبًا إنْ طالَا،

ويُؤخذُ شعرُ إبْطيْهِ،

  • ويُجعلُ المأخوذُ معَهُ؛ كعضوٍ ساقطٍ.

وحرُمَ:

  • حلقُ رأسٍ،
  • وأخذُ عانَةٍ،
  • كختنٍ.

(ولَا يسرِّحُ شعرَهُ)؛ أيْ: يُكرهُ ذلكَ؛ لمَا فيهِ منْ تقطيعِ الشّعرِ منْ غيرِ حاجَةٍ إليهِ.

٤٢١

(ثمَّ يُنشَّفُ) ندبًا (بثوبٍ)؛ كمَا فُعلَ بهِ چ (١٩).

(ويُضْفرُ) ندبًا (شعرُهَا)؛ أيِ: الأنثَى (ثلاثَةَ قرونٍ، ويُسْدَلُ وراءَهَا)؛ لقولِ أمِّ عطيَّةَ ږ : «فَضَفَّرْنَا شعرَهَا ثلاثَةَ قرونٍ، وألقيناهُ خلفَهَا» رواهُ البخارِيُّ(٢٠).

(وإنْ خرجَ مِنهُ)؛ أيِ الميِّتِ (شيءٌ بعدَ سبعِ) غسلاتٍ:

  • (حُشِيَ) المحَلُّ:
  • (بقطنٍ)؛ ليمنعَ الخارجَ، كالمستحاضَةِ،
  • (فإنْ لمْ يستمسكْ) بالقطنِ: (فبِطينٍ حُرٍّ)؛ أيْ خالصٍ؛ لأنَّ فيهِ قوَّةً تمنعُ الخارجَ.
  • (ثمَّ يُغسلُ المحَلُّ) المتنجِّسُ بالخارجِ،
  • (ويُوضّأُ) الميِّتُ وجوبًا، كالجُنبِ إذَا أحدثَ بعدَ الغُسلِ.

(وإنْ خرجَ) مِنهُ شيءٌ (بعدَ تكفينِهِ: لمْ يُعَدِ الغَسلُ)؛ دفعًا للمشقَّةِ.

ولَا بأسَ:

  • بقولِ غاسلٍ لهُ: انقلبْ يرحمُكَ اللهُ ونحوَهُ،
٤٢٢
  • ولَا بغَسلِهِ فِي حمّامٍ.

(ومحرِمٌ) بحجٍّ أوْ عمرَةٍ (ميِّتٌ: كحيٍّ؛

  • يُغسلُ بماءٍ وسدرٍ) لَا كافورٍ،
  • (ولَا يُقرَّبُ طيبًا) مطلقًا،
  • (ولَا يُلبَسُ ذكرٌ مخيطًا) منْ قميصٍ ونحوِهِ،
  • (ولَا يُغطَّى رأسُهُ، ولَا وجهُ أنثَى) محرِمَةٍ،
  • ولَا يُؤخذُ شيءٌ منْ شعرِهمَا أوْ ظُفُرِهمَا؛
  • لمَا فِي الصّحيحَيْنِ منْ حديثِ ابنِ عبّاسٍ ژ : أنَّ النَّبِيَّ چ قالَ فِي محرِمٍ ماتَ: «غسِّلُوهُ بماءٍ وسدرٍ، وكفِّنُوهُ فِي ثوبيْهِ، ولَا تحنِّطُوهُ، ولَا تخمِّرُوا رأسَهُ؛ فإنَّهُ يُبعثُ يومَ القيامَةِ ملبِّيًا»(٢١).

ولَا تُمنعُ معتدَّةٌ منْ طِيبٍ.

وتزالُ اللَّصوقُ لغَسلٍ واجبٍ إنْ لمْ يسقطْ منْ جسدِهِ شيءٌ بإزالتِهَا، فيُمسحُ عَلَيْهَا؛ كجبيرَةِ الحيِّ.

ويُزالُ خاتمٌ ونحوُهُ ولوْ بِبَرْدٍ.

ڭ ڭ ڭ

(ولَا يُغسَّلُ: شهيدُ) معركَةٍ، ومقتولٌ ظلمًا، ولوْ أنثيَيْنِ أوْ غيرَ مكلّفَيْنِ؛

  • لأنَّهُ چ فِي شهداءِ أُحدٍ أمرَ بدفنِهِمْ بدمائِهِمْ ولمْ يغسِّلْهُمْ(٢٢)،
٤٢٣
  • وروَى أبُو داودَ عنْ سعيدِ بنِ زيدٍ ڔ قالَ: سمِعْتُ رسولَ اللهِ چ يقولُ: «منْ قُتلَ دونَ دينِهِ فهوَ شهيدٌ، ومَنْ قُتلَ دونَ دمِهِ فهوَ شهيدٌ، ومَنْ قُتلَ دونَ مالِهِ فهوَ شهيدٌ، ومَنْ قُتلَ دونَ أهلِهِ فهوَ شهيدٌ» وصحّحَهُ التِّرمذِيُّ(٢٣).
  • (إلَّا أنْ يكونَ) الشّهيدُ أوِ المقتولُ ظلمًا:
  • (جُنبًا)،
  • أوْ وجبَ عَلَيْهِمَا الغُسلُ: لحيضٍ أوْ نِفاسٍ أوْ إسلامٍ.

(ويُدفنُ) وجوبًا:

  • بدمِهِ؛ إلَّا أنْ تُخالِطَهُ نجاسَةٌ فيُغسلَا.
  • و(فِي ثيابِهِ) الَّتِي قُتلَ فِيهَا،
  • (بعدَ نزعِ السِّلاحِ والجلودِ عنهُ)؛
  • لمَا روَى أبُو داودَ وابنُ ماجهْ عنِ ابنِ عبّاسٍ ژ : «أنَّ رسولَ اللهِ چ أمرَ بقتلَى أحدٍ أنْ يُنزعَ عنهُمُ الحديدُ والجلودُ، وأنْ يُدفنُوا فِي ثيابِهِمْ بدمائِهِمْ»(٢٤).
٤٢٤
  • (وإنْ سُلبَهَا كُفِّنَ بغيرِهَا) وجوبًا.

(ولَا يُصلَّى عليهِ)؛

  • للأخبارِ(٢٥)؛
  • لكونِهِمْ أحياءً عندَ ربِّهِمْ.

(وإنْ:

  • سقطَ منْ دابّتِهِ) أوْ شاهقٍ بغيرِ فعلِ العدوِّ،
  • (أوْ وُجدَ ميِّتًا ولَا أثرَ بهِ)،
  • أوْ ماتَ حتفَ أنفِهِ،
  • أوْ برفسَةٍ،
  • أوْ عادَ سهمُهُ عليهِ،
  • (أوْ حُمِلَ: فأكلَ)، أوْ شرِبَ، أوْ نامَ، أوْ بالَ، أوْ تكلّمَ، أوْ عطسَ،
  • (أوْ طالَ بقاؤُهُ عرفًا:
  • غُسِّلَ وصُلِّيَ عليهِ)؛ كغيرِهِ.

ويُغسَّلُ الباغِي، ويُصلَّى عليهِ.

ويُقتلُ قاطعُ الطّريقِ، ويُغسَّلُ ويُصلَّى عليهِ، ثمَّ يُصلبُ.

(والسِّقطُ إذَا بلغَ أربعَةَ أشهرٍ غُسِّل وصلِّيَ عليهِ) -وإنْ لمْ يستهلَّ(٢٦)-;

٤٢٥

لقولِهِ چ : «والسّقطُ يُصلَّى عليهِ ويُدعَى لوالديْهِ بالمغفرَةِ والرّحمَةِ» رواهُ أحمدُ وأبُو داودَ(٢٧).

  • ويُستحبُّ تسميتُهُ، فإنْ جُهلَ أذكرٌ أمْ أنثَى: سُمِّيَ بصالحٍ لهمَا.

(ومنْ تعذَّرَ غَسلُهُ) لعدمِ الماءِ، أوْ غيرِهِ كالحَرْقِ والجُذامِ والتّبضيعِ: (يُمِّمَ)؛ كالجُنبِ إذَا تعذّرَ عليهِ الغُسلُ،

  • وإنْ تعذّرَ غَسلُ بعضِهِ غُسّلَ مَا أمكنَ ويُمِّمَ للباقِي.

(و) يجبُ (علَى الغاسلِ: سَترُ مَا رآهُ) مِنَ الميِّتِ (إنْ لمْ يكنْ حسنًا) فيلزمُهُ سَترُ الشّرِّ لَا إظهارُ الخيرِ.

ونرجُو للمحسنِ ونخافُ علَى المسيءِ، ولَا نشهدُ إلَّا لمَنْ شهِدَ لهُ النَّبِيُّ چ .

ويحرُمُ سوءُ الظّنِّ بمسلمٍ ظاهرِ العدالَةِ،

  • ويُستحبُّ ظنُّ الخيرِ بالمسلمِ.

ڭ ڭ ڭ

(١) أخرجه أحمد (١/٢١٥)، والبخاري (١٢٦٥)، ومسلم (١٢٠٦).

(٢) أخرجه الدارقطني (١٧٦١-١٧٦٣)، وابن الجوزي في العلل المتناهية (١/٤٢٢-٤٢٤) من حديث ابن عمر ژ .

قال الدارقطني –بعد إيراده لحديثين عَقِبه عن أبي هريرة وعلي-: (ليس فيها شيء يثبت)، وضعَّفه ابن الجوزي كذلك في التحقيق (٢/٤٧٧ مع التنقيح).

(٣) لم نقف على من أخرجه.

(٤) انظر: المبدع (٢/٢٢٤).

(٥) أخرجه مالك في الموطأ (٥٩٣)، وعبدالرزاق (٣/٤٠٨)، وابن أبي شيبة (٣/٢٤٩)، والبيهقي (٣/٣٩٧) من حديث عائشة ږ .

قال البيهقي: (وهذا الحديث الموصول وإن كان راويه محمد بن عمر الواقدي وليس بالقوي، فله شواهد مراسيل).

(٦) أخرجه أحمد في الزهد (١٧٨٨)، وفي العلل رواية عبدالله (٢١٥)، ومن طريقه أبو نعيم في الحلية (٢/٢٦٧).

(٧) سبق تخريجه في (ص٤١٢).

(٨) ذكره حكايةً ولم يسنده في الأوسط (٥/٣٥٦)، وأخرجه الشافعي (مسنده ١/٢٠٦)، وعبدالرزاق (٣/٤١٠)، والدارقطني (١٨٥١)، والبيهقي (٣/٣٩٦) من حديث أم جعفر بنت محمد عن جدتها أسماء بنت عميس قالت: (أوصت فاطمة إذا ماتت أن لا يغسلها إلا أنا وعلي)، قالت: (فغسلتها أنا وعلي).

أنكر أحمد هذا الحديث كما نقله ابن الجوزي في التحقيق (٢/٦٢٤ مع التنقيح)، وكذا البيهقي فيما نقله ابن الملقن في البدر المنير (٥/٣٧٥).

(٩) في (د، ع، م): «أبيحت له».

(١٠) في (د، ز، س) من المتن.

(١١) أخرجه الزبير بن بكار في المنتخب من كتاب أزواج النبي چ (ص٦٠) وفيه أنه توفي وله ثمانية عشر شهرًا وغسَّلته مرضعتُه أم بردة خولة بنت المنذر ږ .

(١٢) في (د، ز، س) من المتن.

(١٣) في (د، ز، س، ل): «فيُيَمَّمُ»، وفي (ب): «فتيمم».

(١٤) أخرجه أحمد (١/ ٤١٧) والبخاري (٥٢٠) ومسلم (١٧٩٤) من حديث ابن مسعود ڔ .

(١٥) أخرجه أحمد (٦/٢٦٧)، وأبو داود (٣١٤١) عن عائشة ژ .

صححه ابن حبان (٦٦٢٧)، والحاكم (٣/٥٩-٦٠)، وحسَّنه النووي في الخلاصة (٣٣٢٠).

(١٦) أخرجه ابن أبي شيبة (٣/٢٤٠)، والبيهقي (٣/٣٨٨) من حديث يزيد بن أبي زياد عن عبدالله بن الحارث بن نوفل: أن عليًّا غسَّل النبي چ وعلى النبي چ قميص، وبيد علي خرقة يتبع بها تحت القميص.

واختلف على يزيد فيه، فرُوي عنه عن مقسم عن ابن عباس بمثله، أخرجه الطبراني في الكبير (٦٢٩)، وضعَّفه الذهبي في (تاريخ الإسلام ١/٨٢٨)، وقال الهيثمي في مجمع الزوائد (١٧/٤٥٢): (فيه يزيد بن أبي زياد، وهو حسن الحديث على ضعفه، وبقية رجاله ثقات).

(١٧) أخرجه أحمد (٦/٤٠٨)، والبخاري (١٦٧)، ومسلم (٩٣٩)، وأبو داود (٣١٤٥)، والترمذي (٩٩٠)، وابن ماجه (١٤٥٩)، والنسائي (٤/٣٠).

(١٨) انظر: المنتهى (١/٣٩٣-٣٩٤).

(١٩) أخرجه أحمد (١/٢٦٠) من حديث حسين بن عبدالله بن عبيدالله بن عباس عن عكرمة عن ابن عباس وفيه: حتى إذا فرغوا من غسل رسول الله چ ، وكان يغسل بالماء والسدر، جففوه.

وحسين هذا تكلَّموا فيه، وتركه أحمد وقال: (له أشياء منكرة) (انظر: تهذيب الكمال ٦/٣٨٤).

(٢٠) أخرجه أحمد (٦/٤٠٨)، والبخاري (١٢٦٣)، ومسلم (٩٣٩).

(٢١) أخرجه أحمد (١/٢١٥)، والبخاري (١٨٥١)، ومسلم (١٢٠٦).

(٢٢) أخرجه البخاري (١٣٤٣) من حديث جابر بن عبدالله ژ .

(٢٣) أخرجه أحمد (١/١٩٠)، وأبو داود (٤٧٧٢)، والترمذي (١٤٢١)، والنسائي (٧/١١٦)، وبالجملة الأولى منه ابن ماجه (٢٥٨٠).

صححه الترمذي، وابن حبان (٣١٩٤).

(٢٤) أخرجه أحمد (١/٢٤٧)، وأبو داود (٣١٣٤)، وابن ماجه (١٥١٥).

ضعَّفه: البزار في مسنده (٥١٠٢)، وابن الملقن في البدر المنير (٥/٢٥٣)، وابن حجر في التلخيص (٣/١١٩٧-١١٩٨).

(٢٥) منها: ما أخرجه البخاري (١٣٤٣) من حديث جابر بن عبدالله ژ في شهداء أحد قال: وأمر بدفنهم في دمائهم ولم يغسلوا ولم يصلَّ عليهم.

(٢٦) في (د): «يستهل صارخًا».

(٢٧) أحمد (٤/٢٤٨-٢٤٩)، وأبو داود (٣١٨٠) بلفظ المصنف، وأخرجه الترمذي (١٠٣١)، وابن ماجه (١٥٠٧)، والنسائي (٤/٥٥-٥٦) من حديث المغيرة بن شعبة ڔ بلفظ: «والطفل يُصلى عليه».

اختلف في وقفه ورفعه؛ وصححه مرفوعًا: أحمد كما في زاد المعاد (١/٤٩٤)، والترمذي، وابن حبان (٣٠٤٩)، والحاكم (١/٣٦٣).
