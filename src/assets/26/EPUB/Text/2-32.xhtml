(بابُ الهدْيِ، والأُضحيَّةِ)، والعقيقَةِ

الهدْيُ: مَا يُهدَى للحرمِ مِنْ نَعَمٍ وغيرِهَا،

  • سُمِّيَ بذلكَ؛ لأنَّهُ يُهدَى إلَى اللهِ سبحانَهُ وتعالَى.

والأُضحيَّةُ: بضمِّ الهمزَةِ وكسرِهَا: واحدَةُ الأضاحِي، ويُقالُ: ضَحِيَّةٌ.

  • وأجمعَ المسلمُونَ علَى مشروعيّتِهَا(١).

(أفضلُهَا:

  • إبِلٌ،
  • ثمَّ بقرٌ)
  • إنْ أُخرجَ كاملًا؛ لكثرَةِ الثّمنِ ونفعِ الفقراءِ،
  • (ثمَّ غنمٌ)،

وأفضلُ كلِّ جنسٍ:

  • أسمنُ،
  • فأغلَى ثمنًا؛
  • لقولِهِ تعالَى: {وَمَن يُعَظِّمۡ شَعَٰٓئِرَ ٱللَّهِ فَإِنَّهَا مِن تَقۡوَى ٱلۡقُلُوبِ} [الحج:٣٢].
٦٨٤
  • فأشهبُ -وهوَ الأملحُ-؛ أيِ: الأبيضُ، أوْ بياضُهُ(٢) أكثرُ مِنْ سوادِهِ،
  • فأصفرُ،
  • فأسودُ.

(ولَا يجزئُ فِيهَا إلَّا:

  • جَذَعُ ضأنٍ) -مَا لَهُ ستَّةُ أشهُرٍ- كمَا يأتِي،
  • (وثَنِيُّ سِواهُ)؛ أيْ: سوَى الضّأنِ مِنْ إبِلٍ، وبقرٍ، ومعزٍ:
  • (فالإبِلُ)؛ أيِ: السِّنُّ المعتبَرُ لإجزاءِ إبِلٍ: (خمسُ) سنينَ.
  • (ولبقرٍ: سنتانِ.
  • ولمعزٍ: سنةٌ.
  • ولضأنٍ(٣): نصفُهَا)؛ أيْ: نصفُ سنةٍ؛ لحديثِ: «الجَذَعُ مِنَ الضّأنِ أضحيَّةٌ»، رواهُ ابنُ ماجهْ(٤).
٦٨٥

(وتجزئُ الشّاةُ عنْ واحدٍ) وأهلِ بيتِهِ وعيالِهِ؛ لحديثِ أبِي أيُّوبَ ڔ : «كانَ الرّجلُ فِي عهدِ رسولِ اللهِ چ يضحِّي بالشّاةِ عنهُ وعنْ أهلِ بيتِهِ، فيأكلُونَ ويُطعِمُونَ»(٥)، قالَ فِي شرحِ المقنعِ: «حديثٌ صحيحٌ»(٦).

(و) تجزئُ (البدنَةُ والبقرَةُ عنْ سبعَةٍ)؛ لقولِ جابرٍ ڔ : «أمرَنَا رسولُ اللهِ چ أنْ نشتركَ فِي الإبِلِ والبقرِ، كلُّ سبعةٍ فِي واحدٍ منهُمَا»، رواهُ مسلمٌ(٧).

  • وشاةٌ أفضلُ مِنْ سُبعِ بدنَةٍ أوْ بقرَةٍ.

ڭ ڭ ڭ

(ولَا تجزئُ:

  • العَوْرَاءُ) بيِّنَةُ العوَرِ -بأنِ انخسفَتْ عينُهَا- فِي الهدْيِ، ولَا الأُضحيَّةِ.
  • ولَا العمياءُ.
  • (ولَا العَجْفَاءُ) الهزيلَةُ الَّتِي لَا مخَّ فِيهَا.
  • (و) لَا (العَرْجاءُ) الَّتِي لَا تُطيقُ مشيًا معَ صحيحَةٍ.
  • (و) لَا (الهَتْمَاءُ) الَّتِي ذهبَتْ ثناياهَا مِنْ أصلِهَا.
٦٨٦
  • (و) لَا (الجدّاءُ)؛ أيْ: مَا شابَ ونشِفَ ضَرْعُهَا،
  • (و) لَا (المريضَةُ) بيِّنَةُ المرضِ؛
  • لحديثِ البراءِ بنِ عازبٍ ڔ : قامَ فينَا رسولُ اللهِ چ فقالَ: «أربعٌ لَا تجوزُ فِي الأضاحِي: العَوْرَاءُ البيِّنُ عَوَرُهَا، والمريضَةُ البيِّنُ مرضُهَا، والعَرْجَاءُ البيِّنُ ضلَعُهَا(٨)، والعَجْفَاءُ الَّتِي لَا تُنقِي»، رواهُ أبُو داودَ والنَّسائِيُّ(٩).
  • (و) لَا (العَضْبَاءُ) الَّتِي ذهبَ أكثرُ أذنِهَا أوْ قرنِهَا.

(بلْ) تجزئُ(١٠):

  • (البَتراءُ) الَّتِي لَا ذَنَبَ لهَا (خِلْقةً) أوْ مقطوعًا،
  • والصّمْعَاءُ وهيَ: صغيرَةُ الأذنِ،
  • (والجمّاءُ) الَّتِي لمْ يُخلقْ لهَا قرنٌ،
  • (وخَصِيٌّ غيرُ مجبوبٍ)؛ بأنْ قُطعَ خُصيتاهُ فقطْ.
٦٨٧

(و) يجزئُ معَ الكراهَةِ (مَا بأذنِهِ أوْ قرنِهِ):

  • خَرْقٌ،
  • أوْ شَقٌّ،
  • أوْ (قَطْعٌ أقلُّ مِنَ النِّصفِ)،
  • أوِ النِّصفُ فقطْ، علَى مَا نصَّ عليهِ فِي روايَةِ حنبلٍ وغيرِهِ(١١). قالَ فِي شرحِ المنتهَى: «وهذَا المذهبُ»(١٢).

ڭ ڭ ڭ

(والسُّنَّةُ نحرُ الإبِلِ قائمةً معقولةً يدُهَا اليُسرَى:

  • فيطعنُهَا بالحربَةِ) أوْ نحوِهَا (فِي الوَهْدَةِ الَّتِي بينَ أصلِ العُنُقِ والصّدرِ)؛ لفعلِهِ چ وفعلِ أصحابِهِ، كمَا رواهُ أبُو داودَ عنْ عبدِ الرّحمنِ بنِ سابطٍ(١٣).
٦٨٨

(و) السُّنَّةُ أنْ (يذبحَ غيرَهَا)؛ أيْ: غيرَ الإبِلِ:

  • علَى جنبِهَا الأيسرِ،
  • موجَّهةً إلَى القِبلَةِ.

(ويجوزُ عكسُهَا)؛ أيْ: ذَبحُ مَا يُنحرُ، ونَحرُ مَا يُذبَحُ؛

  • لأنَّهُ لمْ يتجاوزْ محَلَّ الذّبحِ،
  • ولحديثِ: «مَا أَنهرَ الدّمَ وذُكرَ اسمُ اللهِ عليهِ فكلْ»(١٤)،

(ويقولُ) حينَ يحرِّكُ يدَهُ بالنّحرِ أوِ الذّبحِ:

  • (بسمِ اللهِ) -وجوبًا-
  • (واللهُ أكبرُ) -استحبابًا-
  • (اللَّهُمَّ هذَا منْكَ ولَكَ)،
  • ولَا بأسَ بقولِهِ: اللّهمَّ تقبّلْ مِنْ فلانٍ.

ويَذبحُ واجبًا قبلَ نفلٍ.

(ويتولَّاهَا)؛ أيِ: الأضحيَّةَ (صاحبُهَا) إنْ قدرَ،

  • (أوْ يوكِّلُ مسلمًا ويشهدُهَا)؛ أيْ: يحضرُ ذبْحَهَا إنْ وكّلَ فيهِ.
  • وإنِ استنابَ ذمِّيًّا فِي ذبحِهَا: أجزأَتْ معَ الكراهَةِ.

(ووقْتُ الذّبحِ) لـ:

  • أُضحيَّةٍ،
٦٨٩
  • وهدْيِ نذرٍ، أوْ تطوُّعٍ، أوْ متعةٍ، أوْ قِرانٍ،
  • (بعدَ صلَاةِ العيدِ) بالبلدِ.
  • فإنْ تعدّدَتْ فيهِ: فبأسبقِ صلاةٍ.
  • فإنْ فاتَتِ الصّلَاةُ بالزّوالِ: ذبَحَ(١٥).
  • (و) إنْ كانَ بمحَلٍّ لَا تُصلَّى بهِ العيدُ: فالوقْتُ بعدَ (قدرِهِ)؛ أيْ: قدرِ زمنِ صلَاةِ العيدِ.

ويستمرُّ وقْتُ الذّبحِ (إلَى) آخِرِ (يومَيْنِ بعدَهُ)؛ أيْ: بعدَ يومِ العيدِ. قالَ أحمدُ: «أيّامُ النّحرِ ثلاثةٌ، عنْ غيرِ واحدٍ مِنْ أصحابِ رسولِ اللهِ چ »(١٦).

والذّبحُ:

  • فِي اليومِ الأوّلِ،
  • عَقِبَ الصّلَاةِ،
  • والخُطبَةِ،
  • وذبحِ الإمامِ:
  • أفضلُ،
  • ثمَّ مَا يليهِ.

(ويُكرَهُ) الذّبحُ (فِي ليلتَيْهِما(١٧))؛ أيْ: ليلتَيِ اليومَيْنِ بعدَ يومِ العيدِ؛ خروجًا مِنْ خلافِ مَنْ قالَ بعدمِ الإجزاءِ فِيهِمَا.

٦٩٠

  • (فإنْ فاتَ) وقْتُ الذّبحِ:
  • (قضَى واجبَهُ)، وفعَلَ بهِ كالأداءِ،
  • وسقطَ التّطوُّعُ؛ لفواتِ وقتِهِ.

ووقْتُ ذبحٍ واجبٍ بفعلِ محظورٍ: مِنْ حينِهِ،

  • فإنْ أرادَ فِعْلَهُ لعذرٍ: فلهُ ذبحُهُ قبلَهُ.

وكذَا: مَا وجبَ لتركِ واجبٍ: وقتُهُ مِنْ حينِهِ.

(١) في (ب، ز، ع، ل، م): «مشروعيتهما»، وأشار في هامش (ض) إلى أنها نسخة.

(٢) في (ب، ز، ض، ل، م): «الأبيض أو ما بياضه»، وفي (د): «أبيض أو ما بياضه»، وفي (ع): «أبيض أو بياضه».

(٣) في (س): «والبقر .. والمعز .. والضأن ..» وفي (د، م): «ولبقر .. ولمعز .. والضأن..»، وفي (ب): «والبقر .. ومعز .. والضأن ..».

(٤) أخرجه أحمد (٦/٣٦٨)، وابن ماجه (٣١٣٩) موصولًا، وأخرجه الإمام أحمد أيضًا من طريق أخرى مرسلًا، ورجح البيهقي إرساله في معرفة السنن (١٤/٢٩).

وضعَّف الحديث ابن حزم في المحلى (٧/٣٦٥) لجهالة بعض رواته، وتبعه ابن الملقن في البدر المنير (٩/٢٧٩ - ٢٨٠).

وللحديث شواهد، منها ما أخرجه الإمام أحمد (٤/١٤٩)، والبخاري واللفظ له (٥٥٤٧)، ومسلم (١٩٦٥) من حديث عقبة بن عامر ڔ : قَسَم النبي چ بين أصحابه ضحايا، فصارت لعقبة جذعة، فقلت: يا رسول الله صارت لي جذعة؟ قال: «ضحِّ بها».

(٥) أخرجه الترمذي (١٥٠٥)، وابن ماجه (٣١٤٧) من حديث أبي أيوب ڔ ، قال الترمذي: (حديث حسن صحيح).

(٦) الشرح الكبير (٩/٣٤٠).

(٧) أخرجه أحمد (٣/٢٩٢ - ٢٩٣)، ومسلم (١٣١٨).

(٨) كذا في (الأصل، ز، س، ض)، وفي (ب، د، ع): «ظلعها»، قال البهوتي في الكشاف (٦/٣٩٠): (وصوابه: بالظاء المشالة، كما يعلم من الصحاح وغيره) [وانظر: الصحاح ٣/١٢٥٦ (ظلع)].

(٩) أخرجه أحمد (٤/٢٨٤)، وأبو داود (٢٨٠٢)، والترمذي (١٤٩٧)، وابن ماجه (٣١٤٤)، والنسائي (٧/٢١٤ - ٢١٥) واللفظ له، وعندهم سوى الترمذي وأحد ألفاظ النسائي، بدل العجفاء: الكسيرة.

صححه الترمذي، وابن حبان (٥٩١٩).

(١٠) «تجزئ» ليست في (الأصل).

(١١) انظر: زاد المسافر (٤/٣٤).

(١٢) انظر: معونة أولي النهى (٤/٢٧٧).

(١٣) أخرجه أبو داود (١٧٦٧) من حديث ابن جريج عن أبي الزبير عن جابر ڔ وأخبرني عبد الرحمن بن سابط أن النبي چ وأصحابه كانوا ينحرون البدنة معقولة اليسرى قائمة على ما بقي من قوائمها.

ساقه البخاري في التاريخ الكبير (٥/٣٠٢) وقال: (ولا يصح) أي: ذكرُ أبي الزبير وجابر، وكذا أخرجه ابن أبي شيبة في الجزء المفرد (ص٢٠٦) عن ابن جريج أخبرني ابن سابط به، وعبد الرحمن بن سابط تابعي، فحديثه مرسل؛ لذا رجَّح إرساله ابن القطان في بيان الوهم والإيهام (٢/٢٩ - ٣٠)، والمجد ابن تيمية في منتقى الأخبار (٢١٢٤).

وصحح الحديث ابن السكن، وقوَّاه ابن الملقن في تحفة المحتاج (٢/٥٢٥).

(١٤) أخرجه أحمد (٣/٤٦٣)، والبخاري (٢٤٨٨)، ومسلم (١٩٦٨) من حديث رافع بن خديج ڔ .

(١٥) في (ز): «ذبح بعده».

(١٦) انظر: زاد المسافر (٤/٤١).

(١٧) في (د، ز، ع، م): «في ليلتهما».
