٤٢٦

(فصلٌ) فِي الكفنِ

(يجبُ تكفينُهُ فِي مالِهِ)؛ لقولِهِ چ فِي المحرِمِ: «كفِّنُوهُ فِي ثوبيْهِ»(١)،

  • (مقدَّمًا علَى:
  • دَينٍ) ولوْ برهنٍ،
  • (وغيرِهِ) منْ وصيَّةٍ وإرثٍ؛
  • لأنَّ المفلسَ يُقدَّمُ بالكسوَةِ علَى الدَّينِ، فكذَا الميِّتُ.

فيجبُ لحقِّ اللهِ تعالَى وحقِّ الميِّتِ، ثوبٌ:

  • لَا يصفُ البشَرَةَ،
  • يسترُ جميعَهُ،
  • منْ ملبوسِ مثلِهِ، مَا لمْ يوصِ بدونِهِ،
  • والجديدُ أفضلُ.

(فإنْ لمْ يكنْ لهُ)؛ أيْ: للميِّتِ (مالٌ فـ)ـكفنُهُ(٢) ومؤنَةُ تجهيزِهِ:

  • (علَى مَنْ تلزمُهُ نفقتُهُ)؛ لأنَّ ذلكَ يلزمُهُ حالَ الحيَاةِ فكذَا بعدَ الموتِ،
٤٢٧
  • (إلَّا الزّوجَ لَا يلزمُهُ كفنُ امرأتِهِ) ولوْ غنيًّا؛ لأنَّ الكسوَةَ وجبتْ عليهِ بالزّوجيَّةِ والتّمكُّنِ مِنَ الاستمتاعِ، وقدِ انقطعَ ذلكَ بالموْتِ.
  • فإنْ عُدمَ مالُ الميِّتِ ومنْ تلزمُهُمْ نفقتُهُ: فمنْ بيْتِ المالِ، إذَا كانَ مسلمًا.
  • فإنْ لمْ يكنْ: فعلَى المسلمينَ العالِمِينَ بحالِهِ. قالَ الشّيخُ تقِيُّ الدّينِ: مَنْ ظنَّ أنَّ غيرَهُ لَا يقومُ بهِ تعيّنَ عليهِ(٣).
  • فإنْ أرادَ بعضُ الورثَةِ أنْ ينفردَ بهِ لمْ يلزمْ بقيَّةَ الورثَةِ قَبولُهُ،
  • لكنْ ليسَ للبقيَّةِ نبشُهُ وسلبُهُ منْ كفنِهِ بعدَ دفنِهِ.

وإذَا ماتَ إنسانٌ معَ جماعَةٍ فِي سفرٍ:

  • كفّنُوهُ منْ مالِهِ،
  • فإنْ لمْ يكنْ: كفّنُوهُ ورجعُوا علَى تركتِهِ، أوْ مَنْ تلزمُهُ نفقتُهُ إنْ نوَوُا الرّجوعَ.

(ويُستحبُّ تكفينُ رجلٍ فِي:

  • ثلاثِ لفائفَ بِيضٍ) منْ قطنٍ؛ لقولِ عائشةَ ږ : «كُفِّنَ رسولُ اللهِ چ فِي ثلاثَةِ أثوابٍ بِيضٍ سُحوليَّةٍ جدُدٍ يمانيَّةٍ، ليسَ فِيهَا قميصٌ ولَا عمامَةٌ أُدرجَ فِيهَا إدراجًا» متّفقٌ عليهِ(٤).
٤٢٨
  • ويقدَّمُ بتكفينٍ مَنْ يقدَّمُ بغَسلٍ(٥)، ونائبُهُ كهوَ، والأَوْلَى تولّيهِ بنفسِهِ،
  • (تُجمَّرُ)؛ أيْ: تُبخَّرُ بعدَ رشِّهَا بماءِ وردٍ أوْ غيرِهِ؛ ليَعلَقَ،
  • (ثمَّ تُبسطُ بعضُهَا فوقَ بعضٍ) أوسعُهَا وأحسنُهَا أعلاهَا؛ لأنَّ عادَةَ الحيِّ جعلُ الظّاهرِ أفخرَ ثيابِهِ،
  • (ويُجعلُ الحَنُوْطُ) وهوَ أخلاطٌ منْ طيبٍ يُعَدُّ للميِّتِ خاصَّةً (فيمَا بينَهَا)؛
  • لَا فوقَ العليَا؛ لكراهَةِ عمرَ وابنِهِ وأبِي هريرةَ ښ (٦).
  • (ثمَّ يوضعُ) الميِّتُ (عَلَيْهَا)؛ أيِ اللّفائفِ (مستلقيًا)؛ لأنَّهُ أمكنُ لإدراجِهِ فِيهَا،
  • (ويُجعلُ مِنهُ)؛ أيْ مِنَ الحنوطِ (فِي قطنٍ: بينَ أليتيْهِ)؛ ليردَّ مَا يخرجُ عندَ تحريكِهِ، (ويُشدُّ فوقَهَا خِرقَةٌ مشقوقَةُ الطّرَفِ كالتُّبَّانِ) وهوَ: السّراويلُ بلَا أكمامٍ (تجمعُ أليتيْهِ ومثانتَهُ،
  • ويُجعلُ الباقِي) مِنَ القطنِ المحنَّطِ:
٤٢٩
  • (علَى منافذِ وجهِهِ) عينيْهِ ومنخريِهِ وأذنيْهِ وفمِهِ؛ لأنَّ فِي جعلِهَا علَى المنافذِ منعًا منْ دخولِ الهوامِّ،
  • (و) علَى (مواضعِ سجودِهِ) ركبتيْهِ ويديْهِ وجبهتِهِ وأنفِهِ وأطرافِ قدميْهِ؛ تشريفًا لهَا،
  • وكذَا: مغابنُهُ؛ كطيِّ ركبتيْهِ، وتحْتِ إبْطيْهِ، وسُرّتِهِ؛ لأنَّ ابنَ عمرَ كانَ يتتبّعُ مغابنَ الميِّتِ ومرافقَهُ بالمسكِ(٧)،
  • (وإنْ طُيِّبَ) الميِّتُ (كلُّهُ فحسنٌ)؛ لأنَّ أنسًا ڔ طُلِيَ بالمسكِ(٨)، وطلَى ابنُ عمرَ ژ ميِّتًا بالمسكِ(٩)،
  • وكُرهَ:
  • داخلَ عينيْهِ،
  • وأنْ يُطيَّبَ بوَرْسٍ وزعفرانٍ،
  • وطليُهُ بمَا يمسكُهُ كصَبِرٍ -مَا لمْ يُنْقَلْ-.
  • (ثمَّ يَردُّ طرَفَ اللِّفافَةِ العليَا) مِنَ الجانبِ الأيسرِ (علَى شِقِّهِ الأيمنِ، ويَردُّ طرَفَهَا الآخرَ فوقَهُ)؛ أيْ فوقَ الطّرَفِ الأيمنِ،
٤٣٠
  • (ثمَّ) يَفعلُ (الثّانيَةَ(١٠) والثّالثَةَ كذلكَ)؛ أيْ: كالأُولَى،
  • (ويَجعلُ أكثرُ الفاضلِ) منْ كفنِهِ (عندَ رأسِهِ)؛ لشرفِهِ،
  • ويُعيدُ الفاضلَ علَى وجهِهِ ورجليْهِ بعدَ جَمعِهِ؛ ليصيرَ الكفنُ كالكيسِ فلَا ينتشرُ،
  • (ثمَّ يعقِدُهَا)؛ لئلَّا تنتشرَ،
  • (وتُحلُّ فِي القبرِ)؛ لقولِ ابنِ مسعودٍ ڔ : «إذَا أدخلْتُمُ الميِّتَ القبرَ فحُلُّوا العُقدَ» رواهُ الأثرمُ(١١).

وكُرهَ تخريقُ اللّفائفِ؛ لأنَّهُ إفسادٌ لهَا.

(وإنْ كُفِّنَ فِي: قميصٍ، ومئزرٍ، ولِفافَةٍ: جازَ)؛

  • «لأنَّهُ چ ألبسَ عبدَ اللهِ بنَ أُبَيٍّ قميصَهُ لمَّا ماتَ» رواهُ البخارِيُّ(١٢).
  • وعنْ عمرِو بنِ العاصِ ڔ : «أنَّ الميِّتَ يُؤزَّرُ ويُقمَّصُ ويُلفُّ بالثّالثَةِ»(١٣)،
٤٣١
  • وهذَا عادَةُ الحيِّ،
  • ويكونُ القميصُ بكُمَّيْنِ ودَخَاريصَ، لَا بِزِرٍّ(١٤).

(وتُكفَّنُ المرأةُ) والخنثَى ندبًا (فِي خمسَةِ أثوابٍ) بِيضٍ منْ قطنٍ،

  • (إزارٍ،
  • وخمارٍ،
  • وقميصٍ،
  • ولِفافتَيْنِ)؛
  • لمَا روَى أحمدُ وأبُو داودَ -وفيهِ ضعفٌ- عنْ ليلَى الثّقفيَّةِ ږ قالتْ: «كنْتُ فيمَنْ غسّلَ أمَّ كلثومٍ بنتَ رسولِ اللهِ چ ، فكانَ أوّلَ مَا أعطانَا: الحِقَا(١٥)، ثمَّ الدِّرعُ، ثمَّ الخمارُ، ثمَّ المِلحفَةُ، ثمَّ أُدرجَتْ بعدَ ذلكَ فِي الثّوبِ الآخِر»(١٦). قالَ
٤٣٢
  • أحمدُ: «الحِقَا: الإزارُ، والدِّرْعُ: القميصُ»(١٧)، فتُؤزَّرُ بالمئزرِ، ثمَّ تُلْبَسُ القميصَ، ثمَّ تُخمَّرُ، ثمَّ تُلفُّ باللِّفافتَيْنِ.

ويُكفَّنُ:

  • صبيٌّ فِي ثوبٍ،
  • ويُباحُ: فِي ثلاثَةٍ مَا لمْ يرثْهُ غيرُ مكلّفٍ،
  • وصغيرَةٌ فِي قميصٍ ولِفافتَيْنِ.

(والواجبُ) للميِّتِ مطلقًا: (ثوبٌ يسترُ جميعَهُ)؛ لأنَّ العورَةَ المغلّظَةَ يُجزئُ فِي سترِهَا ثوبٌ واحدٌ، فكفَنُ الميّتِ أَولَى،

  • وكُرهَ: بصوفٍ وشعرٍ.
  • ويحرُمُ: بجلودٍ.
  • ويجوزُ فِي حريرٍ لضرورَةٍ فقطْ.

فإنْ لمْ يجدْ إلَّا بعضَ ثوبٍ: ستَرَ العورةَ؛ كحالِ الحيَاةِ، والباقِي بحشيشٍ أوْ ورقٍ.

وحرُمَ: دفنُ حُلِيٍّ وثيابٍ غيرِ الكفنِ؛ لأنَّهُ إضاعَةُ مالٍ.

ولِحيٍّ أخذُ كفنِ ميِّتٍ لحاجَةِ حرٍّ أوْ بردٍ بثمنِهِ.

ڭ ڭ ڭ

(١) سبق تخريجه في (ص٤٢٢).

(٢) في (الأصل، ز، س، ل) الفاء من الشرح، والمثبت هو الموافق لما في زاد المستقنع (ص٩٠١ت: القاسم).

(٣) نقله في: الفروع، للشمس ابن مفلح (٣/٣١٥)، الاختيارات، للبعلي (ص١٢٩).

(٤) أخرجه أحمد (٦/١١٨)، والبخاري (١٢٦٤، ١٣٨٧)، ومسلم (٩٤١).

(٥) وتفصيله عند قوله: «وأوْلَى النّاسِ بغسلِهِ..» في (ص٤١٢).

(٦) أثر عمر أخرجه ابن أبي شيبة (٣/٢٥٧) عنه ڔ قال: (لا تحنِّطوني بمسك)، وبنحوه أخرجه ابن المنذر في الأوسط (٥/٣٩٧).

وأثر ابن عمر أخرجه ابن أبي شيبة (٣/٢٧٠) أنه كره الحنوط على النعش.

وأثر أبي هريرة أخرجه ابن المنذر في الأوسط (٥/٥٠٦) أن أبا هريرة أوصى أهله حين توفي أن لا يُظهِروا عليه الطيب.

(٧) أخرجه عبدالرزاق (٣/٤١٤)، وابن المنذر في الأوسط (٥/٣٩٥).

(٨) أخرج ابن أبي شيبة (٣/٢٥٦) ومن طريقه ابن المنذر في الأوسط (٥/٣٩٥-٣٩٦) عن حميد الطويل أنه جُعل في حنوطه صرة من مسك أو سك فيه شعرٌ من شعرِ النبي چ .

(٩) أخرج ابن أبي شيبة (٣/٢٥٧) عن ابن علية عن أيوب عن نافع عن ابن عمر أنه حنَّط ميتا بمسك. وأخرج عبدالرزاق (٣/٤١٤) ومن طريقه ابن المنذر في الأوسط (٥/٣٩٥) عن معمر عن أيوب به نحوه وزاد: يذرُّه عليه ذرًّا.

(١٠) في (د، ز): «بالثانية»، إلا أن الباء في (ز) من الشرح.

(١١) لم نجده عن ابن مسعود، وأخرج ابن أبي شيبة (٣/٣٢٦)، وأبو داود في المراسيل (٤١٩)، والبيهقي (٣/٤٠٧) عن خلف بن خليفة عن أبيه بلغه: أن رسول الله چ وضع نعيم بن مسعود في القبر ونزع الأخِلَّةَ بفيه، يعني العقد. وفي بعض طرقه قال خلف: أظنه –أي أباه- سمعه من مولاه معقل بن يسار.

ضعَّفه النووي في الخلاصة (٣٤١٠).

(١٢) أخرجه البخاري (١٢٧٠)، ومسلم (٢٧٧٣) من حديث جابر بن عبدالله ڔ .

(١٣) لم نجده عن عمرو بن العاص؛ وأخرج مالك في الموطأ (٥٩٩) ومن طريقه عبدالرزاق في المصنف (٣/٤٢٦)، وكذلك البيهقي (٣/٤٠٢) من حديث عبدالله بن عمرو بن العاص بلفظ المصنف ولعله ما أراد.

(١٤) في (د، ز، ع، ل، م) «لا يُزّرّ».

(١٥) هكذا مقصوراً في نسخنا المعتمدة، وهو الموافق لأكثر نسخ السنن لأبي داود (٥/ ٢٨٤ ط. التأصيل)، قال العيني في شرح السنن (٦/ ٨٧): («الحقا» بكسر الحاء المهملة مقصور، ولعله لغة في الحقو).

(١٦) أخرجه أحمد (٦/٣٨٠)، وأبو داود (٣١٥٧) من حديث نوح بن حكيم، عن رجل من بني عروة بن مسعود يقال له: داود، عن ليلى الثقفية به.

قال المنذري في مختصر السنن (٤/٣٠٤): (وفيه أيضًا من ليس بالمشهور، والصحيح أن هذه القصة في زينب، لأن أم كلثوم توفيت، ورسول الله چ غائب ببدر)، وضعَّفه ابن القطان في بيان الوهم والإيهام (٥/٥٢-٥٤)، وحسَّنه النووي في الخلاصة (٣٣٩٠).

(١٧) انظر: المبدع، للبرهان ابن مفلح (٢/٢٤٨).
