٤٠٣

(كتابُ الجنائزِ)

بفتحِ الجيمِ، جمعُ جِنازَةٍ بالكسرِ(١)، -والفتحُ(٢) لغَةٌ-: اسمٌ للميِّتِ أوْ للنّعشِ عليهِ ميِّتٌ، فإنْ لمْ يكنْ عليهِ ميِّتٌ فلَا يُقالُ: نعشٌ ولَا جِنازةٌ، بلْ سريرٌ، قالَهُ الجوهرِيُّ(٣).

واشتقاقُهُ مِنْ: جَنَزَ إذَا سَتَرَ.

وذكرَهُ هنَا لأنَّ أهمَّ مَا يُفعلُ بالميِّتِ الصّلَاةُ.

ويُسنُّ: الإكثارُ منْ ذكرِ الموْتِ، والاستعدادُ لهُ؛ لقولِهِ چ : «أكثِرُوا منْ ذكرِ هاذِمِ اللّذّاتِ»(٤)، هوَ بالذّالِ المعجَمَةِ.

ويُكرهُ: الأنينُ وتمنِّي الموتِ.

٤٠٤

ويُباحُ التّداوِي بمباحٍ، وتركُهُ أفضلُ،

  • ويحرُمُ بمحرَّمٍ، مأكولٍ وغيرِهِ منْ صوْتِ مِلهاةٍ وغيرِهِ.
  • ويجوزُ ببولِ إبِلٍ فقطْ، قالَهُ فِي المبدعِ(٥).

ويُكرهُ أنْ يَسْتَطِبَّ مسلمٌ ذمّيًّا لغيرِ ضرورَةٍ، وأنْ يأْخذَ مِنهُ دواءً لمْ يبيِّنْ مفرداتِهِ المباحةَ.

و(تُسنُّ عيادَةُ المريضِ) والسُّؤالُ عنْ حالِهِ؛ للأخبارِ(٦)،

  • ويُغِبُّ بِهَا،
  • وتكونُ بُكرةً أوْ عشيًّا،
٤٠٥
  • ويأخذُ بيدِهِ، ويقولُ: «لا بأس ، طَهورًا(٧) إنْ شاءَ اللهُ تعالَى»؛ لفعلِهِ چ (٨)،
  • ويُنفِّسُ لهُ فِي أجلِهِ؛ لخبرٍ رواهُ ابنُ ماجهْ عنْ أبِي سعيدٍ ڔ : «فإنَّ ذلكَ لَا يردُّ شيئًا»(٩)،
  • ويدعُو لهُ بمَا وردَ.

(و) يُسنُّ:

  • (تذكيرُهُ التّوبَةَ)؛ لأنَّهَا واجبَةٌ علَى كلِّ حالٍ، وهوَ أحوجُ إلَيْهَا منْ غيرِهِ.
  • (والوصيَّةُ)؛ لقولِهِ چ : «مَا حقُّ امرئٍ مسلمٍ لهُ شيءٌ يوصِي بهِ يبيتُ ليلتَيْنِ إلَّا ووصيّتُهُ مكتوبَةٌ عندَهُ» متّفقٌ عليهِ عنِ ابنِ عمرَ ڔ (١٠).

(وإذَا نُزِلَ بهِ)؛ أيْ: نَزَلَ بهِ الملَكُ لقبضِ رُوحِهِ (سُنَّ:

  • تعاهدُ) أرفقِ أهلِهِ وأتقاهُمْ لربِّهِ (بلَّ حلْقِهِ بماءٍ أوْ شرابٍ،
٤٠٦
  • وندْيَ(١١) شَفتيْهِ(١٢))؛ لأنَّ ذلكَ يُطفئُ مَا نزلَ بهِ مِنَ الشِّدّةِ، ويسهِّلُ عليهِ النُّطقَ بالشّهادةِ،
  • (ولَقْنُهُ(١٣) لَا إلهَ إلَّا اللهُ) لقولِهِ چ : «لقِّنُوا موتَاكُمْ لَا إلهَ إلَّا اللهُ» رواهُ مسلمٌ عنْ أبِي سعيدٍ ڔ (١٤)،
  • (مرَّةً ولمْ يزدْ علَى ثلاثٍ)؛ لئلَّا يُضجرَهُ،
  • (إلَّا أنْ يتكلّمَ بعدَهُ فيُعيدُ تلقينَهُ)؛ ليكُونَ آخرُ كلامِهِ لَا إلهَ إلَّا اللهُ،
  • ويكونُ (برفقٍ)؛ أيْ: بلطفٍ ومدارَاةٍ؛ لأنَّهُ مطلوبٌ فِي كلِّ موضعٍ فهنَا أَولَى.
  • (وَيَقْرَأُ عندَهُ) سورَةَ (يس)؛
  • لقولِهِ چ : «اقرأُوا علَى موتاكُمْ سورَةَ يس» رواهُ أبُو داودَ(١٥).
  • ولأنَّهُ يُسهِّلُ خروجَ الرُّوحِ.
٤٠٧
  • ويقرأُ عندَهُ أيضًا الفاتحةَ.
  • (ويوجِّهُهُ إلَى القبلةِ)؛ لقولِهِ چ عنِ البيْتِ الحرامِ: «قبلتُكُمْ أحياءً وأمواتًا» رواهُ أبُو داودَ(١٦)،
  • وعلَى جنبِهِ الأيمنِ أفضلُ إنْ كانَ المكانُ واسعًا،
  • وإلَّا فعلَى ظهرِهِ مستلقيًا ورجلاهُ إلَى القبلةِ، ويرفعُ رأسَهُ قليلًا ليصيرَ وجهُهُ إلَى القبلةِ.

(فإذَا ماتَ سُنَّ:

  • تغميضُهُ)؛ لأنَّهُ چ أغمضَ أبَا سلمَةَ ڔ وقالَ: «إنَّ الملائكَةَ يؤمِّنُونَ علَى مَا تقولُونَ» رواهُ مسلمٌ(١٧)، ويقولُ: بسمِ اللهِ وعلَى وفَاةِ(١٨) رسولِ اللهِ چ ،
  • ويُغمِضُ ذاتَ مَحرمٍ وتُغمِضُهُ،
٤٠٨
  • وكُرهَ منْ حائضٍ وجُنبٍ، وأنْ يَقرَباهُ،
  • ويُغمِضُ(١٩) الأنثَى مثلُهَا أوْ صبيٌّ.
  • (وشَدُّ لَحييْهِ)؛ لئلَّا يدخُلَهُ الهوامُّ،
  • (وتليِينُ مفاصِلِهِ)؛ ليَسهُلَ تغسيلُهُ،
  • فيردُّ ذراعيْهِ إلَى عَضُديْهِ، ثمَّ يردُّهُمَا إلَى جنبِهِ(٢٠)، ثمَّ يردُّهُمَا،
  • ويردُّ ساقيْهِ إلَى فخذيْهِ، وهمَا إلَى بطنِهِ ثمَّ يردُّهُمَا،
  • ويكونُ ذلكَ عقِبَ موتِهِ قبلَ قسوتِهَا، فإنْ شقَّ ذلكَ ترَكَهُ.
  • (وخلْعُ ثيابِهِ)؛ لئلَّا يحمَى جسدُهُ فيسرعَ إليهِ الفسادُ.
  • (وسترُهُ بثوبٍ)؛ لمَا روَتْ عائشةُ ږ : «أنَّ النَّبِيَّ چ حينَ تُوفِّيَ سُجِّيَ ببُردِ حِبَرةٍ» متّفقٌ عليهِ(٢١).
  • وينبغِي أنْ يُعطفَ فاضلُ الثّوبِ عندَ رأسِهِ ورجليْهِ؛ لئلَّا يرتفعَ بالرِّيحِ.
  • (ووضعُ حديدَةٍ) أوْ نحوِهَا (علَى بطنِهِ)؛
  • لقولِ أنسٍ ڔ : «ضعُوا علَى بطنِهِ شيئًا منْ حديدٍ»(٢٢)؛
  • ولئلَّا ينتفخَ بطنُهُ،
٤٠٩
  • (ووضعُهُ علَى سريرِ غَسلِهِ)؛ لأَنَّهُ يبعدُ عنِ الهوامِّ،
  • (متوجِّهًا) إلَى القبلَةِ علَى جنبِهِ الأيمنِ،
  • (منحدِرًا نحوَ رجليْهِ)؛ أيْ: يكونُ رأسُهُ أعلَى مِنْ رجليْهِ؛ ليَنصبَّ عنهُ الماءُ ومَا يخرجُ مِنهُ،
  • (وإسراعُ تجهيزِهِ إنْ ماتَ غيرَ فجأَةٍ)؛ لقولِهِ چ : «لَا ينبغِي لجيفَةِ مسلمٍ أنْ تُحبسَ بينَ ظهرانَيْ أهلِهِ» رواهُ أبُو داودَ(٢٣)،
  • ولَا بأسَ أنْ يُنتظرَ بهِ مَنْ يحضُرُهُ مِنْ وليِّهِ وغيرِهِ إنْ كانَ قريبًا ولمْ يُخْشَ أوْ يشُقَّ(٢٤) علَى الحاضرِينَ،
  • فإنْ ماتَ فجأةً أوْ شُكَّ فِي موتِهِ: انتُظرَ بهِ حتَّى يُعلَمَ موتُهُ؛
  • بانخسافِ صُدْغيْهِ،
  • وميلِ أنفِهِ،
٤١٠
  • وانفصالِ كفّيْهِ،
  • واسترخاءِ رجليْهِ.
  • (وإنفاذُ وصيّتِهِ)؛ لمَا فيهِ منْ تعجيلِ الأجرِ،
  • (ويجبُ) الإسراعُ (فِي قضاءِ دَيْنِهِ) سواءٌ كانَ للهِ تعالَى أوْ لآدمِيٍّ؛ لمَا روَى الشّافعيُّ وأحمدُ والتّرمذيُّ وحسّنهُ عنْ أبِي هريرةَ ڔ مرفوعًا: «نفسُ المؤمنِ معلَّقَةٌ بدَيْنِهِ حتَّى يُقضَى عنهُ»(٢٥)،

ولَا بأسَ بتقبيلِهِ والنّظرِ إليهِ ولوْ بعدَ تكفينِهِ.

ڭ ڭ ڭ

(١) في (الأصل): «وبالكسر».

(٢) في (د، ع): «بالفتح والكسر».

(٣) انظر: الصحاح (٣/٨٧٠).

(٤) أخرجه أحمد (٢/٢٩٢-٢٩٣)، والترمذي (٢٣٠٧)، وابن ماجه (٤٢٥٨)، والنسائي (٤/٤) من حديث محمد بن عمرو عن أبي سلمة عن أبي هريرة ڔ .

قال الترمذي: (حسن غريب)، وصححه ابن حبان (٢٩٩٢)، والحاكم (٤/٣٢١)، والنووي في الخلاصة (٣١٥٠)، وضعَّفه ابن الجوزي في العلل المتناهية (٢/٤٠١).

والحديث اختلف فيه على محمد بن عمرو؛ فرُويَ عنه عن أبي سلمة مرسلًا، أخرجه ابن أبي شيبة (١٣/٢٢٥)، ورجَّح إرساله أحمد في مسائل أبي داود (١٩٢٢)، والدارقطني في العلل (س١٣٩٧).

(٥) المبدع (٢/٢١٧).

(٦) أما عيادته فقد أخرج أحمد (٢/٥٤٠)، والبخاري (١٢٤٠)، ومسلم (٢١٦٢) من حديث أبي هريرة ڔ عن النبي چ قال: «حق المسلم على المسلم خمس» وذكر منها «وعيادة المريض».

وأما السؤال عن حاله: فأخرج الترمذي (٩٨٣) وابن ماجه (٤٢٦١) من حديث ثابت البناني عن أنس بن مالك ڔ أن النبي چ دخل على شاب وهو في الموت، فقال: «كيف تجدك؟» الحديث.

قال الترمذي: (حديث غريب). ورجَّح إرساله البخاري في العلل الكبير للترمذي (٢٤٤)، وأبو حاتم في العلل لابنه (١٨٠٦)، والدارقطني في علله (٢٣٦٨).

ومما ورد في السؤال عن حال المريض: ما جاء في حديث الإفك لما اشتكت أم المؤمنين ږ فقالت: (فقدمنا المدينة، فاشتكيت حين قدمت شهرًا ... وهو يريبني في وجعي أني لا أعرف من رسول الله چ اللطف الذي كنت أرى منه حين أشتكي، إنما يدخل علي رسول الله چ فيسلم، ثم يقول: «كيف تيكم») أخرجه البخاري (٤١٤١)، ومسلم (٢٧٧٠).

(٧) في (ز، س): «طهورٌ».

(٨) أخرجه البخاري (٣٦١٦) من حديث ابن عباس، وليس في شيء من طرقه ذكر أخذ اليد أو نحوه.

(٩) أخرجه الترمذي (٢٠٨٧)، وابن ماجه (١٤٣٨) من حديث موسى بن محمد بن إبراهيم عن أبيه عن أبي سعيد الخدري ڔ به.

وأُعل هذا الحديث بعلتين : فقيل في موسى: منكر الحديث، والثانية عدم سماع والده من أبي سعيد، بذا أعلَّه البخاري في العلل الكبير للترمذي (٥٩١)، وأبو حاتم في العلل لابنه (س٢٢١٤)، وقال الترمذي: (غريب).

(١٠) أخرجه أحمد (٢/٥٠)، والبخاري (٢٧٣٨)، ومسلم (١٦٢٧).

(١١) في (ب، د): «ندَّى)، وفي (ز، ل، م): «يندي».

(١٢) في (ب، د، س، ض، ع، ل، م): «شفتيه بقطنة»، وهو الموافق لما في زاد المستقنع (ص١٠٣ت: القاسم).

(١٣) في (ع، ل، م): «تلقينه».

(١٤) أخرجه أحمد (٣/٣)، ومسلم (٩١٦).

(١٥) أخرجه أحمد (٥/٢٦-٢٧)، وأبو داود (٣١٢١)، وابن ماجه (١٤٤٨) من حديث أبي عثمان –وليس بالنهدي- عن أبيه عن معقل بن يسار ڔ مرفوعًا.

صححه ابن حبان (٣٠٠٢)، وضعَّفه ابن القطان في بيان الوهم والإيهام (٥/٤٩-٥٠)، والنووي في الخلاصة (٣٢٧٨)، ونقل ابن الملقن في البدر المنير (٥/١٩٥) عن أبي بكر بن العربي عن الدارقطني أنه قال: (حديث ضعيف الإسناد مجهول المتن، ولا يصح في الباب حديث).

(١٦) أخرجه أبو داود (٢٨٧٥) والنسائي (٧/٨٩) من حديث عبدالحميد بن سنان عن عبيد بن عمير عن أبيه –وكانت له صحبة- عن رسول الله چ أن رجلًا سأله فقال: يا رسول الله ما الكبائر؟ فقال: «هنَّ تسعٌ» وذكر منهن: «واستحلال البيت الحرام قبلتكم أحياء وأمواتا».

صححه الحاكم (١/٥٩)، وقال البخاري: (عبدالحميد بن سنان عن عبيد بن عمير، في حديثه نظر) أي هذا الحديث كما قاله العقيلي في الضعفاء (٣/٥١٦) وأشار إلى ضعفه، وكذا الذهبي في ترجمته من الميزان (٢/٤٨٠).

(١٧) أخرجه أحمد (٦/٢٩٧)، ومسلم (٩٢٠) من حديث أم المؤمنين أم سلمة ږ .

(١٨) في (ز): «ملة».

(١٩) في (د، م): «وتغمض».

(٢٠) في (ض): « جنبيه».

(٢١) أخرجه أحمد (٦/٨٩)، والبخاري (٥٨١٤)، ومسلم (٩٤٢).

(٢٢) أخرجه ابن حبان في الثقات (٤/٢٨)، والبيهقي (٣/٣٨٥) من حديث أنس بن مالك ڔ ، وفي إسناده محمد بن عقبة، قال الذهبي في المهذب (٣/١٣١٨): (محمد ضعيف).

(٢٣) أخرجه أبو داود (٣١٥٩) ومن طريقه البيهقي (٣/٣٨٦-٣٨٧) من حديث عروة بن سعيد الأنصاري عن أبيه عن الحصين بن وَحْوَح به مرفوعًا.

ضعفَّه عبد الحق في الأحكام الوسطى (٢/١٢٥)، ووافقه ابن القطان في بيان الوهم والإيهام (٢/٥٥٦، ٣/٤١٠)، وابن كثير في إرشاد الفقيه (١/٢١٩) وقال الذهبي في المهذَّب (٣-١٣٢٠): (غريبٌ جدًا، وفي سنده سعيد، مجهول). وحسَّن إسناده ابن حجر في الفتح (٣/١٨٤).

وثبت الأمر بالتعجيل بالجنازة من حديث أبي هريرة عند البخاري (١٣١٥)، ومسلم (٩٤٤) ويأتي لفظه عن المصنف في (حمل الميت ودفنه).

(٢٤) في (د، ز، ل): «لم يخش عليه أو يشق»، في (ب، ض): «لم يخش أن يشق».

(٢٥) أخرجه الشافعي (مسنده ٢/١٩٠)، وأحمد (٢/٤٤٠)، والترمذي وحسَنه (١٠٧٨)، وابن ماجه (٢٤١٣) من حديث عمر بن أبي سلمة عن أبيه عن أبي هريرة ڔ .

صححه يحيى بن معين (انظر: التمهيد ضمن موسوعة شروح ١٢/٤٠٦)، وابن حبان (٣٠٦١).
