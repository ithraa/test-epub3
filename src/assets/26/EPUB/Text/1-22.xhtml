٣٠٤

(فصلٌ) فِي موقفِ الإمامِ والمأمومِينَ

السُّنَّةُ أنْ: (يقفَ المأمومُونَ) رجالًا كانُوا، أوْ نساءً، إنْ كانُوا اثنيْنِ فأكثرَ (خلفَ الإمامِ)؛ لفعلِهِ(١) چ : كانَ إذَا قامَ إلَى الصّلاةِ، قامَ أصحابُهُ ښ خلفَهُ.

  • ويُستثنَى مِنهُ:
  • إمامُ العرَاةِ يقفُ وسْطَهُمْ: وجوبًا.
  • والمرأَةُ إذَا أمَّتِ النّساءَ تقفُ وسْطَهُنَّ: استحبابًا؛ ويأتِي(٢).

(ويصحُّ) وقوفُهُمْ (معَهُ)؛ أيْ: معَ الإمامِ، (عنْ يمينِهِ، أوْ عنْ جانبيْهِ)؛ لأنَّ ابْنَ مسعودٍ ڔ صلَّى بينَ علقمَةَ والأسودِ، وقالَ: هكذَا رأيْتُ النَّبِيَّ چ فعلَ، رواهُ أحمدُ، وقالَ ابنُ عبدِ البرِّ: «لَا يصحُّ رفعُهُ، والصّحيحُ أنَّهُ منْ قولِ ابنِ مسعودٍ ڔ »(٣)(٤).

  • (لَا قُدّامَهُ)؛ أيْ: لَا قُدّامَ الإمامِ، فلَا تصحُّ للمأمومِ ولوْ بإحرامٍ؛ لأنَّهُ ليسَ موقفًا بحالٍ.
  • والاعتبارُ بمُؤَخَّرِ القدمِ، وإلَّا لمْ يضرَّ،
٣٠٥
  • وإنْ صلَّى قاعدًا: فالاعتبارُ بالأَلْيَةِ، حتَّى لوْ مدَّ رجليْهِ، وقدّمَهمَا علَى الإمامِ: لمْ يضرَّ،
  • وإنْ كانَ مضطجعًا: فبالجَنْبِ.

وتصحُّ داخلَ الكعبَةِ:

  • إذَا جعلَ وجهَهُ إلَى وجْهِ إمامِهِ،
  • أوْ ظهرَهُ إلَى ظهرِهِ،
  • لَا إنْ جعلَ ظهرَهُ إلَى وجْهِ إمامِهِ؛ لأنَّهُ متقدِّمٌ عليهِ.

وإنْ وقفُوا حولَ الكعبَةِ مستديرِينَ: صحَّتْ، فإنْ كانَ المأمومُ فِي جهتِهِ أقربَ مِنَ الإمامِ فِي جهتِهِ: جازَ،

  • إنْ لمْ يكونَا فِي جهَةٍ واحدَةٍ: فتبطلُ صلَاةُ المأمومِ.

ويُغتفرُ التّقدُّمُ: فِي شدَّةِ خوفٍ، إذَا أمكنَ المتابعَةُ.

(ولَا) تصحُّ للمأمومِ إنْ وقفَ (عنْ يسارِهِ فقطْ)؛ أيْ: معَ خلوِّ يمينِهِ إذَا صلَّى ركعَةً فأكثرَ؛ لأنَّهُ چ أدارَ ابْنَ عبّاسٍ وجابرًا ښ عنْ يسارِهِ إلَى يمينِهِ(٥)،

٣٠٦

  • وإذَا(٦) كبّرَ عنْ يسارِهِ: أدارَهُ منْ ورائِهِ إلَى يمينِهِ،
  • فإنْ كبّرَ معَهُ آخرُ: وقفَا خلفَهُ،
  • فإنْ كبّرَ الآخرُ عنْ يسارِهِ: أدارهمَا بيدِهِ وراءَهُ،
  • فإنْ شقَّ ذلكَ، أوْ تعذّرَ تقدَّمَ الإمامُ: فصلَّى بينَهُمَا، أوْ عنْ يسارِهمَا.
  • ولوْ تأخّرَ الأيمنُ قبلَ إحرامِ الدّاخلِ ليصلّيَا خلفَهُ: جازَ.
  • ولوْ أدركَهمَا الدّاخلُ جالسَيْنِ: كبّرَ وجلسَ عنْ يمينِ صاحبِهِ، أوْ يسارِ الإمامِ، ولَا تأخُّرَ إذًا؛ للمشقَّةِ،
  • فالزَّمْنَى لَا يتقدّمُونَ، ولَا يتأخّرُونَ.

(ولَا) تصحُّ صلَاةُ (الفذِّ)؛ أيِ: الفردِ (خلفَهُ)؛ أيْ: خلفَ الإمامِ، (أوْ خلفَ الصّفِّ)، إنْ صلَّى ركعةً فأكثرَ، عامدًا، أوْ ناسيًا، عالمًا، أوْ جاهلًا؛

  • لقولِهِ چ : «لَا صلَاةَ لفردٍ(٧) خلفَ الصّفِّ»، رواهُ أحمدُ وابنُ ماجهْ(٨)،
  • ورأَى(٩) چ رجلًا يصلِّي خلفَ الصّفِّ، فأمرَهُ أنْ يُعيدَ الصّلاةَ،
٣٠٧
  • رواهُ أحمدُ، والتّرمذِيُّ وحسّنَهُ، وابنُ ماجهْ، وإسنادُهُ ثقاتٌ(١٠).
  • (إلَّا أنْ يكونَ) الفذُّ خلفَ الإمامِ، أوِ الصّفِّ (امرأَةً) خلفَ رجلٍ: فتصحُّ صلاتُهَا؛ لحديثِ أنسٍ ڔ (١١)،

وإنْ وقفَتْ:

  • بجانبِ الإمامِ: فكرجلٍ،
  • وبصفِّ رجالٍ: لمْ تَبطلْ صلَاةُ مَنْ يليهَا، أوْ خلفَهَا، فصفٌّ تامٌّ منْ نساءٍ لَا يمنعُ اقتداءَ مَنْ خلفَهُنَّ منْ رجالٍ.

(وإمامَةُ النّسـاءِ: تقـفُ فِي صفِّهِنَّ) ندبًا؛ رُويَ عنْ عائشةَ وأمِّ سلمَةَ ژ (١٢)،

٣٠٨

  • فإنْ أمَّتْ واحدَةً؛ وقفَتْ عنْ يمينِهَا، ولَا يصحُّ خلفَهَا.

(ويليهِ)؛ أيِ: الإمامَ مِنَ المأمومِينَ:

  • (الرّجالُ) الأحرارُ ثمَّ العبيدُ الأفضلُ فالأفضلُ؛ لقولِهِ چ : «ليَلِيْنِي منْكُمْ أولُو الأحلامِ والنُّهَى»، رواهُ مسلمٌ(١٣)،
  • (ثمَّ الصّبيانُ): الأحرارُ، ثمَّ العبيدُ،
  • (ثمَّ النّساءُ)؛ لقولِهِ چ : «أخِّروهُنَّ منْ حيثُ أخّرَهُنَّ اللهُ»(١٤)،
  • ويُقدَّمُ منْهُنَّ:
  • البالغاتُ: الأحرارُ، ثمَّ الأرقّاءُ،
  • ثمَّ مَنْ لمْ تبلُغِ: الأحرارُ(١٥)، فالأرقّاءُ، الفُضْلَى(١٦)، فالفُضْلَى،

وإنْ وقفَ الخَناثَى صفًّا: لمْ تصحَّ صلاتُهُمْ.

(كـ)ـالتّرتيبِ فِي (جنائزِهِمْ) إذَا اجتمعَتْ(١٧): فيقدَّمُونَ إلَى الإمامِ، وإلَى القبلَةِ فِي القبرِ، علَى مَا تقدَّمَ فِي صفوفِهِمْ.

(ومَنْ لمْ يقفْ معَهُ) فِي الصّفِّ (إلَّا:

  • كافرٌ،
٣٠٩
  • أوِ امرأَةٌ)،
  • أوْ خنثَى، وهوَ رجلٌ،
  • (أوْ مَنْ علِمَ حدثَهُ)، أوْ نجاستَهُ (أحدُهمَا)؛ أيِ: المصلِّي، أوِ المصافِفُ لهُ،
  • (أوْ) لمْ يقفْ معَهُ إلَّا (صبِيٌّ فِي فرضٍ:
  • ففذٌّ)؛ أيْ: فردٌ؛ فلَا تصحُّ صلاتُهُ، ركعَةً فأكثرَ.

وعُلِمَ مِنهُ صحَّةُ مصافَّةِ:

  • الصّبِيِّ فِي النّفلِ،
  • أوْ مَنْ جَهِلَ حدثَهُ، أوْ نجَسَهُ حتَّى فرَغَ،

(ومَنْ وجدَ فُرجَةً) بضمِّ الفاءِ؛ وهيَ: الخَلَلُ فِي الصّفِّ-ولوْ بعيدَةً-: (دخلَهَا)،

وكذَا إنْ وجدَ الصّفَّ غيرَ مرصوصٍ: وقفَ فيهِ؛

  • لقولِهِ چ : «إنَّ اللهَ وملائكتَهُ يُصَلُّونَ علَى الَّذِينَ يَصِلُونَ الصّفوفَ»(١٨)،

(وإلَّا) يجدْ فرجةً: وقفَ (عنْ يمينِ الإمامِ)؛ لأنَّهُ موقفُ الواحدِ،

٣١٠

  • (فإنْ لمْ يمكنْهُ: فلهُ أنْ ينبِّهَ مَنْ يقومُ معَهُ)؛ بنحنحةٍ، أوْ كلامٍ، أوْ إشارةٍ.
  • وكرِهَ بجذبِهِ،
  • ويتبعُهُ مَنْ نبّهَهُ وجوبًا.

(فإنْ صلَّى فذًّا ركعَةً: لمْ تصحَّ) صلاتُهُ؛ لمَا تقدَّمَ(١٩)، وكرّرَهُ لأجلِ مَا أعقبَهُ بهِ.

(وإنْ ركعَ فذًّا)؛ أيْ: فردًا، لعذرٍ؛ بأنْ خشِيَ فواتَ الرّكعَةِ،

  • (ثمَّ دخلَ فِي الصّفِّ) قبلَ سجودِ الإمامِ،
  • (أوْ وقفَ معَهُ آخرُ قبلَ سجودِ الإمامِ:
  • صحَّتْ) صلاتُهُ؛ لأنَّ أبَا بكرَةَ ڔ ركعَ دونَ الصّفِّ، ثمَّ مشَى حتَّى دخلَ الصّفَّ؛ فقالَ لهُ النَّبِيُّ چ : «زادَكَ اللهُ حرصًا، ولَا تَعُدْ»، رواهُ البخارِيُّ(٢٠)،
  • وإنْ فعلَهُ ولمْ يخشَ فواتَ الرّكعَةِ: لمْ تصحَّ، إنْ رفعَ الإمامُ رأسَهُ مِنَ الرّكوعِ قبلَ أنْ يَدخلَ الصّفَّ، أوْ يقفَ معَهُ آخرُ.

ڭ ڭ ڭ

(١) في (ب، د): «لأنه».

(٢) أي عند قوله: «(وإمامَةُ النساءِ: تقفُ فِي صفِّهِنَّ) ندبًا» في (ص٣٠٧).

(٣) انظر: التمهيد (١/٢٦٧).

(٤) أخرجه أحمد (١/٤١٤) ومسلم (٥٣٤) به مرفوعًا.

(٥) حديث ابن عباس أخرجه أحمد (١/٢٤٤-٢٤٥)، والبخاري (١١٧)، ومسلم (٧٦٣) في مبيته عند خالته ميمونة ښ ولفظه: (فقمت عن يساره فجعلني عن يمينه).

وحديث جابر أخرجه أحمد (٣/٣٥١)، ومسلم (٧٦٦) ولفظه: (فقمت خلفه فأخذ بأذني فجعلني عن يمينه).

(٦) في (ز): «فإذا».

(٧) في (ز): «لمنفرد»، وفي(ع، م): «لمفرد».

(٨) أخرجه أحمد (٤/٢٣)، وابن ماجه (١٠٠٣) من حديث علي بن شيبان ڔ .

صححه ابن خزيمة (١٥٦٩)، وقال ابن عبدالهادي في تنقيح التحقيق (٢/٤٩٩): (إسناده قوي) ثم نَقَل عن أحمد تحسينه.

(٩) في (س): «رأى» والواو مضروب عليها.

(١٠) أخرجه أحمد (٤/٢٢٨)، وأبو داود (٦٨٢)، والترمذي (٢٣٠)، وابن ماجه (١٠٠٤) من حديث وابصة بن معبد ڔ بنحوه.

صححه أحمد وإسحاق بن راهويه وابن المنذر (انظر: الأوسط ٤/٢٠٨-٢٠٩)، وأبو حاتم في العلل لابنه (س٢٧١)، واستظهر ابن رجب في الفتح (٥/٢٤) عن ابن معين ثبوته، وصححه ابن حبان (٢١٩٨).

وأعلَّه بعض أهل العلم بالاضطراب؛ وإليه مال الشافعي في الجديد، ورجَّح ذلك البزار وابن عبدالبر. وأنكر أحمد على من قال ذلك (انظر: فتح الباري لابن رجب ٥/٢٤).

(١١) أخرجه أحمد (٣/١٣١)، والبخاري (٣٨٠)، ومسلم (٦٥٨) ولفظه: فقام رسول الله چ وصففت أنا واليتيم وراءه والعجوز من ورائنا.

(١٢) أخرجه عنهما عبدالرزاق (٣/١٤٠-١٤١)، وابن أبي شيبة (٢/٨٩)، والبيهقي (٣/١٣١).

ونقل الزيلعي في نصب الراية (٢/٣١) عن النووي تصحيحه أثر عائشة.

(١٣) أخرجه أحمد (٤/١٢٢)، ومسلم (٤٣٢) من حديث أبي مسعود البدري ڔ .

(١٤) أخرجه عبدالرزاق في المصنف (٣/١٤٩)، وابن خزيمة (١٧٠٠)، والطبراني في المعجم الكبير (٩٤٨٤) عن ابن مسعود ڔ موقوفًا بنحوه، قال أبو بكر بن خزيمة: (الخبر موقوف غير مسند).

(١٥) في (ب): «من الحرائر»، وفي (ل): «من الأحرار».

(١٦) في (س): «الفضل»، وضبطها هكذا: «الفَضِلُ»، وفي (ل): «الأفضل فالفضلى»، وفي (م): «الأفضل فالأفضل».

(١٧) فصله في كتاب الجنائز عند قوله: «وإذَا اجتمعَتْ جنائزُ قُدِّمَ إلَى الإمامِ..» في (ص٤٣٤).

(١٨) أخرجه أحمد (٦/١٦٠)، وابن ماجه (٩٩٥) من حديث عروة بن الزبير عن عائشة ږ .

ورجَّح أبو حاتم إرساله (انظر: العلل لابنه س٤١٥)، وصححه ابن خزيمة (١٥٥٠)، وابن حبان (٢١٦٣)، والحاكم (١/٢١٤)، والبيهقي (٣/١٠٣).

(١٩) أي عند قوله: «(ولَا) تصحُّ صلَاةُ (الفذِّ)؛ أيْ: الفردِ..» في (ص٣٠٦).

(٢٠) أخرجه أحمد (٥/٣٩)، والبخاري (٧٨٣).
