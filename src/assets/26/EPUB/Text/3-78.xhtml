١٢٧١

(بابُ وليمَةِ العُرسِ)

أصلُ الوليمَةِ: تمامُ الشّيءِ واجتماعُهُ، ثمَّ نُقِلَتْ لطعامِ العُرْسِ خاصَّةً؛ لاجتماعِ الرّجلِ والمرأَةِ.

(تُسنُّ) الوليمَةُ بعَقْدٍ(١) (بشاةٍ فأقلَّ) منْ شاةٍ؛

  • لقولِهِ چ لعبدِ الرحمنِ بنِ عوفٍ حينَ قالَ لهُ تزوّجتُ: «أَوْلِمْ ولوْ بشاةٍ»(٢)،
  • وأولمَ النَّبِيُّ چ علَى صفيَّةَ بِحَيْسٍ وضعَهُ علَى نِطْعٍ صغيرٍ، كمَا فِي الصّحيحَيْنِ عنْ أنسٍ ڔ (٣)،
  • لكنْ قالَ جمعٌ: يُستحبُّ ألَّا تَنقُصَ عنْ شاةٍ.

(وتجبُ:

  • فِي أوّلِ مرَّةٍ)؛ أيْ: فِي اليومِ الأوّلِ،
  • (إجابةُ مسلمٍ يحرُمُ هجرُهُ)،
  • بخلافِ نحوِ رافِضِيٍّ،
١٢٧٢
  • ومُتجاهِرٍ بمعصيتِهِ(٤)،
  • إنْ دعاهُ (إلَيْهَا)؛ أيْ: إلَى الوليمَةِ (إنْ عيّنَهُ) الدّاعِي،
  • (ولمْ يكنْ ثَمَّ)؛ أيْ: فِي محلِّ الوليمَةِ (منكَرٌ)؛
  • لحديثِ أبِي هُريرةَ ڔ يرفعُهُ: «شرُّ الطّعامِ طعامُ الوليمَةِ، يُمْنَعُهَا مَنْ يأتيهَا ويُدْعَى إلَيْهَا مَنْ يأباهَا، ومَنْ لَا يُجيبُ(٥) فقدْ عصَى الله ورسولَهُ»، رواهُ مسلمٌ(٦).
  • (فإنْ دعاهُ الجَفَلَى) -بفتحِ الفاءِ- كقولِهِ: «أيُّهَا النّاسُ، هلمُّوا إلَى الطّعامِ»: لمْ تَجِبِ الإجابَةُ،
  • (أوْ) دعاهُ (فِي اليومِ الثّالثِ): كُرهَتْ إجابتُهُ؛ لقولِهِ چ : «الوليمَةُ أوّلُ يومٍ حقٌّ، والثّانِي معروفٌ، والثّالثُ رياءٌ وسمعةٌ»، رواهُ أبُو داودَ وغيرُهُ(٧).
١٢٧٣
  • وتُسنُّ فِي ثانِي يومٍ؛ لذلكَ الخبرِ،
  • (أوْ دعاهُ ذمّيٌّ)، أوْ مَنْ فِي مالِهِ حرامٌ: (كُرهَتِ الإجابَةُ)؛ لأنَّ المطلوبَ إذلالُ أهلِ الذِّمَّةِ، والتَّباعدُ عنِ الشُّبهَةِ وَمَا فيهِ الحرامُ؛ لئلَّا يواقعَهُ(٨).

وسائرُ الدَّعْوَاتِ: مباحةٌ،

  • غيرَ عقيقةٍ: فتسنُّ،
  • ومَأتَمٍ: فتُكرَهُ.

والإجابَةُ إلَى غيرِ الوليمَةِ: مستحبَّةٌ،

  • غيرَ مَأتَمٍ: فتُكرَهُ.

(ومَنْ صومُهُ واجبٌ)؛ كنذْرٍ، وقضاءِ رمضانَ، إذَا دُعِيَ للوليمَةِ:

  • حضرَ وجوبًا،
  • و(دعَا) استحبابًا،
  • (وانصرفَ)؛
  • لحديثِ أبِي هريرةَ ڔ يرفعُهُ: «إذْ دُعِيَ أحدُكُمْ فَلْيُجِبْ، فإنْ كانَ صائمًا فَلْيَدْعُ، وإنْ كانَ مُفْطِرًا فَلْيَطْعَمْ»، رواهُ أبُو دوادَ(٩).
١٢٧٤

(و) الصّائمُ (المتنفِّلُ) إذَا دُعِيَ:

  • أَجابَ،
  • و(يفطرُ إنْ جَبَرَ) قلبَ أخيهِ المسلمِ وأدخلَ عليهِ السُّرورَ؛
  • لقولِهِ چ لرجلٍ اعتزلَ مِنَ القومِ ناحيةً وقالَ: إنِّي صائمٌ: «دعاكُمْ أخوكُمْ وتكلّفَ لكُمْ، كُلْ يومًا ثمَّ صُمْ يومًا مكانَهُ إنْ شئْتَ»(١٠).

(ولَا يجبُ) علَى مَنْ حضرَ (الأكلُ) -ولوْ مفطرًا-؛ لقولِهِ چ : «إذَا دُعِيَ أحدُكُمْ فَلْيُجِبْ، فإنْ شاءَ أكلَ، وإنْ شاءَ تركَ» قالَ فِي شرحِ المقنعِ: «حديثٌ صحيحٌ»(١١)(١٢)،

  • ويُستحبُّ الأكلُ؛ لمَا تقدَّمَ.

(وإباحتُهُ)؛ أيْ: إباحةُ الأكلِ (مُتوقِّفَةٌ علَى:

  • صريحِ إذنٍ،
  • أوْ قرينةٍ)
١٢٧٥
  • ولـــوْ مـــنْ بيــتِ قــريــبٍ أوْ صـديـــقٍ لــمْ يُحْــرِزْهُ عـنــهُ؛ لحديثِ ابنِ عمرَ ژ : «مَنْ دخلَ علَى غيرِ دعوةٍ دخلَ سارقًا وخرجَ مُغيرًا»(١٣).
  • والدُّعاءُ إلَى الوليمَةِ، وتقديمُ الطَّعامِ: إذنٌ فيهِ.
  • ولَا يَمْلِكُهُ منْ قُدِّمَ إليهِ، بلْ يهلِكُ علَى مِلكِ صاحبِهِ.

ڭ ڭ ڭ

(وإنْ عَلِمَ) المدعوُّ (أنَّ ثمَّ)؛ أيْ: فِي الوليمةِ (مُنكَرًا)؛

  • كزَمْرٍ، وخمرٍ، وآلاتِ لَهْوٍ، وفَرْشِ حَرِيرٍ، ونحوِهَا،
  • فإنْ كانَ (يقدرُ علَى تغييرِهِ: حضرَ وغيَّرَهُ)؛ لأنَّهُ يؤدِّي بذلكَ فرضَيْنِ: إجابةَ الدّعوَةِ، وإزالةَ المنكَرِ،
  • (وإلَّا) يقدرْ علَى تغييرِهِ: (أبَى) الحضورَ؛ لحديثِ عمرَ ڔ مرفوعًا: «مَنْ كانَ يؤمنُ باللهِ واليومِ الآخِرِ فلَا يقعدْ علَى مائدةٍ يُدارُ عَلَيْهَا الخمرُ»، رواهُ التّرمذِيُّ(١٤).
١٢٧٦

(وإنْ حضرَ) منْ غيرِ علمٍ بالمنكَرِ (ثمَّ علِمَ بهِ

  • أزالَهُ)؛ لوجوبِهِ عليهِ، ويجلسُ بعدَ ذلكَ،
  • (فإنْ دامَ) المنكَرُ (لعجزِهِ)؛ أيْ: المدعوِّ (عنهُ انصرفَ)؛ لئلَّا يكونَ قاصدًا لرؤيتِهِ، أوْ سَمَاعِهِ.

(وإنْ علِمَ) المدعوُّ (بهِ)؛ أيْ: بالمنكَرِ (ولمْ يرَهُ ولمْ يسمعْهُ: خُيِّرَ) بينَ الجلوسِ والأكلِ، أوِ الانصرافِ؛ لعدمِ وجوبِ الإنكارِ حينئذٍ.

(وكُرهَ النِّثارُ والتقاطُهُ)؛

  • لمَا يحصلُ فيهِ مِنَ النُّهْبَةِ والتّزاحمِ،
  • وأخذُهُ علَى هذَا الوجهِ فيهِ دناءةٌ وسُخْفٌ.
  • (ومَنْ أَخَذَهُ)؛ أيْ: أخذَ شيئًا مِنَ النِّثارِ،
  • (أوْ وقعَ فِي حِجْرِهِ) مِنهُ شيءٌ؛
  • (فـ)ـهوَ (لهُ)، قصدَ تملُّكَهُ أوْ لَا؛ لأنَّهُ قدْ حازَهُ، ومالِكُهُ قصدَ تمليكَهُ لمَنْ حازَهُ.

ڭ ڭ ڭ

(ويُسنُّ إعلانُ النِّكاحِ)؛ لقولِه چ : «أعلنُوا النِّكاح»، وفِي لفظٍ: «أظهِرُوا النِّكاح» رواهُ ابنُ ماجهْ(١٥).

١٢٧٧

(و) يُسنُّ (الدُّفُّ) أيْ: الضّربُ بهِ،

  • إذَا كانَ لَا حِلَقَ بهِ(١٦) ولَا صُنُوجَ،
  • (فيهِ) أيْ: فِي النِّكاحِ،
  • (للنِّساءِ)،
  • وكذَا خِتانٌ،
  • وقدومُ غائبٍ،
  • ووِلادةٌ،
  • وإملاكٌ؛
  • لقولِهِ چ : «فَصْلُ مَا بينَ الحلالِ والحرامِ الصّوْتُ والدُّفُّ فِي النِّكاحِ»، رواهُ النَّسائِيُّ(١٧).
١٢٧٨
  • وتَحْرُمُ كلُّ مَلْهَاةٍ سِوَى الدُّفِّ؛
  • كمِزْمَارٍ، وطُنْبُورٍ، وجَنْكٍ، وعُودٍ.
  • قالَ فِي المستوعِبِ والتّرغيبِ: سواءٌ استُعمِلَ لحُزْنٍ أوْ سرورٍ(١٨).

ڭ ڭ ڭ

(١) في (ز، ل، م): «بعقدٍ ولو».

(٢) أخرجه أحمد (٣/١٦٥)، والبخاري (٢٠٤٩)، ومسلم (١٤٢٧) من حديث أنس بن مالك ڔ .

(٣) أخرجه أحمد (٣/١٠١ - ١٠٢)، والبخاري (٥٣٨٧)، ومسلم (١٣٦٥).

(٤) في (ب، ز، ض، م): «بمعصية».

(٥) في (ب، د، ز، ع، ل، م): «لا يُجِبْ».

(٦) أخرجه مسلم (١٤٣٢ / ١١٠) عن أبي هريرة ڔ مرفوعًا، وأخرجه أحمد (٢/٢٤١)، والبخاري (٥١٧٧)، ومسلم (١٤٣٢) من حديث أبي هريرة ڔ موقوفًا.

(٧) أخرجه أحمد (٥/٢٨)، وأبو داود (٣٧٤٥) من حديث زهير بن عثمان -مختلف في صحبته- h.

قال البخاري في التاريخ الكبير (٣/٤٢٥): (لا يصح إسناده).

ورُوي عن الحسن البصري عن النبي چ مرسلًا، أخرجه عبدالرزاق (١٠/٤٤٧)، وابن أبي شيبة (١٤/١٣٠)، ورجَّحه الدارقطني في العلل (س٢٤٣١).

وأخرجه الترمذي (١٠٩٧) من حديث ابن مسعود ڔ به مرفوعًا، وضعَّفه جدًا.

وأخرجه ابن ماجه (١٩١٥) من حديث أبي هريرة ڔ به مرفوعًا، قال البيهقي (٧/٢٦٠): (ليس بشيء).

(٨) في (ب، ز، ل، م): «يوافقه».

(٩) أخرجه أبو داود (٣٧٣٧) من حديث ابن عمر ژ ، وأخرجه أحمد (٢/٢٧٩) ومسلم (١٤٣١) من حديث أبي هريرة ڔ به مرفوعًا، وفيه: «فليصلِّ» بدل «فليدع».

(١٠) أخرجه الطيالسي (٢٢٠٣)، والطبراني في الأوسط (٣٢٤٠) من حديث أبي سعيد الخدري ڔ .

قال العراقي في المغني عن حمل الأسفار (٢/٩٢٣): (لا يصح)، وحسَّن إسناده ابن حجر في الفتح (٢/٢١٠).

(١١) أخرجه أحمد (٣/٣٩٢)، ومسلم (١٤٣٠) من حديث جابر بن عبدالله ژ .

(١٢) الشرح الكبير (٢١/٣٢٨).

(١٣) أخرجه أبو داود (٣٧٤١) وفي إسناده أبان بن طارق.

قال أبو داود: (أبان بن طارق مجهول)، وقال العراقي في المغني عن حمل الأسفار (٢/٩١٥): (إسناده ضعيف).

(١٤) أخرجه أحمد (١/٢٠) من حديث عمر بن الخطاب ڔ .

قال ابن الملقن في البدر المنير (٨/١٩): (في إسناده مجهول).

وأخرجه أحمد (٣/٣٣٩)، والترمذي (٢٨٠١) من حديث جابر بن عبد الله ڔ ، قال الترمذي: (حسن غريب).

ورُوي من حديث ابن عمر وأبي سعيد وابن عباس وعمران بن حصين ښ ، قال ابن حجر في التلخيص الحبير (٥/٢٣٩٨): (وأسانيدها ضعاف).

(١٥) أخرجه أحمد (٤/٥) من حديث عبدالله بن الزبير ژ بلفظ: «أعلنوا».

وصححه ابن حبان (٤٠٦٦)، والحاكم (٢/٢٠٠)، وأشار ابن عبدالهادي إلى إعلاله في حاشية الإلمام (٩٧٦).

وأخرجه الترمذي (١٠٨٩)، وابن ماجه (١٨٩٥) من حديث عائشة ږ به مرفوعًا بلفظ: «أعلنوا»، قال الترمذي: (حديث غريب حسنٌ في هذا الباب).

وأخرجه سعيد بن منصور (٦٣٥)، وإسحاق بن راهويه (٩٤٥)، والبيهقي (٧/٢٩٠) من حديث عائشة ږ بلفظ: «أظهروا»، وضعَّفه ابن الجوزي في العلل المتناهية (٢/١٣٨).

(١٦) في (ز، م): «فيه».

(١٧) أخرجه أحمد (٣/٤١٨)، والترمذي (١٠٨٨)، وابن ماجه (١٨٩٦)، والنسائي (٦/١٢٧) من حديث أبي بلج عن محمد بن حاطب ڔ .

قال الترمذي: (حديث حسن)، وأعلَّه ابن حبان في المجروحين (٣/١١٣)، وقال ابن الملقن (٩/٦٤٤): (قال ابن طاهر: ألزم الدارقطني مسلمًا إخراجه، قال: وهو صحيح).

(١٨) نقله عنهما في: الإنصاف، للمرداوي (٢١/٣٥٥)، وقارن بما في: المستوعب (٢/٨٠٨).
