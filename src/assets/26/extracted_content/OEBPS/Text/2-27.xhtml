(بابُ) ذكرِ (دخولِ مكَّةَ) ومَا يتعلّقُ بهِ مِنَ الطّوافِ والسّعْيِ

(يُسنُّ) دخولُ مكَّةَ (مِنْ أعلاهَا)،

(و) يُسنُّ دخولُ (المسجدِ) الحرامِ (مِنْ بابِ بنِي شيبَةَ)؛ لمَا روَى مسلمٌ وغيرُهُ عنْ جابرٍ ڔ : «أنَّ النَّبِيَّ چ دخلَ مكَّةَ ارتفاعَ الضُّحَى وأناخَ راحلتَهُ عندَ بابِ بنِي شيبَةَ ثمَّ دخلَ»(١).

ويُسنُّ أنْ يقولَ عندَ دخولِهِ: بسمِ اللهِ، وباللهِ، ومنَ اللهِ، وإلَى اللهِ، اللّهمَّ افتحْ لِي أبوابَ فضلِكَ. ذكرَهُ فِي أسبابِ الهدايَةِ(٢).

(فإذَا رأَى البيْتَ:

٦٤٠

(فيحاذِي الحجرَ الأسودَ بكلِّهِ)؛ أيْ: بكلِّ بدنِهِ فيكونُ مبدأَ طوافِهِ؛ لأنَّهُ چ كانَ يبتدئُ بهِ(٨)،

(ويستلمُهُ)؛ أيْ: يمسحُ الحجرَ بيدِهِ اليمنَى، وفِي الحديثِ: «إنَّهُ نزلَ مِنَ الجنَّةِ أشدَّ بياضًا مِنَ اللّبنِ فسوّدتْهُ خطايَا بنِي آدمَ»، رواهُ التّرمذِيُّ وصحَّحَهُ(٩).

٦٤٢

(ويقبِّلُهُ)؛ لمَا روَى عمرُ ڔ : أنَّ النَّبِيَّ چ استقبلَ الحجرَ ووضعَ شَفَتَيْهِ عليهِ يبكِي طويلًا، ثمَّ التفتَ فإذَا بعمرَ بنِ الخطّابِ يبكِي، فقالَ: «يَا عمرُ، هَا هنَا تُسكبُ العَبَراتُ»، رواهُ ابنُ ماجهْ(١٠)،

نقَلَ الأثرمُ: ويسجدُ عليهِ(١١)، وفعَلَهُ ابنُ عمرَ وابنُ عبّاسٍ ښ (١٢)،

(فإنْ شقَّ) استلامُهُ وتقبيلُهُ لمْ يزاحمْ، واستلمَهُ بيدِهِ، و(قبّلَ يدَهُ)؛ لمَا روَى مسلمٌ عنِ ابنِ عبّاسٍ ژ : «أنَّ النَّبِيَّ چ استلمَهُ وقبّلَ يدَهُ»(١٣)،

٦٤٣

(فإنْ شقَّ) استلمَهُ بشيءٍ وقبّلَهُ؛ رُويَ عنِ ابنِ عبّاسٍ ژ (١٤)،

فإنْ شقَّ (اللّمسُ أشارَ إليهِ)؛ أيْ: إلَى الحجرِ بيدِهِ أوْ بشيءٍ ولَا يقبِّلُهُ؛ لمَا روَى البخارِيُّ عنِ ابنِ عبّاسٍ ژ قالَ: «طافَ النَّبِيُّ چ علَى بعيرٍ فلَمَّا أتَى الحجرَ أشارَ إليهِ بشيءٍ فِي يدِهِ وكبّرَ»(١٥)،

(ويقولُ) مستقبِلَ الحجرِ بوجهِهِ كلَّمَا استلمَهُ (مَا وردَ) ومنْهُ: «بسمِ اللهِ واللهُ أكبرُ، اللّهمَّ إيمانًا بكَ، وتصديقًا بكتابِكَ، ووفاءً بعهدِكَ، واتِّباعًا لسُنَّةِ نبيِّكَ مُحمَّدٍ» چ ؛ لحديثِ عبدِ اللهِ بنِ السّائبِ ڔ أنَّ النَّبِيَّ چ كانَ يقولُ ذلكَ عندَ استلامِهِ(١٦).

  • (ثلاثًا)؛ أيْ: فِي ثلاثَةِ أشواطٍ،

ويُسنُّ أنْ (يستلمَ الحجرَ والرُّكْنَ اليمانيَّ كلَّ مرَّةٍ) عندَ محاذاتِهِمَا؛ لقولِ ابنِ عمرَ ژ : «كانَ رسولُ اللهِ چ لَا يدَعُ أنْ يستلمَ الرّكنَ اليمانيَّ والحجرَ فِي طوافِهِ»، قالَ نافعٌ: وكانَ ابنُ عمرَ يفعلُهُ، رواهُ أبُو داودَ(١٩).

  • فإنْ شقَّ استلامُهُمَا أشارَ إليْهِمَا.
  • لَا الشّاميَّ وهوَ: أوّلُ ركنٍ يمرُّ بهِ،
  • ولَا الغربيَّ وهوَ: مَا يليهِ،

ويقولُ بينَ الرُّكنِ اليمانيِّ والحجرِ الأسودِ: {رَبَّنَآ ءَاتِنَا فِي ٱلدُّنۡيَا حَسَنَةٗ وَفِي ٱلۡأٓخِرَةِ حَسَنَةٗ وَقِنَا عَذَابَ ٱلنَّارِ} [البقرة:٢٠١]،

وفِي بقيَّةِ طوافِهِ: «اللّهمَّ اجعلْهُ حجًّا مبرورًا، وسعيًا مشكورًا، وذنبًا مغفورًا، ربِّ اغفرْ وارحمْ واهدنِي السّبيلَ الأقومَ، وتجاوزْ عمّا تعلمُ، وأنْتَ الأعزُّ الأكرمُ».

وتُسنُّ القراءَةُ فيهِ.

(١) لم نقف عليه عند مسلم ولا غيره بهذا اللفظ.

وأخرج الطبراني في الأوسط (٤٩١) من حديث ابن عمر ژ قال: (دخل رسول الله چ ودخلنا معه من باب بني عبد مناف وهو الذي يسميه الناس باب بني شيبة وخرجنا معه إلى المدينة من باب الحزورة وهو باب الخياطين)، قال البيهقي (٥/٧٢): (إسناده غير محفوظ)، وقال ابن حجر (التلخيص ٤/١٥٧١): (وفي إسناده عبد الله بن نافع وفيه ضعف).

(٢) نقله في: المبدع للبرهان ابن مفلح (٣/١٩٢).

(٣) أخرجه الشافعي في الأم (٣/٤٢٢) عن ابن جريج مرسلًا.

قال البيهقي (٥/٧٣): (هذا منقطع، وله شاهد مرسل)، وهو ما أخرجه ابن أبي شيبة (٤/٩٧) عن مكحول مرسلًا، قال الذهبي في المهذب (٤/١٨١٨): (والآخر منقطع).

(٤) أخرجه ابن ابي شيبة (٤/٩٧)، وابن معين (انظر: تاريخ الدوري ٣/٢١١ برقم ٩٧٨)، وأحمد في مسائل عبد الله (٧٩٤)، والبخاري في التاريخ الكبير (١/٢٩٤) عن سعيد بن المسيب عن عمر بن الخطاب ڔ أنه كان يقوله.

وأخرجه الشافعي في الأم (٣/٤٢٣)، وابن أبي شيبة (٤/٩٧) عن سعيد قوله.

(٥) كما ورد في أثر ابن جريج المتقدم تخريجه قريبًا.

(٦) ذكره ابن قدامة في الكافي (٢/٤٠٤) وعزاه للأثرم، ولم نقف عليه مسندًا.

(٧) أخرجه البخاري (١٦١٤)، ومسلم (١٢٣٥) من حديث عائشة ږ قالت: (أول شيء بدأ به حين قدم النبي چ أنه توضأ، ثم طاف بالبيت).

(٨) أخرجه أحمد (٣/٣٢٠)، ومسلم (١٢١٨) من حديث جابر بن عبد الله ژ ، في سياق حجَّة النبي چ ، وفيه: (حتى إذا أتينا البيت معه چ استلم الرُّكنَ)، أي الحَجَر الأسود.

(٩) أخرجه أحمد (١/٣٠٧)، والترمذي (٨٧٧)، والنسائي مختصرًا (٥/٢٢٦) من حديث ابن عباس ژ .

وصححه ابن خزيمة (٢٧٣٣).

(١٠) أخرجه ابن ماجه (٢٩٤٥) من حديث محمد بن عون عن نافع عن ابن عمر ژ به.

قال أبو حاتم الرازي (انظر: الجرح والتعديل لابنه ٨/٤٧) عن محمد هذا: (منكر الحديث، روى عن نافع حديثًا ليس له أصل)، قال المزي في تهذيب الكمال في ترجمته عن هذا الحديث: (وكأنه الذي أشار إليه أبو حاتم)، وضعَّف الحديث البوصيري في مصباح الزجاجة (١٠٤١).

(١١) انظر: زاد المسافر (٢/٥٤٩).

(١٢) لم نجده عن ابن عمر ژ ، ولعله عن والده عمر ڔ والله أعلم، فقد أخرج الطيالسي (٢٨)، والحاكم (١/٤٥٥)، والبيهقي (٥/٧٤) من طريقهما واللفظ له، عن جعفر بن عبد الله قال: رأيت محمد بن عبَّاد بن جعفر قبَّل الحجر وسجد عليه، ثم قال: (رأيت خالَكَ ابنَ عباس يُقبِّله ويسجد عليه، وقال ابن عباس: رأيت عمر بن الخطاب ڔ قبَّله وسجد عليه ثم قال: رأيت رسول الله چ فعل هكذا، ففعلت).

صححه ابن خزيمة (٢٧١٤)، ونقل شيخ الإسلام ابن تيمية في شرح العمدة (الحج ٢/٤٣٠) عن الإمام أحمد أنَّه حسَّنه.

(١٣) أخرجه مسلم (١٢٦٨)، وعبد الله بن أحمد في زوائد المسند (٢/١٠٨) عن نافع قال: (رأيت ابن عمر ژ استلم الحجر ثم قبل يده وقال: ما تركته منذ رأيت رسول الله چ يفعله).

ولم نقف عليه من حديث ابن عباس ژ .

(١٤) أخرجه البخاري (١٦٠٧)، ومسلم (١٢٧٢) عن ابن عباس ژ قال: (طاف النبي چ في حجَّة الوداع على بعير يستلم الرُّكن بمحجن)، زاد أحمد (١/٣٣٨) والنسائي في الكبرى (٤١١٥): (ويقبل المحجن)، وأخرجه مسلم (١٢٧٥) من حديث أبي الطفيل ڔ ، بتمامه.

(١٥) أخرجه أحمد (١/٢٦٤)، والبخاري (١٦٣٢).

(١٦) لم نقف عليه من حديث ابن السائب ڔ ولا غيره مرفوعًا، وأنكرَ ابنُ الملقن رفعه إلى النبي چ في البدر المنير (٦/١٩٥).

ورُوي نحوه من حديث علي بن أبي طالب ڔ موقوفًا، أخرجه الطبراني في الدعاء (٨٦٠)، والطيالسي (١٧٤)، وضعَّفه البوصيري في إتحاف الخيرة المهرة (٣/١٨٩).

ومن حديث ابن عباس ژ موقوفًا، أخرجه عبد الرزاق (٥/٣٣)، والطبراني في الدعاء (٨٦١).

ومن حديث ابن عمر ژ موقوفًا، أخرجه الطبراني في الأوسط (٥٤٨٢)، وقال البخاري في التاريخ الكبير (١/٢٣٠): (لا يتابع عليه)، ووافقه العقيلي في الضعفاء (٥/٣٨٩).

(١٧) أخرجه مسلم (١٢٩٧) من حديث جابر بن عبد الله ژ .

(١٨) أخرجه أحمد (٣/٣٢٠)، ومسلم (١٢١٨) من حديث جابر بن عبد الله ژ ، في سياق حجَّة النبي چ وفيه: (فرَمَلَ ثلاثًا ومشى أربعًا).

(١٩) أخرجه أحمد (٢/١٨)، وأبو داود (١٨٧٦)، والنسائي (٥/٢٣١) من دون ذكر قول نافع من حديث عبد العزيز بن أبي رواد عن نافع به.

قال ابن حبان في المجروحين (٢/١٣٦): (روى عبد العزيز عن نافع عن ابن عمر ژ نسخةً موضوعةً لا يحل ذكرها إلا على سبيل الاعتبار)، وقال المنذري في مختصر سنن أبي داود (٢/٣٧٥): (في إسناده عبد العزيز بن أبي رواد وفيه مقال).

(٢٠) سبق تخريجه في (ص٦٤٤).

(٢١) سبق تخريجه في (ص٥٤٣).

(٢٢) سبق تخريجه في (ص٦٤٤).

(٢٣) أخرجه الترمذي (٩٦٠) من حديث ابن عباس ژ ، وأخرجه أحمد (٣/٤١٤) والنسائي (٥/٢٢٢) عن طاوس عن رجل أدرك النبي چ .

والحديث اختلف في رفعه ووقفه، ورجح جماعة من الأئمة وقفه.

قال شيخ الإسلام ابن تيمية في مجموع الفتاوى (٢١/٢٧٤): (وأهل المعرفة بالحديث لا يصححونه إلا موقوفًا ويجعلونه من كلام ابن عباس، ولا يثبتون رفعه)، وللاستزادة: انظر ما سبق (ص٨٦).

(٢٤) في (ب، ز، ع، م) «بالكافرون».
