٤٨٩

(بابُ زكَاةِ النّقدَيْنِ)

أيِ: الذّهبِ والفضَّةِ.

(يجبُ:

  • فِي الذّهبِ إذَا بلغَ عشرِينَ مثْقالًا،
  • وفِي الفضَّةِ إذَا بلغَتْ مِائتَيْ درهمٍ) إسلاميٍّ:
  • (رُبعُ العُشْرِ منهُمَا)؛
  • لحديثِ ابنِ عمرَ وعائشَةَ ښ مرفوعًا: «أنَّهُ كانَ يأخذُ منْ كلِّ عشرِينَ مِثْقَالًا نصفَ مِثْقالٍ» رواهُ ابنُ ماجهْ(١).
  • وعنْ عليٍّ ڔ نحوُهُ(٢).
٤٩٠
  • وحديثِ أنسٍ ڔ مرفُوعًا: «فِي الرِّقَةِ رُبعُ العُشرِ» متّفقٌ عليهِ(٣).

والاعتبارُ: بالدِّرهمِ الإسلاميِّ الَّذِي وزنُهُ ستَّةُ دَوَانِقَ، والعَشرةُ مِنَ الدّراهمِ: سبعَةُ مَثَاقِيلَ، فالدِّرهمُ نصفُ مِثْقالٍ وخُمُسُهُ، وهوَ: خمسُونَ حبَّةً وخُمُسَا حبَّةِ شعيرٍ.

والعِشرُونَ مثقالًا: خمسةٌ وعشرونَ دينارًا وسُبُعَا دينارٍ وتُسُعُهُ، علَى التّحديدِ بالّذِي زِنَتُهُ دِرهمٌ وثُمنُ درهمٍ.

ويزُكَّى مغشوشٌ إذَا بلغَ خالصُهُ نصابًا وزنًا.

(ويُضمُّ الذّهبُ إلَى الفضَّةِ فِي تكميلِ النِّصابِ) بالأجزاءِ، فلوْ مَلكَ عَشرةَ مَثَاقِيلَ ومِائَةَ درهمٍ، فكلٌّ منهُمَا نصفُ نصابٍ، ومجموعُهمَا نصابٌ، ويجزئُ إخراجُ زكَاةِ أحدِهمَا مِنَ الآخَرِ؛ لأنَّ مقاصدَهمَا وزكاتَهمَا متّفقةٌ، فهمَا كنوعَيْ جنسٍ.

  • ولَا فرقَ بينَ الحاضرِ والدَّينِ.

(وتُضمُّ قيمَةُ العُروضِ)؛ أيْ: عُروضِ التِّجارَةِ (إلَى كلٍّ منهُمَا)؛ كمَنْ لهُ عشرةُ مَثَاقِيلَ ومتاعٌ قيمتُهُ عشرةٌ أخرَى، أوْ لهُ مِائةُ درهمٍ ومتاعٌ قيمتُهُ مثلُهَا.

٤٩١

ولوْ كانَ ذهبٌ وفضَّةٌ وعُروضٌ: ضُمَّ الجميعُ فِي تكميلِ النِّصابِ.

ويُضمُّ جيِّدُ كلِّ جنسٍ ومَضْروبُهُ إلَى رَدِيئِهِ وتِبْرِهِ، ويُخرجُ منْ كلِّ نوعٍ بحصّتِهِ، والأفضلُ مِنَ الأعلَى، ويجزئُ إخراجُ رَدِيءٍ عنْ أعلَى معَ الفضلِ.

ڭ ڭ ڭ

(ويُباحُ للذّكرِ مِنَ الفضَّةِ: الخاتمُ)؛ لأنَّهُ چ «اتّخذَ خاتَمًا منْ وَرِقٍ» متّفقٌ عليهِ(٤).

  • والأفضلُ جعلُ فصِّهِ ممّا يلِي كفَّهُ، ولهُ جعلُ فصِّهِ مِنهُ ومنْ غيرِهِ،
  • والأولَى جعلُهُ فِي يسارِهِ.
  • ويُكرهُ بسبّابةٍ ووسْطَى،
  • ويُكرَهُ أنْ يُكتبَ عليهِ ذكرُ اللهِ قرآنٌ أوْ غيرُهُ.

ولوِ اتّخذَ لنفسِهِ عدَّةَ خواتيمَ: لمْ تسقطِ الزّكَاةُ فيمَا خرجَ عنِ العادَةِ، إلَّا أنْ يتّخِذَ ذلكَ لولدِهِ أوْ عبدِهِ.

(و) يُباحُ لهُ: (قَبِيعَةُ السّيفِ)، وهيَ: مَا يُجعلُ علَى طرفِ القبضَةِ. قالَ أنسٌ ڔ : «كانتْ قَبِيعَةُ سيفِ رسولِ اللهِ چ فضَّةً»، رواهُ الأثرمُ(٥).

٤٩٢

(و) يُباحُ لهُ: (حِلْيَةُ المِنْطَقَةِ)، وهيَ: مَا يُشدُّ بهِ الوسَطُ، وتسمِّيهَا العامَّةُ: الحِيَاصَةَ، واتّخذَ الصَّحابَةُ المناطقَ مُحلَّاةً بالفضَّةِ.

(ونحوُهُ)؛ أيْ: نحوُ مَا ذُكرَ؛ كحِلْيَةِ الجَوْشَنِ، والخُوذَةِ، والخُفِّ، والرَّانِ، وحَمَائلِ سيفٍ؛ لأنَّ ذلكَ يساوِي المِنْطَقَةَ معنًى، فوجبَ أنْ يساويَهَا حُكْمًا.

  • قالَ الشّيخُ تقيُّ الدِّينِ: «وتَرْكاشُ النُّشّابِ، والكَلَاليبُ؛ لأنَّهُ يسيرٌ تابعٌ»(٦).

ولَا يُباحُ غيرُ ذلكَ: كتَحْلِيَةِ المراكبِ، ولِباسِ الخيلِ كاللُّجُمِ، وتَحْلِيَةِ الدَّوَاةِ، والمِقْلَمَةِ، والكِمْرانِ، والـمُشطِ، والمُكحُلَةِ، والمِـيلِ، والمِرآةِ، والقِنْدِيلِ.

(و) يُباحُ للذَّكرِ (مِنَ الذّهبِ:

  • قَبِيعَةُ السّيفِ)؛ لأنَّ عمرَ ڔ كانَ لهُ سيفٌ فيهِ سَبائِكُ منْ ذهبٍ، وعثمانُ بنُ حُنَيْفٍ ڔ كانَ فِي سيفِهِ مسمارٌ منْ ذهبٍ، ذكرَهمَا أحمدُ(٧) وقيّدَهَا باليسيرِ، معَ أنَّهُ ذكرَ أنَّ قَبِيعَةَ سيفِ النَّبِيِّ چ كانَ
٤٩٣
  • وزنُهَا ثمانيةَ مَثَاقِيلَ(٨)، فيُحتملُ أنَّهَا كانتْ ذهبًا وفضَّةً، وقدْ رواهُ التِّرمذِيُّ كذلكَ(٩).
  • (ومَا دعَتْ إليهِ ضرورةٌ؛ كأنفٍ ونحوِهِ)؛ كرِباطِ(١٠) أسْنَانٍ؛ «لأنَّ عرفجَةَ بْنَ أسعدَ ڔ قُطعَ أنفُهُ يومَ الكُلابِ فاتّخذَ أنفًا منْ فضَّةٍ، فأنتَنَ عليهِ، فأمرَهُ النَّبِيُّ چ فاتّخذَ أنفًا منْ ذهبٍ»، رواهُ أبُو داودَ وغيرُهُ وصحّحَهُ الحاكمُ(١١)، وروَى الأثرمُ عنْ: موسَى بنِ طلحَةَ،
٤٩٤
  • وأبِي حمْزَةَ(١٢) الضُّبَعيِّ، وأبِي رافعٍ وثابتٍ(١٣) البُنانيِّ، وإسماعيلَ بنِ زيدِ بنِ ثابتٍ، والمغيرَةِ بنِ عبدِ اللهِ، أنَّهُمْ شدُّوا أسنانَهُمْ بالذّهبِ(١٤).

(ويُباحُ للنِّساءِ مِنَ الذّهبِ والفضَّةِ مَا جرَتْ عادتُهنَّ بِلُبْسِهِ ولوْ كثُرَ)؛

  • كالطَّوْقِ، والخَلْخَالِ، والسِّوارِ، والقُرْطِ، ومَا فِي المخَانِقِ والمقَالِدِ والتّاجِ، ومَا أشبهَ ذلكَ؛
  • لقولِهِ چ : «أُحلَّ الذّهبُ والحريرُ للإناثِ منْ أُمّتِي، وحُرِّمَ علَى ذكورِهَا»(١٥).
٤٩٥

ويُباحُ لهمَا: تَحلٍّ بجوهرٍ ونحوِهِ.

وكُرهَ تختُّمُهمَا:

  • بحديدٍ،
  • وصُفْرٍ،
  • ونُحَاسٍ،
  • ورَصَاصٍ.

(ولَا زكَاةَ فِي حُليِّهِمَا)؛ أيْ: حُليِّ الذّكرِ والأنثَى المباحِ (المعَدِّ للاستعمالِ أوِ العاريَّةِ)؛

  • لقولِهِ چ : «ليسَ فِي الحُليِّ زكاةٌ» رواهُ الطّبريُّ(١٦) عنْ جابرٍ ڔ (١٧)،
  • وهوَ قولُ أنسٍ(١٨)، وجابرٍ(١٩)، وابنِ عمرَ(٢٠)، وعائشَةَ(٢١)، وأسماءَ
٤٩٦
  • أختِهَا ښ (٢٢).
  • حتَّى ولوِ اتَّخذَ الرَّجلُ حُليَّ النِّساءِ؛ لإعارتِهنَّ أوْ بالعكسِ، إنْ لمْ يكنْ فرارًا.

(وإنْ أُعدَّ) الحُليُّ:

  • (للكِراءِ،
  • أوِ النّفقَةِ،
  • أوْ كانَ مُحرَّمًا)؛ كسَرْجٍ، ولِجامٍ، وآنيةٍ:
  • (فَفِيهِ الزّكَاةُ) إنْ بلغَ نصابًا وزنًا؛ لأنَّهَا إنَّمَا سقطَتْ ممّا أُعدَّ للاستعمالِ بصرفِهِ عنْ جهَةِ النّماءِ، فيبقَى مَا عداهُ علَى مقتضَى الأصلِ.

فإنْ كانَ مُعَدًّا للتِّجارَةِ وجبتِ الزّكَاةُ فِي قيمتِهِ كالعُروضِ.

ومباحُ الصِّناعَةِ إذَا لمْ يكنْ للتِّجارَةِ: يُعتبرُ فِي النِّصاب بوزنِهِ، وفِي الإخراجِ بقيمتِهِ.

ويَحرُمُ أنْ يُحلَّى مسجدٌ، أوْ يُموَّهَ سقفٌ أوْ حائطٌ بنقدٍ، وتجبُ إزالتُهُ، وزكاتُهُ بشرطِهِ، إلَّا إذَا استُهلكَ فلمْ يجتمعْ مِنهُ شيءٌ.

ڭ ڭ ڭ

(١) أخرجه ابن ماجه (١٧٩١) من حديث إبراهيم بن إسماعيل عن عبدالله بن واقد عن ابن عمر وعائشة ښ أن النبي چ كان يأخذ من كل عشرين دينارًا، نصف دينار.

قال البوصيري في مصباح الزجاجة (٦٤٦): (هذا إسناد فيه إبراهيم بن إسماعيل وهو ضعيف)، وضعَّفه ابن حجر في الدراية (١/٢٥٨)، وللحديث شواهد انظرها في الإرواء (٣/٢٨٩-٢٩١).

(٢) أخرجه عبدالرزاق (٤/٨٩)، وأبو داود (١٥٧٣)، ومن طريقه البيهقي (٤/١٣٧) من حديث أبي إسحاق السبيعي عن عاصم بن ضمرة والحارث الأعور عن علي مرفوعًا، في ذكر الصدقات، وفيه: «فإذا كان لك عشرون دينارًا وحال عليها الحول ففيها نصف دينار».

ورُويَ عن علي موقوفًا، رواه أبو عبيد في الأموال (١٠٦١)، وابن أبي شيبة (٣/١١٩)، وابن خزيمة (٢٢٨٤).

ورجَّح البخاري فيما نقله عنه الترمذي (٦٢٠)، وكذا الدارقطني في العلل (س٣٢٦) صحةَ الوجهين، وصحح الحديث ابن حزم في المحلى (٦/٧٤)، وضعَّفه أبو عبيد في الأموال (عقب الحديث ٩٢١) للفظة منكرة لم تروَ إلا فيه.

(٣) أخرجه أحمد (١/١١)، والبخاري (١٤٥٤)، ولم يروه مسلم.

(٤) أخرجه أحمد (٢/١٨)، والبخاري (٥٨٦٥)، ومسلم (٢٠٩١) من حديث عبدالله بن عمر ژ .

(٥) أخرجه أبو داود (٢٥٨٣)، والترمذي (١٦٩١)، والنسائي (٨/٢١٩) من حديث جرير بن حازم عن قتادة عن أنس به، ورُويَ عن قتادة عن سعيد بن أبي الحسن مرسلًا، أخرجه أبو داود (٢٥٨٤)، والنسائي (٨/٢١٩)، والترمذي في الشمائل (١٠٦) من طرق عن قتادة به.

فاختلف في وصله وإرساله؛ ورجَّح إرساله أحمد في العلل برواية عبدالله (٣١٢)، والدارمي في سننه (٢٦١٥)، وأبو حاتم في العلل لابنه (٩٣٨) وأبو داود، وغيرهم.

(٦) انظر: الأخبار العلمية (اختيارات شيخ الإسلام)، لابن اللحام (ص١١٦).

(٧) انظر: المغني (٤/٢٢٧)، وأثر عمر أخرجه عبدالله بن الإمام أحمد في زوائده على فضائل الصحابة (٣٢٥)، عن ابن عمر قال: (كان سيف عمر بن الخطاب الذي شهد بدرًا فيه سبائك من ذهب).

ولم نقف على من أخرج أثر عثمان بن حنيف، وأخرج ابن أبي شيبة (٨/٢٨٧) قال: حدثنا ابن نمير، قال: حدثنا عثمان بن حكيم، قال: (رأيت في قائم سيف سهل بن حنيف مسمارًا من ذهب).

(٨) لم نقف عليه مسندًا.

(٩) أخرجه الترمذي (١٦٩٠) من حديث طالب بن حُجير عن هود –ويقال هوذة- بن عبدالله بن سعد عن جدِّه مَزِيدَة العصري وله صحبة قال: دخل رسول الله چ مكة يوم الفتح وعلى سيفه ذهبٌ وفضة.

قال الترمذي: (حديث غريب)، وضعَّفه ابن عبدالبر في الاستيعاب (٣/٥٢٦-٥٢٧ بهامش الإصابة)، وابن القطان في بيان الوهم والإيهام (٣/٤٨١)، والذهبي في ميزان الاعتدال في ترجمة طالب (٢/٣٠٥)، وقال الدارقطني في أطراف الغرائب والأفراد (٤/٣٦٢): (تفرَّد به طالب بن حجير عن هوذة عن جده).

(١٠) في (س): «كرياط».

(١١) أخرجه أحمد (٥/٢٣)، أبو داود (٤٢٣٢)، والترمذي (١٧٧٠)، والنسائي (٨/١٦٣) من حديث عبدالرحمن بن طرفة بن عرفجة عن جدِّه به.

قواه ابن المديني في علله (١٥٤)، وحسَّنه الترمذي، وصححه ابن حبان (٥٤٦٢)، وتكلَّم فيه ابن القطان في بيان الوهم والإيهام (٤/٦٠٩)، ولم نجده عند الحاكم، وذكر ابن حجر هذا الحديث في إتحاف المهرة (١١/١٥١) ولم يعزه إليه.

(١٢) كذا في النسخ المعتمدة لدينا: «حمزة» بالحاء والزاي، وهو كذلك في النسخ الأخرى التي بين أيدينا، وهو مخالف لما في كتب التراجم وصوابه: «أبي جمرة» بالجيم والراء، وهو: نصر بن عمران الضبعي [انظر: الإكمال، لابن ماكولا (٢/٥٠٦)].

(١٣) في (الأصل، د، ز، س، ض، ع، ل، م) بدون واو العطف قبل «ثابت»، كما أنه كذلك في أكثر النسخ التي بين أيدينا، وفي (ب) أُثبتت الواو، وهو الصواب.

(١٤) أما موسى بن طلحة: فأخرجه ابن أبي شيبة (٨/٤٩٨)، والطحاوي في مشكل الآثار (٤/٣٦).

وأما أبو جمرة الضبعي وأبو رافع: فأخرجه الطحاوي (٤/٣٧-٣٨) عنهما.

وأما ثابت البناني: فأخرجه ابن أبي شيبة (٨/٤٩٩)، وأبو زرعة الدمشقي في تاريخه (ص٦٢٤).

وأما إسماعيل بن زيد بن ثابت: فلم نقف عليه مسندًا، وأخرج أبو زرعة الدمشقي في تاريخه (ص٦٢٤) أن إسماعيل بن معدي كرب يضبب ثنيته بالذهب.

وأما المغيرة بن عبدالله: فأخرجه عبدالله بن الإمام أحمد المسند (٥/٢٣)، وابن أبي شيبة (٨/٤٩٨)، والطحاوي في مشكل الآثار (٤/٣٦).

(١٥) أخرجه أحمد (٤/٣٩٤)، والترمذي (١٧٢٠)، والنسائي (٨/١٦١).

صححه الترمذي، وأعله ابن حبان في صحيحه (٥٤٣٤)، والدارقطني في العلل (١٣٢٠).

(١٦) في (ب، د، ز،ض، ع، ل، م): «الطبراني».

(١٧) أخرجه ابن الجوزي في التحقيق (٣/٦٦ مع التنقيح).

قال البيهقي في معرفة السنن والآثار (٦/١٤٣): (مرفوعًا .. لا أصل له، إنما يروى عن جابر من قوله)، ورجَّحه ابن عبدالهادي في التنقيح (٣/٦٧ مع التحقيق).

(١٨) أخرجه أبو عبيد في الأموال (١١٩١)، والدارقطني (١٩٦٥)، والبيهقي (٤/١٣٨).

(١٩) أخرجه عبدالرزاق (٤/٨٢)، وأبو عبيد في الأموال (١١٨٩)، والبيهقي (٤/١٣٨). وأخرجه عبدالرزاق (٤/٨٢)، وابن أبي شيبة (٣/١٥٥) بنحوه.

(٢٠) أخرجه عبدالرزاق (٤/٨٢)، وابن أبي شيبة (٣/١٥٤)، والبيهقي (٤/١٣٨). وأخرجه مالك في الموطأ (٦٧٤)، وأبو عبيد في الأموال (١١٩٠) بنحوه.

(٢١) أخرجه مالك في الموطأ (٦٧٣)، ومن طريقه البيهقي (٤/١٣٨)، وأخرجه عبدالرزاق (٤/٨٣)، وابن أبي شيبة (٣/١٥٥).

(٢٢) أخرجه ابن أبي شيبة (٣/١٥٥)، والدارقطني (١٩٦٩)، والبيهقي (٤/١٣٨).
