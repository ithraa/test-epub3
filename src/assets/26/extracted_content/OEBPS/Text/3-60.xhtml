١١٩٧

(كتابُ النِّكاحِ)

هوَ لغةً: الوطءُ، والجمعُ بينَ الشّيئَيْنِ، وقدْ يطلقُ علَى العَقْدِ.

  • وإذَا قالُوا: «نكحَ فُلانَةً، أوْ بنْتَ فُلانٍ» أرادُوا تزوَّجَهَا وعَقَدَ عَلَيْهَا.
  • وإذَا قالُوا: «نكحَ امرأتَهُ» لمْ يريدُوا إلَّا المُجامعَةَ.

وشرعًا: عَقْدٌ يُعتبَرُ فيهِ لفظُ إنكاحٍ أوْ تزويجٍ فِي الجملَةِ.

والمعقودُ عليهِ منفعَةُ الاستمتاعِ.

و(هوَ سُنَّةٌ) لذِي شهوةٍ لَا يخافُ زنًا، منْ رجلٍ وامرأةٍ؛ لقولِهِ چ : «يَا معشرَ الشّبابِ، مَنِ استطاعَ منكُمُ الباءَةَ فليتزوّجْ، فإنَّهُ أغضُّ للبصرِ وأحصنُ للفرجِ، ومَنْ لمْ يستطعْ فعليهِ بالصّومِ فإنَّهُ لهُ وجاءٌ»، رواهُ الجماعةُ(١).

ويُباحُ لمَنْ لَا شهوَةَ لهُ؛ كالعِنِّينِ والكبيرِ.

(ونفْلُهُ(٢) معَ الشَّهوَةِ أفضلُ منْ نوافلِ العبادَةِ)؛ لاشتمالِهِ علَى مصالحَ كثيرةٍ؛ كتحصينِ فَرْجِهِ وفَرْجِ زوجتِهِ، والقيامِ بِهَا، وتحصيلِ النّسلِ، وتكثيرِ

١١٩٨

الأمَّةِ، وتحقيقِ مُباهَاةِ النَّبِيِّ چ (٣)، وغيرِ ذلكَ.

  • ومَنْ لَا شهوَةَ لهُ نَوافِلُ العبادَةِ أفضلُ لهُ.

(ويجبُ) النِّكاحُ (علَى مَنْ يخافُ زنًا بتركِهِ)، ولوْ ظنًّا، منْ رجلٍ وامرأةٍ؛ لأنَّهُ طريقُ إعفافِ نفسِهِ وصَوْنِهَا عنِ الحرامِ.

  • ولَا فرقَ بينَ القادرِ علَى الإنفاقِ والعاجزِ عنهُ،
  • ولَا يكتفِي بمرَّةٍ، بلْ يكونُ فِي مجموعِ العمرِ.

ويحرُمُ بدارِ حَرْبٍ،

  • إلَّا لضرورةٍ فيُباحُ لغيرِ أسيرٍ.

(ويُسَنُّ نكاحُ واحدةٍ)؛ لأنَّ الزِّيادَةَ عَلَيْهَا تعريضٌ للمُحَرَّمِ، قالَ تعالَى: {وَلَن تَسۡتَطِيعُوٓاْ أَن تَعۡدِلُواْ بَيۡنَ ٱلنِّسَآءِ وَلَوۡ حَرَصۡتُمۡ} [النساء:١٢٩]،

١١٩٩
  • (دَيِّنَةٍ)؛ لحديثِ أبِي هريرةَ ڔ مرفوعًا: «تُنكَحُ المرأَةُ لأربعٍ: لمالِهَا، ولحَسَبِهَا، ولجمالِهَا، ولِدِينِهَا، فاظفرْ بذاتِ الدِّينِ تَرِبَتْ يداكَ»، متَّفقٌ عليهِ(٤).
  • (أجنبيَّةٍ)؛
  • لأنَّ ولدَهَا يكونُ أنْجَبَ،
  • ولأنَّهُ لَا يَأمَنُ الطّلاقَ فيُفضِي -معَ القرابَةِ- إلَى قطيعَةِ الرَّحِمِ.
  • (بِكْرٍ)؛ لقولِهِ چ لجابرٍ: «فهلَّا بِكْرًا تُلاعبُهَا وتُلاعِبُكَ»، متَّفقٌ عليهِ(٥).
  • (ولودٍ)؛ أيْ: منْ نساءٍ يُعْرَفْنَ بكثرَةِ الأولادِ؛ لحديثِ أنسٍ ڔ يرفعُهُ: «تزوّجُوا الودُودَ الوَلُودَ، فإنِّي مُكاثِرٌ بكُمُ الأممَ يومَ القيامَةِ»، رواهُ سعيدٌ(٦).
  • (بلَا أمٍّ)؛ لأنَّهَا ربّمَا أفسدتْهَا عليهِ.
  • ويُسنُّ أنْ يتخيّرَ الجميلَةَ؛ لأنَّهُ أغضُّ لبصَرِهِ.

ڭ ڭ ڭ

١٢٠٠

(و) يُباحُ (لهُ)؛ أيْ:

  • لمَنْ أرادَ خِطبَةَ امرأةٍ،
  • وغلبَ علَى ظنِّهِ إجابتُهُ:
  • (نَظَرُ مَا يظهرُ غالبًا)؛ كوجهٍ، ورقَبَةٍ، ويدٍ، وقَدَمٍ؛ لقولِهِ چ : «إذَا خطبَ أحدُكُمُ امرأةً فقدرَ أنْ يرَى مِنْهَا بعضَ مَا يدعُوهُ إلَى نكاحِهَا فليفعلْ»، رواهُ أحمدُ وأبُو داودَ(٧)،
  • (مرارًا)؛ أيْ: يكرِّرُ النّظرَ،
  • (بلَا خلوةٍ)،
  • إنْ أَمِنَ ثَوَرانَ الشَّهوَةِ.

ولَا يحتاجُ إلَى إذنِهَا.

ويُباحُ نَظَرُ ذلكَ ورأسٍ وساقٍ مِنْ أمَةٍ وذاتِ مَحْرَمٍ،

  • ولعبدٍ نظرُ ذلكَ منْ مَوْلاتِهِ،
  • ولشاهدٍ، ومُعَامِلٍ، نَظَرُ وجهِ مشهودٍ عَلَيْهَا ومَنْ تُعَامِلُهُ وكفَّيْهَا لحاجةٍ.
  • ولطبيبٍ ونحوِهِ نَظَرٌ ولمسٌ دعَتْ(٨) إليهِ حاجةٌ.
  • ولامرأةٍ نَظَرٌ مِنَ امرأةٍ ورَجُلٍ إلَى مَا عدَا مَا بينَ سُرَّةٍ وركبةٍ.

ويحرمُ خَلْوَةُ ذَكَرٍ غيرِ مَحْرَمٍ بامرأَةٍ.

(ويحرُمُ التَّصريحُ بخِطبَةِ المُعتدَّةِ)؛ كقولِهِ: أريدُ أنْ أتزوَّجَكِ؛ لمفهومِ قولِهِ تعالَى: {لَا(٩) جُنَاحَ عَلَيۡكُمۡ فِيمَا عَرَّضۡتُم بِهِۦ مِنۡ خِطۡبَةِ ٱلنِّسَآءِ} [البقرة:٢٣٥]. وسواءٌ:

١٢٠١
  • المُعتدَّةُ(١٠) (منْ وفاةٍ،
  • والمُبانَةُ) حالَ الحيَاةِ.
  • (دونَ التّعرِيضِ)، فيُباحُ؛ لمَا تقدَّمَ.

ويحرُمُ التّعريضُ كالتّصريحِ لرجعيَّةٍ.

(ويباحانِ لمَنْ أبَانَهَا بدونِ الثّلاثةِ(١١))؛ لأنَّهُ يُباحُ لهُ نكاحُهَا فِي عِدّتِهَا، (كرجعيَّةٍ(١٢))؛ فإنَّ لهُ رجعتَهَا فِي عِدّتِهَا.

(ويحرُمانِ)؛ أيْ: التّصريحُ والتّعريضُ (مِنْهَا علَى غيرِ(١٣)زوجِهَا)، فيحرمُ علَى الرّجعيَّةِ أنْ تُجيبَ مَنْ خطبَهَا فِي عدّتِهَا تصريحًا أوْ تعريضًا.

  • وأمَّا البائنُ فيُباحُ لهَا إذَا خُطبَتْ فِي عدّتِهَا التّعريضُ دونَ التّصريحِ.

(والتّعريضُ: إنِّي فِي مِثْلِكِ لراغبٌ، وتجيبُهُ) إذَا كانتْ بائنًا: (مَا يُرغبُ عنْكَ، ونحوُهُمَا)؛ كقولِهِ: لَا تُفَوِّتِينِي بنفسِكِ، وقولِهَا: إنْ قُضِيَ شيءٌ كانَ.

(فإنْ أجابَ وليُّ مُجْبَرَةٍ) ولوْ تعريضًا لمسلمٍ، (أوْ أجابَتْ غيرُ المُجْبَرَةِ لمسلمٍ: حَرُمَ علَى غيرِهِ خِطْبَتُهَا) بلَا إذنِهِ؛ لحديثِ أبِي هريرةَ ڔ مرفوعًا:

١٢٠٢

«لَا يخطِبُ الرّجلُ علَى خِطبَةِ أخيهِ حتَّى ينكحَ أوْ يترُكَ»، رواهُ البخارِيُّ والنَّسائِيُّ(١٤).

  • (وإنْ رُدَّ) الخاطبُ الأوّلُ،
  • (أوْ أَذِنَ)،
  • أوْ تركَ،
  • أوِ استأذنَ الثَّانِي الأوَّلَ فسكتَ،
  • (أوْ جَهِلَ(١٥) الحالَ)؛ بأنْ لمْ يعلمِ الثّانِي إجابَةَ الأوّلِ:
  • (جازَ) للثّانِي أنْ يخطِبَ.

(ويُسَنُّ العقدُ:

  • يومَ الجمعَةِ مساءً)؛ لأنَّ فيهِ ساعَةَ الإجابَةِ.
  • ويُسَنُّ بالمسجدِ، ذكرَهُ ابنُ القيِّمِ(١٦).
  • ويُسَنُّ أنْ يَخْطُبَ قبلَهُ (بخُطبَةِ ابنِ مسعودٍ ڔ ) وهيَ: إنَّ الحمدَ للهِ نحمدُهُ، ونستعينُهُ، ونستغفرُهُ، ونتوبُ إليهِ، ونعوذُ باللهِ منْ شرورِ أنفسِنَا وسيّئاتِ أعمالِنَا، مَنْ يهدِ الله فلَا مُضلَّ لهُ، ومَنْ يُضلِلْ فلَا هادِيَ لهُ، وأشهدُ(١٧) أنْ لَا إلهَ إلَّا الله وأشهدُ
١٢٠٣
  • أنَّ مُحمَّدًا عبدُهُ ورسولُهُ(١٨).

ويُسَنُّ أنْ يُقالَ لمتزوِّجٍ: «باركَ الله لكمَا وعليكُمَا، وجمعَ بينكمَا فِي خيرٍ وعافيةٍ»(١٩).

  • فإذَا زُفَّتْ إليهِ قالَ: «اللَّهُمَّ إنِّي أسألُكَ خيرَهَا وخيرَ مَا جَبَلْتَهَا عليهِ؛ وأعوذُ بِكَ منْ شرِّهَا وشرِّ مَا جَبَلْتَهَا عليهِ»(٢٠).

ڭ ڭ ڭ

(١) أخرجه أحمد (١/٣٧٨)، والبخاري (١٩٠٥)، ومسلم (١٤٠٠)، وأبو داود (٢٠٤٦)، والترمذي (١٠٨١)، وابن ماجه (١٨٤٥)، والنسائي (٦/٥٧) من حديث عبد الله بن مسعود ڔ .

(٢) في (ب، ض، ل، م): «وفعله»، وفي (ز، ع) كانت كالمثبت ثم عُدِّلت إلى: «وفعله»، والمثبت من (الأصل، د، س) وكَتَب في هامش (س): (الذي في النسخ: «وفعله» بدل: «ونفله»)، وقد اختلفت نسخ الزاد على نحو هذا الاختلاف في نسخ الروض، إلا أن مما يرجِّحُ المثبتَ موافقتُه لعبارةِ صاحب الوجيز حيث قال (ص٣٣١): (وهو سنة، ونَفْلُهُ مع الشهوة أفضل من نَفْلِ العبادة)، ويؤيده أيضًا قول صاحب الفروع (٨/١٧٨): (ونَفْلُهُ مقدَّمٌ على نَفْلِ العبادة على الأصح)، إلا أن صاحب الفروع استغنى بقوله: «ونفله» عن الإشارة إلى قيد: (مع الشهوة) الذي هو مناط استحباب النكاح، بخلاف عبارة الحجاوي.

ومما يؤيِّدُ النسخَ الأخرى عبارةُ المصنف في الإقناع حيث قال (٣/٢٩٥): (يسن لمن له شهوة ... واشتغالُهُ به أفضلُ من التخلِّي لنوافل العبادة)، ونحوه عبارة المقنع (٢٠/١٣) والمنتهى (٤/٤٩)؛ فتعبيرهم بـ: «واشتغاله» موافق للتعبير بـ: «وفِعْله»، فالله أعلم.

(٣) أخرج أبو داود (٢٠٥٠)، والنسائي (٦/٦٥ - ٦٦) من حديث معقل بن يسار ڔ مرفوعًا: «تزوجوا الودود الولود، فإني مكاثر بكم الأمم»، صححه ابن حبان (٤٠٥٦).

(٤) أخرجه أحمد (٢/٤٢٨)، والبخاري (٥٠٩٠)، ومسلم (١٤٦٦).

(٥) أخرجه أحمد (٣/٣١٤)، والبخاري (٢٣٠٩)، ومسلم (٧١٥).

(٦) أخرجه أحمد (٣/١٥٨)، وسعيد بن منصور في سننه (٤٩٠) بلفظ «الأنبياء» بدل «الأمم»، صححه ابن حبان (٤٠٢٨)، وتقدم قريبًا بلفظ المؤلف من حديث معقل بن يسار ڔ (ص١١٩٨).

(٧) أخرجه أحمد (٣/٣٣٤)، وأبو داود (٢٠٨٢) من حديث جابر بن عبدالله ژ .

وصححه الحاكم (٢/١٦٥)، وقوَّاه الحافظ ابن عبدالهادي في المحرر (١٠٠٨)، وقال ابن حجر في فتح الباري (٩/٨٧): (إسناده حسن وله شاهد من حديث محمد بن مسلمة)، وانظر: نصب الراية (٤/٢٤١ - ٢٤٢)، وضعَّفه ابن القطان في بيان الوهم والإيهام (٤/٤٢٨).

(٨) في (ب، ع، ل): «ولمس ما دعت».

(٩) في (ب، ض): «ولا»، وكثيرًا ما يحذف العلماء ڤ حروف العطف عند ذكرهم الآية للاستدلال والاستشهاد، وانظر تعليقنا على روضة الناظر (٢/ ٣٨١ ط. الثالثة).

(١٠) في (د، ز، ض، ل، م): «كانت المعتدة».

(١١) في (ب، د، ز، س، م): «الثلاث»، والمثبت هو الموافق لما في زاد المستقنع (ص٥٩٢ت: القاسم).

(١٢) في (ز، م): «كرجعيته»، وفي (الأصل) كتبت كذلك ثم عدلت إلى «كرجعية» فيما يظهر لنا، والمثبت من (ب، د، س، ض، ع، ل).

(١٣) في (الأصل، س): من الشرح.

(١٤) أخرجه البخاري (٥١٤٤)، والنسائي (٦/٧٣).

(١٥) في (ب، د، ز، ض، ع، ل، م): «جهلت»، والمثبت هو الموافق لما في زاد المستقنع (ص٦٩٢ت: القاسم).

(١٦) انظر: إعلام الموقعين (٤/٥٤٢).

(١٧) في (ز، ع، ل): «أشهد».

(١٨) أخرجه أحمد (١/٣٩٢)، وأبو داود (٢١١٨)، والترمذي (١١٠٥)، وابن ماجه (١٨٩٢)، والنسائي (٦/٨٩).

قال الترمذي: (حديث حسن)، وصححه أبو عوانة في مستخرجه (٤١٤٣)، وابن الملقن في البدر المنير (٧/٥٣١).

(١٩) أخرجه أحمد (٢/٣٨١)، وأبو داود (٢١٣٠)، والترمذي (١٠٩١)، وابن ماجه (١٩٠٥) من حديث أبي هريرة ڔ .

صححه الترمذي، وابن الملقن في البدر المنير (٧/٥٣٤)، وقال ابن عبد الهادي في حاشية الإلمام (٩٦٦): (إسناده على شرط مسلم).

(٢٠) أخرجه أبو داود (٢١٦٠)، وابن ماجه (١٩١٨) من حديث عمرو بن شعيب عن أبيه عن جده مرفوعًا.

قال ابن عبد الهادي في حاشية الإلمام (١٠١٧): (وهو صحيح الإسناد إلى عمرو).
